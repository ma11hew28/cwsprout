  6 require('views/view') ;
  7 
  8 /**
  9   @class
 10   
 11   This is a basic popup menu. You can show the popup menu by calling the 
 12   view's popup() method.  Pass in the root element and event so that is can
 13   be positioned.
 14   
 15   @extends SC.View
 16 */
 17 SC.PopupMenuView = SC.View.extend({
 18   emptyElement: "<ul></ul>",
 19   
 20   acceptsFirstResponder: true,
 21 
 22 
 23   keyDown: function(evt)
 24   {
 25     return this.interpretKeyEvents(evt);
 26   },
 27 
 28   // interpretKeyEvents callbacks...
 29   cancel: function()
 30   {
 31     this.set('currentSelectedMenuItem', null);
 32     this.set('isVisible', false);
 33   },
 34   moveUp: function()
 35   {
 36     this.selectPreviousMenuItem();
 37   },
 38   moveDown: function()
 39   {
 40     this.selectNextMenuItem();
 41   },
 42   
 43   _currentSelectedMenuItem: null,
 44   currentSelectedMenuItem: function( key, value )
 45   {
 46     if (value !== undefined)
 47     {
 48       if (this._currentSelectedMenuItem) this._currentSelectedMenuItem.set('isDefault', false);
 49       this._currentSelectedMenuItem = value;
 50       if (this._currentSelectedMenuItem) this._currentSelectedMenuItem.set('isDefault', true);
 51     }
 52     return this._currentSelectedMenuItem;
 53   }.property(),
 54   
 55   selectPreviousMenuItem: function()
 56   {
 57     var item = this.previousValidMenuItem();
 58     if (!item) return false;
 59     this.set('currentSelectedMenuItem', item);
 60   },
 61   
 62   selectNextMenuItem: function()
 63   {
 64     var item = this.nextValidMenuItem();
 65     if (!item) return false;
 66     this.set('currentSelectedMenuItem', item);
 67   },
 68   
 69   previousValidMenuItem: function()
 70   {
 71     return this._validMenuItemInDirection('previousSibling', 'lastChild');
 72   },
 73   nextValidMenuItem: function()
 74   {
 75     return this._validMenuItemInDirection('nextSibling', 'firstChild');
 76   },
 77   _validMenuItemInDirection: function( direction, begin )
 78   {
 79     var curr = this.get('currentSelectedMenuItem');
 80     var view = curr ? curr.get(direction) : this.get(begin);
 81     if (!view) return null;
 82     do {
 83       if (view.get('isEnabled')) return view;
 84     } while (view = view.get(direction));
 85     return null;
 86   },
 87   
 88   
 89   _show: function()
 90   {
 91     // clear out any previously used selection value...
 92     this.set('currentSelectedMenuItem', null);
 93 
 94     var menuItems = this.get('childNodes');
 95     for (var i=0, n=menuItems.length; i < n; i++)
 96     {
 97       // for each menu item, get the action that it performs and the current target for that action
 98       var sender = menuItems[i];
 99 
100       // old function style actions... skip validating and let them be
101       if (sender._hasLegacyActionHandler()) continue;
102 
103       var action = sender.get('action');
104       var target = SC.app.targetForAction(action, sender.get('target'), sender);
105       
106       //console.log( "action: %s, target: %o", action, target );
107       
108       // found a target and the target validates, menu item is enabled if the validator returns true
109       if (target && target.respondsTo('validateMenuItem')) sender.set('isEnabled', target.validateMenuItem(sender));
110       // found a target and it chooses not to validate, menu item is enabled
111       if (target && !target.respondsTo('validateMenuItem')) sender.set('isEnabled', true);
112       // unable to resolve a target for this action, therefor disable the menu item
113       if (!target) sender.set('isEnabled', false);
114     }
115     // call the superclasses _show method to make the menu visible
116     sc_super();
117   },
118   
119   resizeWithOldParentSize: function() {
120     
121     // loop through child views (which should be menu items), and get their
122     // required widths.  Use the maximum returned width.
123     var requiredWidth = 0;
124     var child = this.get('firstChild') ;
125     while(child) {
126       var w = (child.computedRequiredWidth) ? child.computedRequiredWidth():0;
127       if (w > requiredWidth) requiredWidth = w ;
128       child = child.get('nextSibling') ;
129     }
130    var oldWidth = this.get("size").width;
131    if(requiredWidth != oldWidth)
132    {
133      var size = { width: requiredWidth };
134      // set this view to the same size.
135      this.set('size',size) ;
136    }
137   }
138   
139 });
