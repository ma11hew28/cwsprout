 10 /**
 11   FormView provides a simple way for you to focus a group of views onto a 
 12   single content object or set of content objects.  It can also be used to
 13   "stage" input by buffering data from your views into the form before it is 
 14   set on your actual content object.  This approach is generally deprecated 
 15   however as the Controller objects now provide this same buffering feature
 16   in a more robust way.
 17 
 18   h2. How It Works
 19 
 20   On startup, FormView will walk any child views looking for any views 
 21   with the property "fieldKey".  Any views with this key will be saved as
 22   fields on the form.  You can access their value directly on the form by
 23   getting the value of fieldKey.
 24 
 25   Used this way, you can easily combine multiple views into a single,
 26   bindable object.  To gain the full functionality however, you should use
 27   the content commit capability.
 28 
 29   IMPORTANT:  FormView defines some properties of its own.  If you try to
 30   name your fields with properties already declared in FormView, they will
 31   be ignored and a warning will be logged to the console.
 32 
 33   h2. Content Commit
 34 
 35   If you set the content object on a form view, then the form fields will
 36   automatically be bound to the same named keys on the content object.  When
 37   the content object value's change, the fields on the form will update
 38   automatically.  Likewise, if you change the value of the field, the value
 39   of the content object will generally be changed as wel.
 40   
 41   If you prefer, you can also turn on buffered commits by setting the 
 42   commitChangesImmediately property to NO.  This will cause the form to 
 43   buffer any changes to the field values and only forwards them to the content
 44   when you call commitChanges() on the form.  This method of buffering is
 45   still supported for some simple cases, however it is better to use the newer
 46   Controller objects that provide this same buffering in a more robust way.
 47 
 48 */
 49 SC.FormView = SC.View.extend({
 50 
 51   // PROPERTIES
 52   
 53   // set this to point to an object and the value of the object will be used
 54   // to auto-populate the form.  When the form is committed, its values will
 55   // applied to this content object.
 56   content: null,
 57   contentBindingDefault: SC.Binding.Single,
 58 
 59   // this is set to true when the values of the field have changed since the
 60   // last commit or reset.
 61   isDirty: false,
 62 
 63   // this is set to true while a form is in the process of committing changes
 64   // from the form.  This is useful when your content is an object controller
 65   // that actually commits the form to the server.
 66   isCommitting: true,
 67   
 68   // set to false to disable form input.  this will set the isEnabled property
 69   // of all fields to false.
 70   isEnabled: true,
 71   
 72   passThroughToContent: false,
 73   
 74   // computed property returns true if you have no errors.
 75   isValid: function() {
 76     return this.get('errors').length == 0;
 77   }.property('errors'),
 78   
 79   // RO - computed property returns true if the form can current be committed.
 80   // The default version returns true if the form is enabled, valid, and 
 81   // dirty.  You can override this with your own changes if you prefer some
 82   // other behavior.
 83   canCommit: function() {
 84     return this.get('isValid') && this.get('isEnabled');
 85   }.property('isValid','isEnabled'),
 86   
 87   // Set this to an error or array of errors to be included in the
 88   // overall errors property. 
 89   generalErrors: null,
 90   
 91   // RO - this property returns any current errors on the form.  Note that if
 92   // you are implementing your own view, you should set your own value to
 93   // an insteand of SC.FieldError and it will be displayed.
 94   errors: function() {
 95     if (!this._fields) return [] ;
 96     
 97     // compute cached errors.
 98     if (!this._errors) {
 99       var fview =this ;
100       this._errors = [] ;
101       this.get('fieldKeys').each(function(k) {
102         var value = fview.get(k) ;
103         if ($type(value) == T_ERROR) fview._errors.push(value) ;
104       }) ;
105     }
106     
107     // return set of errors.
108     return this._errors.concat(this.get('generalErrors') || []) ;
109   }.property('generalErrors'),
110 
111   
112   fieldKeys: function() {
113     if (!this._fieldKeys && this._fields) {
114       var keys = [];
115       for(var key in this._fields) {
116         if (!this._fields.hasOwnProperty(key)) continue ;
117         keys.push(key) ;
118       }
119       this._fieldKeys = keys ;
120     } 
121     return this._fieldKeys ;
122   }.property(),
123   
124   // SUPPORT METHODS  
125   
126   // Call this method to perform a full validation on the form fields.
127   // Returns true if the form is valid, false if it is not.  If validation
128   // fails, the errors property will be set to the approrpiate value.
129   validate: function() {
130     if (!this._fields) return true; // ok if now fields.
131     
132     // validate all fields.
133     for(var key in this._fields) {
134       if (this._fields.hasOwnProperty(key)) {
135         var field = this._fields[key] ;
136         if (field.validateSubmit) field.validateSubmit() ;
137       }
138     } 
139     
140     // check for errors
141     return this.get('isValid') ;
142   },
143   
144   // This action can be called by a button to commit change to the form.
145   commit: function() {
146 
147 
148     // validate the form.  Return false if validation fails.
149     if (!this.validate()) return false ;
150 
151     var ret = true ;
152     var content = this.get('content') ;
153     if (!content || !this._fields) return ;
154     
155     // disable form during commit.
156     var wasEnabled = this.get('isEnabled') ;
157     this.beginPropertyChanges() ;
158     this.set('isEnabled', false) ;
159     this.set('isCommitting',true) ;
160     this.endPropertyChanges() ;
161     
162     ret = this.get('passThroughToContent') ? this._commitChanges() : this._copyContentAndCommitChanges();
163 
164     // clean up property settings.
165     this.beginPropertyChanges() ;
166     this.set('isCommitting',false) ;
167     this.set('isEnabled',wasEnabled) ;
168     this.endPropertyChanges() ;
169     
170     return ret ;
171   },
172 
173   _copyContentAndCommitChanges: function()
174   {
175     var ret = true ;
176     var content = this.get('content');
177     if (!content || !this._fields) return false;
178 
179     // copy all the properties back to the content object.
180     // if the content object throws an error for some reason, catch it
181     // and log it.  Also add it to the list of errors.
182     try {
183       content.beginPropertyChanges();
184       for (var key in this._fields)
185       {
186         if (key.match(/Button$/)) continue; // ignore buttons.
187         
188         if (this._fields.hasOwnProperty(key)) {
189           var newValue = this.get(key);
190           content.set(key,newValue);
191         }
192       }
193       content.endPropertyChanges();
194 
195       // attempt to save changes...
196       ret = this._commitChanges();
197       
198       // once a commit is complete, set isDirty to false.  If the commit
199       // fails or an exception occurs, then don't set to false.
200       this.set('isDirty', !ret) ;
201     }
202     catch(e) {
203       console.log("commit() exception: " + e) ;
204       ret = false ;
205     }
206     
207     return ret;
208   },
209   _commitChanges: function()
210   {
211     var content = this.get('content');
212     var success = false;
213     
214     // If the content object supports a commit method, call it so it can
215     // commit changes to the server.
216     if (content && content.commit) {
217       success = content.commit(this);
218     } else if (content && content.commitChanges) {
219       success = content.commitChanges();
220     }
221     
222     return success;
223   },
224   
225   
226   // This action will reset the form, copying the current values from the
227   // content object onto the field values.
228   reset: function()
229   {
230     if (!this._fields) return; // EXIT POINT
231     
232     var content = this.get('content');
233     
234     if (content && content.discardChanges) content.discardChanges();
235 
236     this.beginPropertyChanges();
237     for(var key in this._fields) {
238       if (this._fields.hasOwnProperty(key)) {
239         var value = (content) ? content.get(key) : null;
240         this.set(key, value);
241       }
242     }
243     this.set('isDirty',false);
244     this.endPropertyChanges();
245     
246     //if (content && content.discardChanges) content.discardChanges();
247   },
248   
249   // This method will crawl through its child views looking for any view
250   // with the fieldKey property set.  This does not go inside of other
251   // FormViews.
252   rebuildFields: function() {
253     this.beginPropertyChanges(); 
254     
255     // if fields are already registered, remove them.  Do it this way so
256     // that we can remove observer from the target object as well.
257     if (this._fields) {
258       for (var key in this._fields) {
259         if (this._fields.hasOwnProperty(key)) this.removeField(key) ;
260       }
261     }
262     
263     // reset the fields hash.
264     this._fields = {} ;
265     this._buttons = {} ;
266     this._values = {} ;
267     
268     // now rebuild field nodes for children.
269     this._rebuildFieldsForNode(this, true) ;
270     this.endPropertyChanges() ;
271   },
272   
273   // You can add a field manually by calling this method. 
274   // key - the key to respond to.  
275   // field - the view to map to the key.  This should be a child view of the
276   //         form, but it is not required.
277   addField: function(key, field) {
278     
279     // if the key is already defined on the form view, then we can't use it
280     // as a field.  Throw an exception.
281     if (this[key] !== undefined) {
282       throw "FormView cannot add the field '%@' because that property already exists.  Try using another name.".fmt(key);
283     }
284     
285     // if this field is a submitButton or resetButton and the actio is
286     // not set, set it...
287     var form = this ;
288     if (key == 'submitButton' && (field.action == SC.ButtonView.prototype.action)) {
289       field.action = function() { form.commit(); } ;
290     }
291     
292     if (key =="resetButton" && (field.action == SC.ButtonView.prototype.action)) {
293       field.action = function() { form.reset(); } ;    
294     }
295     
296     // save this field in the key.
297     this._fields[key] = field ;
298     if (key.substr(-6,6) == "Button") {
299       this._buttons[key] = field ; 
300     };
301     
302     // also add property of field to cache and notify of change.
303     this.propertyWillChange(key) ;
304     this.setValueForField(key, field.get('value'));
305     this.propertyDidChange(key,this.getValueForField(key)) ;    
306     
307     // and add us as an observer.
308     field.addObserver('value', this._fieldValueObserver_b()) ;
309     field.set('ownerForm',this) ;
310     
311     this.propertyWillChange('fieldKeys') ;
312     this._fieldKeys = null ;
313     this.propertyDidChange('fieldKeys', null) ;
314   },
315   
316   // This will remove the field with the named key from the list of fields.
317   removeField: function(key) {
318     // first remove the form as an observer to this field.
319     var field = this._fields[key] ;
320     if (field) {
321       field.removeObserver('value', this._fieldValueObserver_b());
322       field.set('ownerForm',null) ;
323     }
324     
325     // now delete the field from our hash and cache and notify.
326     this.propertyWillChange(key) ;
327     delete this._fields[key] ;  
328     delete this._values[key] ;
329     delete this._buttons[key] ;
330     this.propertyDidChange(key, null) ;
331 
332     
333     this.propertyWillChange('fieldKeys') ;
334     this._fieldKeys = null ;
335     this.propertyDidChange('fieldKeys', null) ;
336   },
337   
338   // public accessor for retrieving the field View object
339   getField: function(key) {
340     return this._fields[key];
341   },
342   
343   
344   // KEYBOARD SUPPORT METHODS
345   
346   // Process keyboard events...
347   keyDown: function(evt) {
348     return this.interpretKeyEvents(evt) ; // start bubbling key events...
349   },
350   keyUp: function() {},
351 
352   // Handle default button.
353   insertNewline: function(sender, evt) {
354     
355     // find the default button to use by scanning for the isDefault button.
356     var button = this._findDefaultButton(this) ;
357     
358     // if not isDefault-button was found, look for the submitButton:
359     if (!button && this._fields && this._fields.submitButton) {
360       button = this._fields.submitButton ;
361     }
362 
363     if (button && button.triggerAction) button.triggerAction(evt) ;
364     return true ;
365   },
366   
367   // search child views looking for the default button view.
368   _findDefaultButton: function(view) {
369     if (view.triggerAction && view.get('isDefault')) return view;
370     view = view.firstChild;
371     while(view) {
372       var ret = this._findDefaultButton(view) ;
373       if (ret) return ret ;
374       view = view.nextSibling ;
375     }   
376     
377     return null ;
378   },
379   
380   // INTERNAL SUPPORT METHODS
381   
382   // This is called anytime you try to get or set an unknown property.  When
383   // you get a property, this will look in the fields as well.
384   unknownProperty: function(key, value) {
385 
386     var field = (this._fields) ? this._fields[key] : null ;
387     
388     // setter
389     if (value !== undefined) {
390       if (field) {
391         
392         var oldValue = this.getValueForField(key);
393         
394         // save in our own cache first.  This way when we get notified of
395         // the new value by the field, we won't renotify everyone else.
396         this.setValueForField(key, value);
397         
398         // set the value on the field itself as well...
399         field.set('value',value) ;
400         
401         // notify errors if the newValue changed to or from an error.
402         var isOldError = $type(oldValue) == T_ERROR ;
403         var isNewError = $type(value) == T_ERROR ;
404 
405         if (isOldError != isNewError) {
406           this.propertyWillChange('errors') ;
407           this._errors = null ;
408           this.propertyDidChange('errors', null) ;
409         } 
410         
411         
412       // if this is not on a field, just do the normal thing if no field fnd.
413       } else this[key] = value ; 
414       
415     // getter
416     } else {
417       
418       // return the cached value if there is one.
419       if (field) {
420         if (this.getValueForField(key) === undefined) {
421           this.setValueForField(key, field.get('value'));
422         }
423         return this.getValueForField(key);
424       }
425       
426     }
427     
428     return value ;
429   },
430   
431   getValueForField: function( key )
432   {
433     if (this.get('passThroughToContent')) {
434       var content = this.get('content');
435       return (content && content.get) ? content.get(key) : undefined;
436     } else {
437       return this._values[key];
438     }
439   },
440   setValueForField: function( key, value )
441   {
442     if (this.get('passThroughToContent')) {
443       var content = this.get('content');
444       if (content && content.get && content.set && (content.get(key) !== value)) 
445       {
446         content.set(key, value);
447       }
448     } else {
449       this._values[key] = value;
450     }
451     return value;
452   },
453   
454   // When the form is first created, go find all the fields and save them.
455   init: function() {
456     sc_super() ;
457     
458     // disable the normal submission system so we can take over.
459     if (this.rootElement && this.rootElement.tagName.toLowerCase() == "form") {
460       this.rootElement.onsubmit = function() { return false; } ;
461       
462     }
463     this.rebuildFields() ;
464   },
465   
466   _rebuildFieldsForNode: function(node, _force) {
467     if (node.fieldKey) this.addField(node.fieldKey, node) ;
468 
469     // other form views may be fields in your current form, but we do not 
470     // examine the children of the other form views because the form views own
471     // those fields.
472     if ((_force != true) && (node instanceof SC.FormView)) return ;
473     
474     // examine children.
475     var nodes = (node.childNodesForFormField) ? node.childNodesForFormField() : node.get('childNodes');
476     var loc = nodes.length ;
477     while(--loc >= 0) {
478       node = nodes[loc] ;
479       this._rebuildFieldsForNode(node, false);
480     }
481   },
482   
483   // this observer is added to all field's value property.  This simply
484   // notifies any observers of the form that the field's property has changed.
485   _fieldValueObserver: function(field, key, value) { 
486     if (!(key = field.fieldKey)) return ; // only notifiy fields with keys...
487 
488     // return if the new value is the same as the old value.  This avoids
489     // infinite loops when the value is changed by our own setter.
490     var oldValue = this.getValueForField(key);
491     if (oldValue == value) return ;
492 
493     // value did change so...
494     
495     this.beginPropertyChanges() ;
496 
497     // notify changes to field.
498     this.propertyWillChange(key) ;
499     this.setValueForField(key, value); // save the changed value.
500     this.propertyDidChange(key, value) ;
501 
502     // notify errors if the newValue changed to or from an error.
503     var isOldError = $type(oldValue) == T_ERROR ;
504     var isNewError = $type(value) == T_ERROR ;
505     
506     if (isOldError != isNewError) {
507       this.propertyWillChange('errors') ;
508       this._errors = null ;
509       this.propertyDidChange('errors', null) ;
510     } 
511 
512     // make form dirty.
513     if (!this.get('isDirty')) this.set('isDirty',true) ;
514 
515     this.endPropertyChanges() ;
516   },
517 
518   // returns a bound observer function...
519   _fieldValueObserver_b: function() {
520     return this._bound_fieldValueObserver = (this._bound_fieldValueObserver || this._fieldValueObserver.bind(this)) ;
521   },
522 
523   // this observer gets called anytime any property changes on the content
524   // object, even those that are not mapped to fields on the form.  This
525   // code simply checks for a change and then updates.
526   _contentPropertyObserver: function(content, key, value) {
527     if (!this._fields || !content) return ;
528     var fields = this._fields ;
529     
530     // if the key changed is one we match, just update with the new value...
531     // NOTE: it is important to call didChangeFor() everytime this is called.
532     // otherwise this might not notify changes property.
533     if (fields[key] && content.didChangeFor(this,key)) {
534       this.set(key, value) ;
535       
536     // otherwise, if the key changes is "*", then check all fields.
537     } else if (key == "*") {
538       for(var key in fields) {
539         if (fields.hasOwnProperty(key) && content.didChangeFor(this,key)) {
540           this.set(key,content.get(key)) ;
541         } // if (fields)
542       } // for(var key)
543     } // else if...
544   },
545 
546   // returns the bound observer function...
547   _contentPropertyObserver_b: function() {
548     return this._bound_contentPropertyObserver = (this._bound_contentPropertyObserver || this._contentPropertyObserver.bind(this)) ;
549   },
550   
551   _isEnabledObserver: function() {
552     var fields = this._fields ;
553     if (!fields) return ;
554     var enabled = this.get('isEnabled') ;
555     var canCommit = this.get('canCommit') ;
556     
557     for(var key in fields) {
558       if (fields.hasOwnProperty(key)) {
559         var field = fields[key] ;
560         if (field.set) if (key == 'submitButton') {
561           field.set('isEnabled', canCommit) ;
562         } else field.set('isEnabled',enabled) ;
563       }
564     }      
565   }.observes('isEnabled'),
566   
567   // Automatically observe the content properties and add/remove form as 
568   // observer.
569   _contentObserver: function() {
570     var content = this.get('content') ;
571     if (content == this._content) return ; // bail if content is same.
572     
573     var func = this._contentPropertyObserver_b() ;
574     
575     // if there was an older content, remove our observer.
576     if (this._content) this._content.removeObserver('*',func) ;
577     
578     // now, add observer to new content
579     this._content = content ;
580     if (!content) return ; // EXIT POINT
581     content.addObserver('*', func) ; 
582 
583     // reset the form to the content values
584     this.reset() ;
585 
586   }.observes('content'),
587   
588   _canCommitObserver: function() {
589     var buttons = this._buttons ;
590     var canCommit = this.get('canCommit') ;
591     if (buttons && buttons.submitButton) {
592       var button = buttons.submitButton ;
593       if (button.set) button.set('isEnabled',canCommit) ;
594     }
595   }.observes('canCommit') 
596   
597 });
