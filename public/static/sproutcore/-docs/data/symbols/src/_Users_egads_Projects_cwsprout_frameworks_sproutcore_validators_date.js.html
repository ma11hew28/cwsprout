  6 require('validators/validator') ;

  8 /**
  9   Handle parsing and display of dates.
 10   
 11   @class
 12   @extends SC.Validator
 13   @author Charles Jolley
 14   @version 1.0
 15 */
 16 SC.Validator.Date = SC.Validator.extend(
 17 /** @scope SC.Validator.Date.prototype */ {
 18 
 19   /**
 20     The standard format you want the validator to convert dates to.
 21   */
 22   format: 'NNN d, yyyy h:mm:ss a',
 23   
 24   /**
 25     if we have a number, then convert to a date object.
 26   */
 27   fieldValueForObject: function(object, form, field) {
 28     var date ;
 29     if (typeof(object) == "number") {
 30       date = new Date(object) ;
 31     } else if (object instanceof Date) { date = object; }
 32       
 33     if (date) object = date.format(this.get('format')) ;
 34     
 35     return object ;
 36   },
 37 
 38   /**
 39     Try to pase value as a date. convert into a number, or return null if
 40     it could not be parsed.
 41   */
 42   objectForFieldValue: function(value, form, field) {
 43     if (value) {
 44       var date = Date.parseDate(value) ;
 45       value = (date) ? date.getTime() : null ;
 46     }
 47     return value ;
 48   }
 49     
 50 }) ;
