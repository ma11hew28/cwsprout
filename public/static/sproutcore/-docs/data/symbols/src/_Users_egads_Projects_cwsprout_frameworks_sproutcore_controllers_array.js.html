 11 /** @class
 12 
 13 An Array Controller provides a way to project the contents of an array
 14 out to a view.  You can use this object any place you might use an
 15 array.  Changes to the array will not propogate to the content array
 16 until you call commitChanges().
 17 
 18 @extends SC.Controller
 19 @extends SC.Array
 20 @extends SC.SelectionSupport
 21 @since SproutCore 1.0
 22 
 23 */
 24 SC.ArrayController = SC.Controller.extend(SC.Array, SC.SelectionSupport,
 25 /** @scope SC.ArrayController.prototype */
 26 {
 27   
 28   /**
 29     If YES the will return controllers for content objects.
 30     
 31     If you want to use an array controller to edit an array contents directly
 32     but you do not want to wrap the values of the array in controller objects
 33     then you should set this property to NO.
 34     
 35     @field
 36     @type {Boolean}
 37   */
 38   useControllersForContent: NO,
 39   
 40   /**
 41     Provides compatibility with CollectionControllers.
 42     @field
 43     @type {SC.ArrayController}
 44   */
 45   arrangedObjects: function() { return this; }.property('content'),
 46 
 47   /**
 48     The content array managed by this controller.  
 49   
 50     In general you can treat an instance of ArrayController as if it were 
 51     the array held in this property.  Any changes you make to the controller
 52     that are not specifically implemented in the controller will pass through
 53     to the Array.
 54   
 55     Also if you set commitsChangesImmediately to false, the controller will
 56     buffer changes against this.
 57 
 58     @property
 59     @type {Array}
 60   */
 61   content: null,
 62   contentBindingDefault: SC.Binding.Multiple,
 63 
 64   /**
 65     Set to true if the controller has any content, even an empty array.
 66   */
 67   hasContent: function() {
 68     return this.get('content') != null ;
 69   }.property('content'),
 70   
 71   /**
 72     Set to true if you want objects removed from the array to also be
 73     deleted.  This is a convenient way to manage lists of items owned
 74     by a parent record object.
 75     
 76     Note that even if this is set to NO, calling destroyObject() instead of
 77     removeObject() will still destroy the object in question as well as 
 78     removing it from the parent array.
 79     
 80     @field
 81     @type {Boolean}
 82   */
 83   destroyOnRemoval: NO,
 84 
 85   /**
 86     Defines the default class to use when creating new content. 
 87     
 88     This property should either contains a class or a string that resolves
 89     to a class that responds to the newRecord() method.
 90   
 91     @property
 92     @type {Class}
 93   */
 94   exampleContentObject: null,
 95   
 96   /**
 97     Creates a new record instance and adds it to the end of the current array.
 98 
 99     This method works just like insertNewObjectAt() but always appends.
100 
101     @param attributes {Hash} optional hash of attributes to pass to the new obejct.
102     @param objectType {Class} optional class of object to create.
103     @returns {Object} the newly created object (also added to the array)
104   */ 
105   newObject: function(attributes, objectType) {
106     return this.insertNewObjectAt(null, attributes, objectType) ;
107   },
108   
109   /**
110     Creates a new content object and inserts it at the passed index or appends
111     it at the end of the array if you pass null.
112 
113     This method takes an optional hash of attributes which will be set on
114     the new record.  You can also pass an optional objectType.  If you do 
115     not pass the objectType, you must instead set the exampleContentObject to 
116     the class of the object you want to use.  The object can be of any type 
117     but it must respond to the newRecord() method.
118     
119     Objects created using this method will be destroyed automatically if you
120     have set commitsChangesImmediately to false and call discardChanges().
121 
122     @param index {Number} the index to insert at or null to append.
123     @param attributes {Hash} optional hash of attributes to pass to the new obejct.
124     @param objectType {Class} optional class of object to create.
125     @returns {Object} the newly created object (also added to the array)
126   */
127   insertNewObjectAt: function(index, attributes, objectType) {
128     
129     // compute the objectType
130     if (!objectType) objectType = this.get('exampleContentObject') ;
131     if ($type(objectType) === T_STRING) {
132       objectType = SC.Object.objectForPropertyPath(objectType) ;
133     }
134     if (objectType == null) {
135       throw "Invalid object type was provided" ;
136     }
137     
138     if ($type(objectType.newObject) !== T_FUNCTION) {
139       throw "content object type does not support newRecord()" ;
140     }
141     
142     // Create a new object...
143     var obj = objectType.newObject(attributes) ;
144     if (!this._createdObjects) this._createdObjects = [] ;
145     this._createdObjects.push(obj) ; // save for discard...
146     
147     // Add to array.
148     if (index) {
149       this.insertAt(index, obj) ;
150     } else this.pushObject(obj) ;
151     
152     return obj ;
153   },
154   
155   /**
156     Watches changes to the content property updates the contentClone.
157     @private
158     @observes content
159   */
160   _contentObserver: function() {
161     var content = this.get('content') ;
162     if (SC.isEqual(content, this._content)) return ; // nothing to do
163 
164     if (!this._boundContentPropertyObserver) {
165       this._boundContentPropertyObserver = this._contentPropertyObserver.bind(this) ;
166     }
167     var func = this._boundContentPropertyObserver ;
168 
169     // remove old observer, add new observer, and trigger content property change
170     if (this._content && this._content.removeObserver) this._content.removeObserver('[]', func) ;
171     if (content && content.addObserver) content.addObserver('[]', func) ;
172     this._content = content; //cache
173     this._contentPropertyRevision = null ;
174     
175     var rev = (content) ? content.propertyRevision : -1 ;
176     this._contentPropertyObserver(this, '[]', content, rev) ; 
177   }.observes('content'),
178 
179   _contentPropertyObserver: function(target, key, value, rev) {  
180       
181     if (!this._updatingContent && (!rev || (rev != this._contentPropertyRevision))) {
182       this._contentPropertyRevision = rev ;
183 
184       this._updatingContent = true ;
185 
186       this.beginPropertyChanges();
187       this.contentCloneReset();
188       this.enumerableContentDidChange() ;
189       this.notifyPropertyChange('length') ;
190       this.updateSelectionAfterContentChange();
191       this.endPropertyChanges() ;
192 
193       this._updatingContent = false ;
194 
195     }
196   },
197   
198   /**
199     The array content that (when committed) will be merged back into the 
200     content property. All array methods will take place on this object.
201 
202     @field
203     @type {SC.Array}
204   */
205   contentClone: null,
206 
207   /**
208     Clones the content property into the contentClone property.
209     @private
210   */
211   contentCloneReset: function() {
212     this._changelog = [];
213     this.set('contentClone', null);
214   },
215 
216   /**
217    SC.Array interface implimentation.
218    
219    @param idx {Number} Starting index in the array to replace.  If idx >= 
220      length, then append to the end of the array.
221    
222    @param amt {Number} Number of elements that should be removed from the 
223      array, starting at *idx*.
224    
225    @param objects {Array} An array of zero or more objects that should be 
226      inserted into the array at *idx* 
227   */
228   replace: function(idx, amt, objects) {
229 
230     var content = this.get('content') ;
231 
232     // in case the passed objects are controllers, convert to source objects.
233     var copyIdx = objects.length ;
234     var sourceObjects = objects ;
235     if (copyIdx > 0) {
236       sourceObjects = [] ;
237       while(--copyIdx >= 0) {
238         sourceObjects[copyIdx] = this._sourceObjectFor(objects[copyIdx]) ;
239       }
240     }
241     
242     // create clone of content array if needed
243     var contentClone = this.get('contentClone') ;
244     if (!contentClone) {
245       this.set('contentClone', contentClone = content.clone()) ;
246     }
247 
248     // now, record the removed objects.  This may be used later.
249     if (this.get('destroyOnRemoval')) {
250       if (!this._deletions) this._deletions = [] ;
251       for (var i=0; i < amt; i++) {
252         this._deletions.push(content.objectAt(idx + i));
253       }
254     }
255 
256     // and record additions
257     if (!this._changelog) this._changelog = []; 
258     this._changelog.push({ idx: idx, amt: amt, objects: sourceObjects });
259     
260     // then actually perform the edit on the contentClone
261     contentClone.replace(idx, amt, sourceObjects);
262     
263     this.editorDidChange() ;
264     this.enumerableContentDidChange();
265     this.updateSelectionAfterContentChange();
266     
267     return this;
268   },
269   
270   /**
271     SC.Array interface implimentation.
272     @param idx {Number} The index of the item to return.  If idx exceeds the 
273       current length, return null.
274   */
275   objectAt: function(idx) {
276     var obj = this._getSourceContent() ;
277     obj = (obj && obj.objectAt) ? obj.objectAt(idx) : null;
278     return this._objectControllerFor(obj) ;
279   },
280   /**
281     SC.Array interface implimentation.
282     @property
283     @type {integer}
284   */
285   length: function( key, value ) {
286     var ret = this._getSourceContent() ;
287     return (ret && ret.get) ? (ret.get('length') || 0) : 0 ;
288   }.property(),
289 
290   /**
291     Returns the index in the array of the specified object.
292     
293     This can handle both controller wrapper objects and source content objects.
294   */
295   indexOf: function( obj ) {
296     return this._getSourceContent().indexOf(this._sourceObjectFor(obj)) ;
297   },
298   
299   _getSourceContent: function() {
300     return this.get('contentClone') || this.get('content') || [];
301   },
302   
303   /** 
304   * @private
305   */
306   performCommitChanges: function()
307   {
308     var content = this.get('content');
309     var ret     = true;
310     
311     // cannot commit changes to null content.  Return an error.
312     if (!content) {
313       return $error("No Content");
314     }
315     
316     if (content.beginPropertyChanges) content.beginPropertyChanges();
317 
318 
319     // apply all the changes made to the clone
320     if (this._changelog) {
321       var changelog = this._changelog ;
322       var max = changelog.length;
323       for(var idx=0;idx<max;idx++) {
324         var change = changelog[idx];
325         content.replace(change.idx, change.amt, change.objects) ;
326       }
327       this._changelog.length = 0 ; // reset changelog
328     }
329 
330     // finally, destroy any removed objects if necessary.  Make 
331     // sure the objects have not been re-added before doing this.
332     if (this.get('destroyOnRemoval') && this._deletions && this._deletions.length>0) {
333       var idx = this._deletions.length;
334       while(--idx >= 0) {
335         var obj = this._deletions[idx] ;
336         if (obj && obj.destroy && (content.indexOf(obj) < 0)) {
337           obj.destroy() ; 
338         }
339       }
340       this._deletions.length = 0; // clear array
341     }
342     
343     // changes commited, clear any created objects from the internal array
344     if (this._createdObjects) this._createdObjects.length = 0 ;
345     
346      // finish commiting changes.
347     if (content.endPropertyChanges) content.endPropertyChanges();
348     if (content.commitChanges) ret = content.commitChanges();
349     
350     if ($ok(ret)) {
351       this.contentCloneReset();
352       this.editorDidClearChanges();
353     }
354     
355     return ret;
356   },
357   /** 
358   * @private
359   */
360   performDiscardChanges: function()
361   {
362     this.contentCloneReset();
363     this.editorDidClearChanges();
364     
365     // if any objects were created before the commit, destroy the objects 
366     // and reset the array.
367     if (this._createdObjects && this._createdObjects.length > 0) {
368       var idx = this._createdObjects.length ;
369       while(--idx >= 0) {
370         var obj = this._createdObjects[idx] ;
371         if ($type(obj.destroy) === T_FUNCTION) obj.destroy() ;
372       }
373       this._createdObjects.length = 0 ;
374     }
375     
376     return true;
377   },
378   
379   /** @private
380     Returns the object controller for a source value.
381   */
382   _objectControllerFor: function(obj) {
383     if (!this.useControllersForContent) return obj;
384     
385     var controllers = this._objControllers = this._objControllers || {} ;
386     var guid = SC.getGUID(obj) ;
387     var ret = controllers[guid] ;
388     if (!ret) {
389       ret = controllers[guid] = this.controllerForValue(obj) ;
390       if (ret) ret.__isArrayController = true ;
391     }
392     return ret ;
393   },
394   
395   /** @private
396     Returns the source object for the passed value.  If the passed value is a 
397     controller, this will map back to the sourceo object.  Otherwise the object itself
398     will be returned.
399   */
400   _sourceObjectFor: function(obj) {
401     return (obj && obj.kindOf && obj.kindOf(SC.Controller)) ? obj.get('content') : obj ;
402   },
403   
404   init: function() {
405     sc_super() ;
406     if (this.get('content')) this._contentObserver() ;
407   }
