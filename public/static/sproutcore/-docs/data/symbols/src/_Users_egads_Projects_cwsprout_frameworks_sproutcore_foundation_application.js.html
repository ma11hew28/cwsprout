  6 require('core');
  7 require('foundation/responder');
  8 
  9 /** 
 10 
 11   @class
 12 
 13   The SC.Application object manages a SproutCore application. A single 
 14   instance is created and placed in a global variable named SC.app. All events 
 15   and actions are routed through this object.
 16  
 17   
 18   @extends SC.Responder
 19   @author Skip Baney
 20   @copyright 2006-2008, Sprout Systems, Inc. and contributors.
 21   @since SproutCore 1.0
 22 */
 23 SC.Application = SC.Responder.extend( 
 24 /** @scope SC.Application.prototype */ {
 25 
 26   /**
 27     The pane that is currently receiving key events.
 28     
 29     This is most often the SC.window object but if you display a popup or 
 30     some other such element, you may set this to some other value.
 31     
 32     @field
 33     @type {SC.View}
 34   */
 35   keyPane: function( key, value )
 36   {
 37     if ( value != undefined )
 38     {
 39       if (this._keyPane) this._keyPane.willResignKeyPane();
 40       if (this._keyPane) this._keyPane.set('isKeyPane', false);
 41       this._keyPane = value;
 42       if (this._keyPane) this._keyPane.set('isKeyPane', true);
 43       if (this._keyPane) this._keyPane.didBecomeKeyPane();
 44     }
 45     return this._keyPane || null;
 46   }.property(),
 47 
 48   /**
 49     The main pane for the application.  
 50     
 51     The main pane is usually the primary pane you expect users to work in. 
 52     It receives keyboard equivalent events and other operations even when
 53     another key pane is positioned over it.
 54     
 55     @field
 56     @type {SC.View}
 57   */
 58   mainPane: function( key, value )
 59   {
 60     if ( value != undefined )
 61     {
 62       if (this._mainPane) this._mainPane.willResignMainPane();
 63       if (this._mainPane) this._mainPane.set('isMainPane', false);
 64       this._mainPane = value;
 65       if (this._mainPane) this._mainPane.set('isMainPane', true);
 66       if (this._mainPane) this._mainPane.didBecomeMainPane();
 67     }
 68     return this._mainPane || null;
 69   }.property(),
 70 
 71   /**
 72     Starts the SproutCore application.
 73     
 74     Typically this will simply setup the SC.window object so that it starts
 75     to receive keyboard and mouse events and then makes the window both
 76     main and key pane.
 77     
 78     @returns {void}
 79   */
 80   run: function()
 81   {
 82     // primary application pane.
 83     SC.window.setup();
 84     SC.window.makeMainPane();
 85     SC.window.makeKeyPane();
 86   },
 87 
 88   /**
 89   * Route an action message to the appropriate responder
 90   * @param {String} action The action to perform - this is a method name.
 91   * @param {SC.Responder} target The object to perform the action upon. Set to null to search the Responder chain for a receiver.
 92   * @param {Object} sender The sender of the action
 93   * @returns return value info
 94   * @type Array
 95   */
 96   sendAction: function( action, target, sender )
 97   {
 98     var target = this.targetForAction(action, target, sender);
 99     return (!!target && (target.tryToPerform(action, sender) != false));
100   },
101   
102   targetForAction: function( action, target, sender )
103   {
104     // no action, no target...
105     if (!action || ($type(action) != T_STRING)) return null;
106 
107     // an explicit target was passed...
108     if (target) {
109       if ($type(target) === T_STRING) {
110         target = SC.Object.objectForPropertyPath(target) ;
111       }
112       return target.respondsTo(action) ? target : null ;
113     }
114 
115     // ok, no target was passed... try to find one in the responder chain
116     var keyPane   = this.get('keyPane');
117     var mainPane  = this.get('mainPane');
118 
119     // TODO: add in a check for Pane and App delegates once we add support for them...
120     if (keyPane)
121     {
122       target = keyPane.get('firstResponder') || keyPane.get('defaultResponder') || keyPane;
123       do {
124         if (target.respondsTo(action)) return target;
125       } while (target = target.get('nextResponder'));
126     }
127     if (mainPane && (mainPane != keyPane))
128     {
129       target = mainPane.get('firstResponder') || mainPane.get('defaultResponder') || mainPane;
130       do {
131         if (target.respondsTo(action)) return target;
132       } while (target = target.get('nextResponder'));
133     }
134     // last stop, SC.app
135     target = this;
136     if (target.respondsTo(action)) return target;
137 
138     return null;
139   },
140   
141 
142   sendEvent: function( evt, target )
143   {
144     var target  = target || null;
145     var handler = null;
146 
147     //console.log( '[SC.Application#sendEvent] type: %s, evt: %o, target: %o', evt._type, evt, target );
148 
149     if (target && target.respondsTo(evt._type))
150     {
151       // explicit target was passed... we send the event to that target only...
152       return (target.tryToPerform(evt._type, evt)) ? target : false;
153     }
154     
155     // ok... then we need to resolve the target based on event type...
156     switch ( evt._type )
157     {
158       case 'keyDown':
159       case 'keyUp':
160       case 'flagsChanged':
161         var pane = this.get('keyPane');
162         if (!pane) return null;
163         target = pane.get('firstResponder') || pane.get('defaultResponder') || pane;
164         break;
165       case 'mouseOver':
166       case 'mouseOut':
167       case 'mouseMoved':
168       case 'mouseDown':
169       case 'mouseUp':
170       case 'click':
171       case 'doubleClick':
172         target = SC.window.firstViewForEvent(evt);
173         break;
174       default:
175         // unrecognized event type.. bail...
176         return null;
177     }
178 
179     // unable to resolve a target... mouseDown on an element not in the view chain, key responder is null, etc...
180     if (!target) return null;
181 
182     // calling doCommand will crawl up the responder chain until someone handles it or it hits the end of the chain (SC.PaneView).
183     handler = target.doCommand( evt._type, evt );
184     
185     // unhandled keyDown event...
186     if ((evt._type == 'keyDown') && !handler)
187     {
188       // lonely keyDown... the responder chain doesn't want you... 
189       // perhaps someone else has defined you as a shortcut?
190       if (this._attemptKeyEquivalent(evt)) return true;
191       // no?... how about keyView navigation?
192       if (this._attemptKeyInterfaceControl(evt)) return true;
193     }
194     
195     //console.log( '[SC.Application#sendEvent] ----- handler: %o', handler );
196     return handler;
197   },
198 
199   _attemptKeyEquivalent: function( evt )
200   {
201     // keystring is a method name representing the keys pressed (i.e 'alt_shift_escape')
202     var keystring = SC.Responder.inputManager.codesForEvent(evt).first();
203     
204     // inputManager couldn't build a keystring for this key event... nothing to do...
205     if (!keystring) return false;
206     
207     var keyPane  = this.get('keyPane');
208     var mainPane = this.get('mainPane');
209     
210     if (keyPane && keyPane.performKeyEquivalent(keystring, evt)) return true;
211     if (mainPane && (mainPane != keyPane) && mainPane.performKeyEquivalent(keystring, evt)) return true;
212     return this.performKeyEquivalent(keystring, evt);
213   },
214   
215   _attemptKeyInterfaceControl: function( evt )
216   {
217     // keystring is a method name representing the keys pressed (i.e 'alt_shift_escape')
218     var keystring = SC.Responder.inputManager.codesForEvent(evt).first();
219 
220     //console.log( '[SC.Application#_attemptKeyInterfaceControl] keystring: %s, evt: %o', keystring, evt );
221 
222     var pane = this.get('keyPane');
223     if ( !pane ) return false;
224     
225     return pane.performKeyInterfaceControl(keystring, evt);
226   }
227
