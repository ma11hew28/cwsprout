  6 require('views/field/field') ;
  7 
  8 /**
  9   @class
 10   
 11   Supports using platform native checkboxs with the input field.  If you 
 12   would like to make use of the extended SproutCore checkbox features
 13   including a mixed state and theming support, use SC.CheckboxView instead.
 14   
 15   @extends SC.FieldView
 16   @author Charles Jolley
 17   @version 1.0
 18 */
 19 SC.CheckboxFieldView = SC.FieldView.extend(
 20 /** @scope SC.CheckboxFieldView.prototype */ {
 21   
 22   emptyElement: '<input type="checkbox" value="1" />',
 23   
 24   setFieldValue: function(value) {
 25     this.rootElement.checked = !!value;
 26   },
 27   
 28   getFieldValue: function() {
 29     return this.rootElement.checked;
 30   },
 31   
 32   valueBindingDefault: SC.Binding.Bool,
 33   
 34   init: function() {
 35     sc_super() ;
 36     var f = this.fieldValueDidChange.bind(this, false) ;
 37     Event.observe(this.rootElement,'click', f) ;
 38   }
 39   
 40 }) ;
