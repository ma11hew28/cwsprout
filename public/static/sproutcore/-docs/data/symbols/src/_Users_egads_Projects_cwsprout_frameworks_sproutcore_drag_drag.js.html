 13 /**
 14 
 15   @class
 16   
 17   An instance of this object is created whenever a drag occurs.  The instance
 18   manages the mouse events and coordinating with droppable targets until the
 19   user releases the mouse button. 
 20 
 21   To initiate a drag, you should call SC.Drag.start() with the options below
 22   specified in a hash. Pass the ones you need to get the drag you want:  
 23    
 24   - *event: (req)* The mouse event that triggered the drag.  This will be used 
 25     to position the element.
 26   
 27    - *source: (req)* The drag source object that should be consulted during 
 28      the drag operations. This is usually the container view that initiated 
 29      the drag.
 30   
 31    - *dragView: (req)*  This should point to a view that will be used as the 
 32      source image for the drag. The drag operation will clone the DOM elements 
 33      for this view and add the class name 'drag-image' to the outermost 
 34      element.
 35   
 36    - *ghost:  YES | NO*  If YES or not passed, then drag view image will show, 
 37      but the source dragView will not be hidden.  set to NO to make it appear 
 38      that the dragView itself is being dragged around.
 39   
 40    - *slideBack: YES | NO*  If YES or not specified, then if the drag 
 41      operation is cancelled, the dragView will slide back to its source 
 42      origin.
 43   
 44    - *origin:*  If passed, this will be used as the origin point for the 
 45      ghostView when it slides back.  You normally do not need to pass this 
 46      unless the ghost view does not appear in the main UI.
 47   
 48    - *data:* Optional hash of data types and values.  You can use this to pass 
 49      a static set of data instead of providing a dataSource.  If you provide
 50      a dataSource, it will override this.
 51   
 52    - *dataSource:*  Optional object that will provide the data for the drag to 
 53      be consumed by the drop target.  If you do not pass this parameter or the 
 54      data hash, then the source object will be used if it implements the 
 55      SC.DragDataSource protocol.
 56   
 57    - *anchorView:* if you pass this optional view, then the drag will only be 
 58      allowed to happen within this view.  The ghostView will actually be added 
 59      as a child of this view during the drag.  Normally the anchorView is the 
 60      window.
 61    
 62   @extends SC.Object
 63 */

115   /**
116     Data types supported by this drag operation.
117     
118     Returns an array of data types supported by the drag source.  This may 
119     be generated dynamically depending on the data source.
120 
121     If you are implementing a drag source, you will need to provide these data
122     types so that drop targets can detect if they can accept your drag data.
123     
124     If you are implementing a drop target, you should inspect this property
125     on your dragEntered() and prepareForDragOperation() methods to determine 
126     if you can handle any of the data types offered up by the drag source.
127     
128     @field {Array} available data types
129   */
130   dataTypes: function() {
131     
132     // first try to use the data source.
133     if (this.dataSource) return this.dataSource.get('dragDataTypes') ;
134     
135     // if that fails, get the keys from the data hash.
136     if (this.data) {
137       var ret = [];
138       for(var key in this._data) {
139         if (this.data.hasOwnProperty(key)) ret.push(key) ;
140       }
141       return ret ;
142     }    
143     
144     // if that fails, then check to see if the source object is a dataSource.
145     var source = this.get('source') ;
146     if (source && source.dragDataTypes) return source.get('dragDataTypes') ;
147     
148     // no data types found. :(
149     return [] ;

175

185     } else {

188         return source.dragDataForType(dataType, this) ;

230   // this will actually start the drag process.
231   startDrag: function() {
232     
233     // create the ghost view and position.
234     this._createGhostView() ;
235     
236     // compute the offset from the original mouse location.
237     var origin = this.dragView.convertFrameToView(this.dragView.get('frame'), null);
238     var pointer = Event.pointerLocation(this.event) ;
239     
240     window.dragEvent = this.event ;
241     
242     this.ghostOffset = { x: (pointer.x-origin.x), y: (pointer.y-origin.y) };
243 
244     // position the ghost view.
245     this._positionGhostView(this.event) ;
246     
247     // notify window a drag is in process. mouseDragged notifications will
248     // go to the drag instead.
249     SC.window.dragDidStart(this) ;
250     
251     if (this.source && this.source.dragDidBegin) {
252       this.source.dragDidBegin(this, pointer) ;
253     }

258   /**  
259     @private
260     
261     This method is called repeatedly during a mouse drag.  It updates the
262     position of the ghost image, then it looks for a current drop target and
263     notifies it.
264   */
265   mouseDragged: function(evt) {
266     
267     var loc = Event.pointerLocation(evt) ;
268     var scrolled = this._autoscroll(evt) ;
269     
270     // ignore duplicate calls.
271     if (!scrolled && (loc.x == this._lastLoc.x) && (loc.y == this._lastLoc.y)) return ;
272     this._lastLoc = loc ;
273     this.set('location', loc) ;
274 
275     this._positionGhostView(evt) ;
276     var last = this._lastTarget ;
277 
278     // STEP 1: Determine the deepest drop target that allows an operation.
279     // if the drop target selected the last time this method was called differs
280     // from the deepest target found, then go up the chain until we either hit the
281     // last one or find one that will allow a drag operation
282     var target = this._findDropTarget(evt) ; // deepest drop target
283     var op = SC.DRAG_NONE ;
284     
285     while (target && (target != last) && (op == SC.DRAG_NONE)) {
286       
287       // make sure the drag source will permit a drop operation on the named target.
288       // if source does not implement this callback, just assume a drop is allowed
289       // by the source.
290       if (target && this.source && this.source.dragSourceOperationMaskFor) {
291         op = this.source.dragSourceOperationMaskFor(target, this) ;
292       } else op = SC.DRAG_ANY ;
293 
294       // now, let's see if the target will accept the drag.  If it does not respond
295       // to dragEntered, then assume NO drag opts.
296       if ((op != SC.DRAG_NONE) && target && target.dragEntered) {
297         op = op & target.dragEntered(this, evt) ;
298       } else op = SC.DRAG_NONE ;
299 
300       // if DRAG_NONE, then look for the next parent that is a drop zone.
301       if (op == SC.DRAG_NONE) target = this._findNextDropTarget(target) ;
302     }
303 
304     // STEP 2: Refocus the drop target.
305     // If a new drop target was found then this part of the method will exit the
306     // last drag target and start the new one.
307     if (target != last) {
308 
309       // if the new target does not match the last target, exit that target.
310       if (last && last.dragExited) last.dragExited(this, evt) ;
311 
312       if (target && this.source && this.source.dragSourceOperationMaskFor) {
313         op = this.source.dragSourceOperationMaskFor(target, this) ;
314       } else op = SC.DRAG_ANY ;
315 
316       // save new op and set target to null if op = DRAG_NONE
317       this.sourceDropOperations = op ;
318       
319       // now notify the new target, if there is one.  Save the allowed drop
320       // operations as the logical AND between the ops allowed by the source
321       // and target.
322       if (target && target.dragEntered) {
323         this.dropOperations = op & target.dragEntered(this, evt) ;
324       } else this.dropOperations = SC.DRAG_NONE ;
325 
326       if (this.dropOperations == SC.DRAG_NONE) target = null ;
327     
328     // if nothing has changed, send dragUpdated
329     } else {
330       if (target && target.dragUpdated) target.dragUpdated(this, evt) ;
331     }
332      
333     // notify source that the drag moved.
334     if (this.source && this.source.dragDidMove) {
335       this.source.dragDidMove(this, loc) ;
336     }   
337     
338     this._lastTarget = target ;
339   },

341   /**
342     @private
343     
344     Called when the mouse is released.  Performs any necessary cleanup and
345     executes the drop target protocol to try to complete the drag operation.
346   */
347   mouseUp: function(evt) {
348     var loc = Event.pointerLocation(evt) ;
349 
350     // try to have the drop target perform the drop...
351     var target = this._lastTarget ;
352     var op = this.dropOperations;
353     
354     if (target && target.prepareForDragOperation(op, this)) {
355       op = target.performDragOperation(op, this) ;  
356     } else {
357       op = SC.DRAG_NONE;
358     }
359     
360     // create cleanupFunc.  This function will be called at the end of this
361     // function or after the ghostView slides back to its origin.
362     var drag = this ;
363     var cleanupFunc = function() {
364       if (target) target.concludeDragOperation(op, this) ;  
365       drag._destroyGhostView() ;
366     };
367     
368     // notify drop target.
369     if (target && target.dragEnded) target.dragEnded(this, evt) ;
370     this._lastTarget = null ;
371     
372     // clean up ghost view.  if sldeBack is true, then do the animation.
373     if ((op == SC.DRAG_NONE) && this.get('slideBack')) {
374       var loc = this.dragView.convertFrameToView(this.dragView.get('origin'), null) ;
375       this._ghostView.transitionTo(1.0, 
376         "left: %@px; top: %@px".fmt(loc.x, loc.y), 
377         { duration: 200, onComplete: cleanupFunc }) ;
378         
379     } else cleanupFunc() ;
380     
381     // notify the source that everything has completed.
382     if (this.source && this.source.dragDidEnd) {
383       this.source.dragDidEnd(this, loc, op) ;
384     }
385     
386     this._dragInProgress = NO ; // required by autoscroll.
387     
388   },

433   // Return an array of drop targets, sorted with any nested drop targets
434   // at the top of the array.  The first time this method is called during
435   // a drag, it will reconstruct this array using the current set of 
436   // drop targets.  Afterwards it uses the cached set until the drop
437   // completes.
438   //
439   // This means that if you change the view hierarchy of your drop targets
440   // during a drag, it will probably be wrong.
441   _getOrderedDropTargets: function() {
442     if (this._cachedDropTargets) return this._cachedDropTargets ;
443     var ret = [];
444     
445     // build array of drop targets
446     var dt = SC.Drag._dropTargets ;
447     for(var key in dt) {
448       if (!dt.hasOwnProperty(key)) continue ;
449       ret.push(dt[key]) ;      
450     }
451 
452     // views must be sorted so that drop targets with the deepest nesting 
453     // levels appear first in the array.  The getDepthFor().
454     var depth = {} ;
455     var getDepthFor = function(x) {
456       if (!x) return 0 ;
457       var guid = SC.guidFor(x);
458       var ret = depth[guid];
459       if (!ret) {
460         ret = 1 ;
461         while((x = x.parentNode) && (x !== SC.window)) {
462           if (dt[SC.guidFor(x)] !== undefined) ret++ ;
463         }
464         depth[guid] = ret ;
465       }
466       return ret ;
467     } ;
468 
469     ret.sort(function(a,b) {
470       if (a===b) return 0;
471       a = getDepthFor(a) ;
472       b = getDepthFor(b) ;
473       return (a > b) ? -1 : 1 ;
474     }) ;
475 
476     this._cachedDropTargets = ret ;
477     
478     return ret ;
479   },
480   
481   // This will search through the drop targets, looking for one in the 
482   // target area.
483   _findDropTarget: function(evt) {
484     var dt = this._getOrderedDropTargets() ;
485     var loc = Event.pointerLocation(evt) ;
486 
487     var ret = null ;
488     for(var idx=0;idx<dt.length;idx++) {
489       var t = dt[idx] ;
490 
491       if(!t.get('isVisibleInWindow')) continue ;
492       
493       // get frame, converted to view.
494       var f = t.convertFrameToView(t.get('clippingFrame'), null) ;
495       
496       // check to see if loc is inside.  If so, then make this the drop
497       // target unless there is a drop target and the current one 
498       // is not deeper.
499       if (SC.pointInRect(loc, f)) return t;
500     } 
501     return null ;
502   },
503   
504   // Search the parent nodes of the target to find another view matching the 
505   // drop target.  Returns null if no matching target is found.
506   _findNextDropTarget: function(target) {
507     while ((target = target.parentNode) && (target != SC.window)) {
508       if (SC.Drag._dropTargets[target._guid]) return target ;
509     }
510     return null ;
511   },

513   // ............................................
514   // AUTOSCROLLING
515   //
516   
517   // Performs auto-scrolling for the drag.  This will only do anything if
518   // the user keeps the mouse within a few pixels of one location for a little
519   // while.
520   //
521   // Returns true if a scroll was performed
522   _autoscroll: function(evt) {
523   
524     // If drag has ended, exit
525     if (!this._dragInProgress) return ;
526     
527     // STEP 1: Find the first view that we can actually scroll.  This view 
528     // must be:
529     // - scrollable
530     // - the mouse pointer must be within a scrolling hot zone
531     // - there must be room left to scroll in that direction. 
532     
533     // NOTE: an event is passed only when called from mouseDragged
534     var loc = (evt) ? Event.pointerLocation(evt) : this._lastMouseLocation ;
535     if (!loc) return false ;
536     this._lastMouseLocation = loc ;
537 
538     var view = this._findScrollableView(loc) ;
539     
540     // these will become either 1 or -1 to indicate scroll direction or 0 for no scroll.
541     var verticalScroll, horizontalScroll ;
542     var min, max, edge ;
543     var scrollableView = null;
544 
545     while(view && !scrollableView) {
546       
547       // quick check...can we scroll this view right now?
548       verticalScroll = view.get('hasVerticalScroller') ? 1 : 0;
549       horizontalScroll = view.get('hasHorizontalScroller') ? 1 : 0;
550       
551       // at least one direction might be scrollable.  Collect some extra
552       // info to investigate further.
553       if ((verticalScroll != 0) || (horizontalScroll != 0)) {
554         var f = view.convertFrameToView(view.get('frame'), null) ;
555         var innerSize = view.get('innerFrame') ;
556         var scrollFrame = view.get('scrollFrame') ;
557       }
558       
559       
560       if (verticalScroll != 0) {
561         
562         // bottom hotzone?
563         max = SC.maxY(f); min = max - SC.DRAG_AUTOSCROLL_ZONE_THICKNESS ; 
564         edge = SC.maxY(scrollFrame) ;
565         
566         if ((edge >= innerSize.height) && (loc.y >= min) && (loc.y <= max)) {
567           verticalScroll = 1 ;
568           
569         // no...how about top hotzone?
570         } else {
571           min = SC.minY(f); max = min + SC.DRAG_AUTOSCROLL_ZONE_THICKNESS ;
572           edge = SC.minY(scrollFrame) ;
573           if ((edge <= innerSize.height) && (loc.y >= min) && (loc.y <= max)) {
574             verticalScroll = -1 ;
575           
576           // no, ok don't scroll vertical
577           } else verticalScroll = 0 ;
578         }
579       }
580 
581       if (horizontalScroll != 0) {
582         // right hotzone?
583         max = SC.maxX(f); min = max - SC.DRAG_AUTOSCROLL_ZONE_THICKNESS ; 
584         edge = SC.maxX(scrollFrame) ;
585         if ((edge >= innerSize.width) && (loc.x >= min) && (loc.x <= max)) {
586           horizontalScroll = 1 ;
587           
588         // no...how about left hotzone?
589         } else {
590           min = SC.minY(f); max = min + SC.DRAG_AUTOSCROLL_ZONE_THICKNESS ;
591           edge = SC.minY(scrollFrame) ;
592           if ((edge <= innerSize.width) && (loc.x >= min) && (loc.x <= max)) {
593             horizontalScroll = -1 ;
594           
595           // no, ok don't scroll vertical
596           } else horizontalScroll = 0 ;
597         }
598       }
599 
600       // if we can scroll, then set this.
601       if ((verticalScroll != 0) || (horizontalScroll != 0)) {
602         scrollableView = view ;
603       } else view = this._findNextScrollableView(view) ;
604     }
605 
606     
607     // STEP 2: Only scroll if the user remains within the hot-zone for a period of
608     // time
609     if (scrollableView && (this._lastScrollableView == scrollableView)) {
610       if ((Date.now() - this._hotzoneStartTime) > 100) {
611         this._horizontalScrollAmount *= 1.05 ;
612         this._verticalScrollAmount *= 1.05 ;
613       }
614       
615     // otherwise, reset everything and disallow scroll
616     } else {
617       this._lastScrollableView = scrollableView ;
618       this._horizontalScrollAmount = 15 ;
619       this._verticalScrollAmount = 15 ;
620       this._hotzoneStartTime = (scrollableView) ? Date.now() : null ;
621       
622       horizontalScroll = verticalScroll = 0 ;
623     }
624     
625     // STEP 3: Scroll!
626     if (scrollableView && ((horizontalScroll != 0) || (verticalScroll != 0))) {
627       var scroll = { 
628         x: horizontalScroll * this._horizontalScrollAmount,
629         y: verticalScroll * this._verticalScrollAmount 
630       } ;
631 
632       scrollableView.scrollBy(scroll) ;
633       
634     }
635 
636     // If a scrollable view was found, then reschedule
637     if (scrollableView) {
638       this.invokeLater('_autoscroll', 100, null);
639       return true ;
640     } else return false ;
641   },
642 
643   // Returns an array of scrollable views, sorted with nested scrollable
644   // views at the top of the array.  The first time this method is called
645   // during a drag, it will reconstrut this array using the current ste of
646   // scrollable views.  Afterwards it uses the cached set until the drop
647   // completes.
648   _scrollableViews: function() {
649     if (this._cachedScrollableView) return this._cachedScrollableView ;
650     var ret = [];
651     
652     // build array of drop targets
653     var dt = SC.Drag._scrollableViews ;
654     for(var key in dt) {
655       if (!dt.hasOwnProperty(key)) continue ;
656       ret.push(dt[key]) ;      
657     }
658     
659     // now resort.  This custom function will sort nested drop targets
660     // at the start of the list.
661     ret = ret.sort(function(a,b) {
662       var view = a;
663       while((view = view.parentNode) && (view != SC.window)) {
664         if (b == view) return -1 ;
665       }
666       return 1; 
667     }) ;
668 
669     this._cachedScrollableView = ret ;
670     
671     return ret ;
672   },
673   
674   // This will search through the scrollable views, looking for one in the 
675   // target area.
676   _findScrollableView: function(loc) {
677     var dt = this._scrollableViews() ;
678 
679     var ret = null ;
680     for(var idx=0;idx<dt.length;idx++) {
681       var t = dt[idx] ;
682 
683       if(!t.get('isVisibleInWindow')) continue ;
684       
685       // get frame, converted to view.
686       var f = t.convertFrameToView(t.get('frame'), null) ;
687       
688       // check to see if loc is inside.  
689       if (SC.pointInRect(loc, f)) return t;
690     } 
691     return null ;
