  6 require('views/view') ;

  8 /** @class
  9 
 10   This is a basic menu item for inclusion in a popup menu.  This is a type
 11   of button that will automatically close the popup menu when it is
 12   pressed.
 13   
 14   @extends SC.ButtonView
 15   
 16 */
 17 SC.MenuItemView = SC.ButtonView.extend({
 18 
 19   emptyElement: [
 20   '<li class="button menu-item">',
 21     '<a href="javascript:;">',
 22       '<span class="sel">✓</span>',
 23       '<span class="mixed">-</span>',
 24       '<span class="inner">',
 25       '<span class="label"></span>',
 26       '</span>',
 27       '<span class="shortcut"></span>',
 28     '</a>',
 29   '</li>'].join(''),
 30   
 31   // this method returns the computed required width.  This is potentially
 32   // expensive, so don't call it often.  It is intended to be used with the
 33   // wrapper MenuView that will decide how wide to make the menu.
 34   computedRequiredWidth: function() {
 35     
 36     var ret = 0;
 37     
 38     // first, get the left edge offset for the .inner span.
 39     // we expect this to make room for any checkboxes that might appear on 
 40     // the left and required spacing on the right.
 41     var el = this.$sel('.inner') ;
 42     if (el) {
 43       ret = el.offsetLeft ;
 44       ret += parseInt(Element.getStyle(el, 'padding-left'),0) ;
 45       ret += parseInt(Element.getStyle(el, 'padding-right'),0) ;
 46     }
 47     
 48     // next, add in the width of any img tag.
 49     var img = Element.$sel(el,'img') ;
 50     if (img) {
 51       ret += Element.getDimensions(img).width ;
 52     }
 53     
 54     // next add in the width of any label.  We assume this includes the width
 55     // of the label text itself.
 56     el = Element.$sel(el, '.label') ;
 57     if (el) {
 58       ret += Element.getDimensions(el).width ;
 59     }
 60     
 61     // finally, add the width of any shortcut.  We assume this includes any
 62     // padding required to go between the label and the shortcut.
 63     el = this.$sel('.shortcut') ;
 64     if (el) {
 65       ret += Element.getDimensions(el).width ;
 66     }
 67     
 68     // that should do it...
 69     return ret ;
 70   },
 71   
 72   mouseMoved: function(evt)
 73   {
 74     if (!this.get('isDefault')) this.get('parentNode').set('currentSelectedMenuItem', this);
 75   },
 76   mouseOut: function(evt)
 77   {
 78     this.set('isDefault', false);
 79     this.setClassName('active', false);
 80   },
 81   
 82   mouseUp: function(evt)
 83   {
 84     sc_super();
 85     this._closeParentMenu();
 86   },
 87   didTriggerAction: function()
 88   {
 89     this._closeParentMenu();
 90   },
 91   _closeParentMenu: function()
 92   {
 93     var menu = this.get('parentNode');
 94     if (menu) menu.set('isVisible', false);
 95   }
 96   
 97 });
