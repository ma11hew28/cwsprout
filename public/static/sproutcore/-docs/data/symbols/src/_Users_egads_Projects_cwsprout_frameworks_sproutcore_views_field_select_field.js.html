  8 /**
  9   @class
 10 
 11   SelectFieldView displays browser-native popup menu.  To use this view,
 12   you should either bake into the HTML the preset list of options, or 
 13   you can set the -objects property to an array of items to show.  The
 14   value is current value of the select.
 15   
 16   @extends SC.FieldView
 17   @author Charles Jolley
 18   @version 1.0
 19 */
 20 SC.SelectFieldView = SC.FieldView.extend(
 21 /** @scope SC.SelectFieldView.prototype */ {
 22 
 23  emptyElement: '<select></select>',
 24  
 25   /**
 26     An array of items that will form the menu you want to show.
 27   */ 
 28  objects: null,
 29 
 30   /**
 31     If you set this to a non-null value, then the name shown for each 
 32     menu item will be pulled from the object using the named property.
 33     if this is null, the collection objects themselves will be used.
 34   */
 35  nameKey: null,
 36 
 37   /**
 38    If you set this to a non-null value, then the value of this key will
 39    be used to sort the objects.  If this is not set, then nameKey will
 40    be used.
 41   */ 
 42   sortKey: null,
 43 
 44   /**
 45      Set this to a non-null value to use a key from the passed set of objects
 46      as the value for the options popup.  If you don't set this, then the
 47      objects themselves will be used as the value.
 48   */ 
 49   valueKey: null,
 50 
 51   /**
 52     set this to non-null to place an empty option at the top of the menu.   
 53   */
 54   emptyName: null,
 55 
 56   /**
 57     if true, the empty name will be localized.
 58   */
 59   localize: false,
 60 
 61   /**
 62     override this to change the enabled/disabled state of menu items as they
 63     are built.  Return false if you want the menu item to be disabled.
 64     
 65     @param itemValue the value for the item to validate
 66     @param itemName the name of the menu item to validate
 67     @returns YES if the item should be enabled, NO otherwise
 68   */  
 69   validateMenuItem: function(itemValue, itemName) {
 70    return true ;
 71   },
 72 
 73   /**
 74     override this method to implement your own sorting of the menu. By
 75     default, menu items are sorted using the value shown or the sortKey
 76     
 77     @param objects the unsorted array of objects to display.
 78     @returns sorted array of objects
 79   */
 80   sortObjects: function(objects) {
 81    var nameKey = this.get('sortKey') || this.get('nameKey') ;
 82    objects = objects.sort(function(a,b) {
 83      if (nameKey) {
 84        a = (a.get) ? a.get(nameKey) : a[nameKey] ;
 85        b = (b.get) ? b.get(nameKey) : b[nameKey] ;
 86      }
 87      return (a<b) ? -1 : ((a>b) ? 1 : 0) ;
 88    }) ;
 89 
 90    return objects ;
 91   },
 92 
 93   /**
 94     call this method to rebuild the menu manually.  Normally you should not
 95     need to do this since the menu will be rebuilt as its data changes.
 96   */
 97   rebuildMenu: function() {
 98    this._rebuildMenu() ;
 99   },
100 
101   // .......................................
102   // PRIVATE
103   //
104 
105   /** @private */
106   mouseDown: function(e) { e._stopWhenHandled = false; return false; },
107 
108   // when fetching the raw value, convert back to an object if needed...
109   /** @private */
110   getFieldValue: function() {
111    var value = this.rootElement.value ; // get raw value...
112    var valueKey = this.get('valueKey') ;
113    var objects = this.get('objects') ;
114  
115    // Handle empty selection.
116    if (value == '***') {
117      value = null ;
118 
119    // If no value key was set and there are objects then match back to an
120    // object.
121    } else if (value && objects) {
122      objects = Array.from(objects) ;
123      var loc = objects.length ;
124      var found = null ; // matching object goes here.
125      while(!found && (--loc >= 0)) {
126        var object = objects[loc] ;
127      
128        // get value using valueKey if there is one or use object
129        // map to _guid or toString.
130        if (valueKey) object = (object.get) ? object.get(valueKey) : object[valueKey] ;
131        ov = (object) ? ((object._guid) ? object._guid : object.toString()) : null ;
132      
133        // use this object value if it matches.
134        if (value == ov) found = object ;
135      }
136    }
137  
138    return value ;
139   },
140 
141   // when setting the raw value, convert from object...
142   /** @private */
143   setFieldValue: function(nv) {
144    if (nv) {
145      nv = (nv._guid) ? nv._guid : nv.toString() ;
146    } else {
147      nv = "***" ;
148    }
149    if (this.rootElement.value != nv) this.rootElement.value = nv ;
150   },
151 
152   // this method is called anytime the objects property or any of its member
153   // objects change.
154   _rebuildMenu: function() {     
155    // get list of objects.
156    var nameKey = this.get('nameKey') ;
157    var valueKey = this.get('valueKey') ;
158    var objects = this.get('objects') ;
159    var fieldValue = this.get('value') ;
160    
161    // get the localization flag.
162    var shouldLocalize = this.get('localize'); 
163    
164    // convert fieldValue to guid, if it is an object.
165    if (!valueKey && fieldValue) fieldValue = fieldValue._guid ;
166    if ((fieldValue == null) || (fieldValue == '')) fieldValue = '***' ;
167 
168    if (objects) {
169      objects = Array.from(objects) ; // make array.
170      objects = this.sortObjects(objects) ; // sort'em.
171      var html = [] ;       
172 
173      var emptyName = this.get('emptyName') ;
174      if (emptyName) {
175        if (shouldLocalize) emptyName = emptyName.loc() ;
176        html.push('<option value="***">%@</option>'.fmt(emptyName)) ;
177        html.push('<option disabled="disabled"></option>') ;
178      }
179 
180      // generate option elements.
181      objects.each(function(object) {
182        if (object) {
183        
184          // either get the name from the object or convert object to string.
185          var name = (nameKey) ? ((object.get) ? object.get(nameKey) : object[nameKey]) : object.toString() ;
186 
187          // localize name if specified.
188          if(shouldLocalize)
189          {
190            name = name.loc();
191          }
192 
193          // get the value using the valueKey or the object if no valueKey.
194          // then convert to a string or use _guid if one of available.
195          var value = (valueKey) ? ((object.get) ? object.get(valueKey) : object[valueKey]) : object ;
196          if (value) value = (value._guid) ? value._guid : value.toString() ;
197        
198          // render HTML
199          var disable = (this.validateMenuItem && this.validateMenuItem(value, name)) ? '' : 'disabled="disabled" ' ;
200          html.push('<option %@value="%@">%@</option>'.fmt(disable,value,name)) ;
201    
202        // null value means separator.
203        } else {
204          html.push('<option disabled="disabled"></option>') ;
205        }
206      }.bind(this) );
207 
208      // replace the contents of this HTML element.
209      this.update(html.join(""));
210      this.rootElement.value = fieldValue ;
211 
212    } else {
213      this.set('value',null);
214    }
215   },
216 
217   // object changes to the objects array of objects if possible.
218   _objectsObserver: function() {
219    if (!this._boundObserver) {
220      this._boundObserver = this._objectsItemObserver.bind(this) ;
221    }
222 
223    if (this.didChangeFor('_objO','objects','nameKey','valueKey')) {
224      var loc ;
225      var objects = Array.from(this.get('objects')) ;
226      var func = this._boundObserver ;
227    
228      // stop observing old objects.
229      if (this._objects) {
230        loc = this._objects.length ;
231        while(--loc >= 0) {
232          var object = this._objects[loc] ;
233          if (object && object.removeObserver) {
234            if (this._nameKey && this._valueKey) {
235              object.removeObserver(this._nameKey, func) ;
236              object.removeObserver(this._valueKey, func) ;
237            } else {
238              object.removeObserver('*', func) ;
239            } // if (this._nameKey)
240          } // if (object &&...)
241        } // while(--loc)
242      } // if (this._objects)
243 
244      // start observing new objects.
245      this._objects = objects ;
246      this._nameKey = this.get('nameKey') ;
247      this._valueKey = this.get('valueKey') ;
248    
249      if (this._objects) {
250        loc = this._objects.length ;
251        while(--loc >= 0) {
252          var object = this._objects[loc] ;
253          if (object && object.addObserver) {
254            if (this._nameKey && this._valueKey) {
255              object.addObserver(this._nameKey, func) ;
256              object.addObserver(this._valueKey, func) ;
257            } else {
258              object.addObserver('*', func) ;
259            } // if (this._nameKey)
260          } // if (object &&...)
261        } // while(--loc)
262      } // if (this._objects)
263 
264     this._rebuildMenu() ;
265    } // if (this.didChangeFor...)
266   }.observes('objects','nameKey','valueKey'),
267 
268   // this is invoked anytime an item we are interested in in the menu changes
269   // rebuild the menu when this happens, but only one time.
270   _objectsItemObserver: function(item, key, value) {
271     if (item.didChangeFor(this._guid, key)) {
272       // console.log('rebuildMenu') ;
273       this._rebuildMenu() ;
274     }
275   },
276 
277   _fieldDidFocus: function() {
278     var isFocused = this.get('isFocused');
279     if (!isFocused) this.set('isFocused', true);
280   },
281 
282   _fieldDidBlur: function() {
283     var isFocused = this.get('isFocused');
284     if (isFocused) this.set('isFocused', false);
285   },
286 
287   _isFocusedObserver: function() {
288     var isFocused = this.get('isFocused');
289     this.setClassName('focus', isFocused);
290   }.observes('isFocused'),
291 
292 
293   init: function() {
294     arguments.callee.base.call(this);
295     this._rebuildMenu();
296     
297     var changeListener = this.fieldValueDidChange.bind(this,false);
298     Element.observe(this.rootElement, 'change', changeListener);
299 
300     var focusListener = this._fieldDidFocus.bindAsEventListener(this);
301     Element.observe(this.rootElement, 'focus', focusListener);
302 
303     var blurListener = this._fieldDidBlur.bindAsEventListener(this);
304     Element.observe(this.rootElement, 'blur', blurListener);
305   }
306  
307 });
