  6 require('views/field/field') ;
  7 require('mixins/editable') ;

  9 /**
 10   @class
 11   
 12   A text field is an input element with type "text".  This view adds support
 13   for hinted values, etc.
 14   
 15   @extends SC.FieldView
 16   @extends SC.Editable
 17   @author Charles Jolley
 18 */
 19 SC.TextFieldView = SC.FieldView.extend(SC.Editable,
 20 /** @scope SC.TextFieldView.prototype */ {
 21   
 22   emptyElement: '<input type="text" value="" />',
 23 
 24   // PROPERTIES
 25 
 26   /**
 27     The hint to display while the field is not active.  Can be a loc key.
 28   */  
 29   hint: null,
 30   
 31   /**
 32     automatically set to YES if the hint is currently showing.
 33   */
 34   isHintShowing: false,
 35 
 36   /**
 37     If YES then the text field is currently editing. 
 38   */
 39   isEditing: NO,
 40 
 41   // PRIVATE SUPPORT METHODS
 42   init: function() {
 43     
 44     // compatibility...
 45     if (this.hint == null) {
 46       this.hint = this.rootElement.getAttribute('hint') ;
 47     }
 48 
 49     if (this.validator == null) {
 50       this.validator = this.rootElement.getAttribute('validate') ;
 51     }
 52 
 53     arguments.callee.base.call(this) ;
 54     
 55     // observe important events for this field.
 56     var focusListener = this._fieldDidFocus.bindAsEventListener(this) ;
 57     Event.observe(this.rootElement, 'focus', focusListener) ;
 58 
 59     var blurListener = this._fieldDidBlur.bindAsEventListener(this) ;
 60     Event.observe(this.rootElement, 'blur', blurListener) ;
 61 
 62     this._updateFieldHint() ;
 63   },
 64 
 65 
 66   // FOCUS AND BLUR EVENTS -
 67   // These should be hooked into the firstResponder system..
 68   _fieldDidFocus: function() {
 69     if (!this._isFocused) {
 70       this._isFocused = true ;
 71       this.becomeFirstResponder() ;
 72     }
 73   },
 74   
 75   _fieldDidBlur: function() {
 76     if (this._isFocused) {
 77       this._isFocused = false ;
 78       this.resignFirstResponder() ;
 79     }
 80   },
 81 
 82   /**
 83     tied to the isEnabled state
 84   */
 85   acceptsFirstResponder: function() {
 86     return this.get('isEnabled');
 87   }.property('isEnabled'),
 88   
 89   // First Responder
 90   // When we become first responder, make sure the field gets focus and
 91   // the hint value is hidden if needed.
 92 
 93   // when we become first responder, focus the text field if needed and
 94   // hide the hint text.
 95   /** @private */
 96   didBecomeFirstResponder: function() {
 97 
 98     // focus the text field.
 99     if (!this._isFocused) {
100       this._isFocused = true ;
101       if (this.get('isVisibleInWindow')) {
102         this.rootElement.focus();
103 		    this.invokeLater(this._selectRootElement, 1) ;
104       }
105     }
106 
107     // hide the hint text if it is showing.
108     this._updateFieldHint() ;
109   },
110 
111   // In IE, you can't modify functions on DOM elements so we need to wrap the 
112   // call to select() like this.
113   _selectRootElement: function() {
114 	this.rootElement.select() ;
115   },
116 
117   // when we lose first responder, blur the text field if needed and show
118   // the hint text if needed.
119   /** @private */
120   willLoseFirstResponder: function() {
121     
122     if (this._isFocused) {
123       this._isFocused = false ;
124       this._updateFieldHint() ;
125       return this.rootElement.blur() ;
126     } else {
127       this._value = this.rootElement.value ;
128       this.fieldValueDidChange() ;
129       this._updateFieldHint() ;
130       return true;
131     }
132   },
133   
134   _isFocused: false,
135   
136   _updateFieldHint: function() {
137     
138     // show the hint if:
139     // - flag is true
140     // - hint != null or empty.
141     // - this._value = null or empty.
142     var hint = this.get('hint') ;
143     var showHint = !!(!this._isFocused && ((this._value == null) || this._value == '') && (hint)) ;
144     
145     this.setClassName('show-hint', showHint);
146     this.rootElement.value = (showHint) ? hint : (this._value || '') ;
147     this.set('isHintShowing', showHint);
148   },
149   
150   // field value updates...
151   /** @private */
152   getFieldValue: function() {
153     return this._value ;
154   },
155   
156   /** @private */
157   setFieldValue: function(value) {
158     if (this._value == value) return ;
159     this._value = value ;
160     this._updateFieldHint() ;
161   },
162   
163   /** @private */
164   mouseDown: function(e)
165   {
166     e._stopWhenHandled = false;
167     return false;
168   },
169   
170   // trap key-press events and notify as needed.
171   /** @private */
172   keyDown: function(evt) {
173     if (this._value != this.rootElement.value) {
174       this._value = this.rootElement.value ;
175       this.fieldValueDidChange(true) ;
176     }
177     
178     return this.interpretKeyEvents(evt); // start bubbling key events...
179     //return false;
180   },
181   
182   /** @private */
183   keyUp: function() {
184     if (this._value != this.rootElement.value) {
185       this._value = this.rootElement.value ;
186       this.fieldValueDidChange(true) ;
187     }
188   },
189 
190   // if make a text field first responder before the view becomes visible,
191   // then focus the text field when it does become visible.
192   _focusOnVisible: function() {
193     if (this.get('isVisibleInWindow') && this._isFocused) {
194       this.rootElement.focus() ;
195       
196       if(SC.isIE()){
197         var selector = function() {
198           Element.select(arguments.callee.it);
199         };
200         selector.it = this.rootElement;
201         setTimeout(selector,0.05);
202       } else{
203         this.rootElement.select.bind(this.rootElement).delay(0.05);  
204       }
205     }
206   }.observes('isVisibleInWindow'),
207   
208   // THESE ARE DUMMY IMPLEMENTATIONS OF THE REPONDER METHODS FOR KEYBOARD
209   // ACTIONS HANDLED BY THE BROWSER.  This avoids having the responder 
210   // bubble up these items.
211 
212   /** @private */
213   deleteBackward: function(evt) { evt._stopWhenHandled = false; return true; },
214 
215   /** @private */
216   deleteForward: function(evt) { evt._stopWhenHandled = false; return true; },
217 
218   /** @private */
219   moveLeft: function(evt) { evt._stopWhenHandled = false; return true; },
220 
221   /** @private */
222   moveRight: function(evt) { evt._stopWhenHandled = false; return true; },
223 
224   /** @private */
225   moveUp: function(evt) { evt._stopWhenHandled = false; return true; },
226 
227   /** @private */
228   moveDown: function(evt) { evt._stopWhenHandled = false; return true; },
229 
230   /** @private */
231   moveLeftAndModifySelection: function(evt) { evt._stopWhenHandled = false; return true; },
232 
233   /** @private */
234   moveRightAndModifySelection: function(evt) { evt._stopWhenHandled = false; return true; },
235 
236   /** @private */
237   moveUpAndModifySelection: function(evt) { evt._stopWhenHandled = false; return true; },
238 
239   /** @private */
240   moveDownAndModifySelection: function(evt) { evt._stopWhenHandled = false; return true; },
241 
242   /** @private */
243   moveToBeginningOfDocument: function(evt) { evt._stopWhenHandled = false; return true; },
244 
245   /** @private */
246   moveToEndOfDocument: function(evt) { evt._stopWhenHandled = false; return true; },
247   
248   /** @private */
249   selectAll: function(evt) { evt._stopWhenHandled = false; return true; }
250   
251 }) ;
