  8 // A relay simply allows you to connect the properties of two objects. 
  9 // Whenever the value of one property changes, the value of the other 
 10 // property will change also.  This allows you to design your system more
 11 // modular.
 12 //
 13 // You can also configure a relay to be one way so only the from -> to path
 14 // changes are permitted. We will also allow transforms and placeholder values 
 15 // eventually.
 16 
 17 SC.Binding = SC.Object.extend({
 18   
 19   // ......................................
 20   // PROPERTIES
 21   // Configure these on setup.  They should be either tuples or property
 22   // paths.  The relay will try to observe both of them.
 23   from: '', to: '',
 24   
 25   // set to true if you don't want changes to -> from to relay.
 26   oneWay: false,
 27   
 28   // set this to some value if you want a placeholder when the source
 29   // value is an empty array or null (if you don't also set a nullPlaceholder)
 30   emptyPlaceholder: null,
 31   
 32   // set this to some value if you want a placeholder value when the source
 33   // value is null.  If you don't set this but you do set an emptyPlaceholder
 34   // then the emptyPlaceholder will be used instead.
 35   nullPlaceholder: null,
 36   
 37   // set this to some value if you want a placeholder when the value is 
 38   // an array with multiple values.  If you set this, arrays with single
 39   // values will be converted to indivdual objects.  If this is set ot null,
 40   // multiple values will be allowed to pass untouched.
 41   multiplePlaceholder: null,
 42   
 43   // set this to a function if you want a transform performed on an input
 44   // value.  This transform is performed both ways and before placeholder
 45   // values are applied.
 46   transform: null,
 47   
 48   // ......................................
 49   // METHODS
 50   
 51   // This will connect the relay so that properties can be forwarded.  If
 52   // the items are already connected, this has no effect.
 53   connect: function() {
 54     if (this._connected) return ;
 55     var funcs = this._boundObservers() ;
 56     
 57     SC.Observers.addObserver(this.get('from'),funcs.from) ;
 58     SC.Observers.addObserver(this.get('to'),funcs.to) ;
 59     this._connected = true ;
 60     return this ;
 61   },
 62   
 63   // this will disconnect the relay from its observing.  Call this if you 
 64   // want to destroy the relay. This returns null so you can use it in
 65   // assignment.
 66   disconnect: function() {  
 67     if (!this._connected) return ;
 68     var funcs = this._boundObservers() ;
 69     SC.Observers.removeObserver(this.get('from'),funcs.from) ;
 70     SC.Observers.removeObserver(this.get('to'),funcs.to) ;
 71     this._connected = false ;
 72     return this ;
 73   },
 74   
 75   // simulate a from -> to relay.
 76   relay: function() { 
 77     var tuple = SC.Object.tupleForPropertyPath(this.get('from')) ;
 78     if (tuple) tuple = this._walkTuple(tuple) ;
 79     if (tuple) this._fromObserver(tuple[0],tuple[1],tuple[0].get(tuple[1]));
 80   },
 81   
 82   // ......................................
 83   // INTERNAL METHODS
 84 
 85   init: function() {
 86     arguments.callee.base.call(this) ;
 87     this.connect() ;
 88   },
 89   
 90   _boundObservers: function() {
 91     var ret = this._boundObserverFuncs ;
 92     if (!ret) {
 93       this._boundObserverFuncs = ret = {
 94         from: this._fromObserver.bind(this),
 95         to: this._toObserver.bind(this)
 96       } ;
 97     }
 98     return ret ;
 99   },
100   
101   _fromObserver: function(target,key,value, propertyRevision) {
102     // to avoid echos, check against last toProperty revision.
103     if (propertyRevision <= this._lastFromPropertyRevision) return ;
104     this._lastFromPropertyRevision = propertyRevision ;
105     
106     // no need to forward values if they haven't actually changed.
107     if (!this._didChange(this._lastFromValue,value)) return ;
108     this._lastFromValue = value ;
109     
110     // try to get the to object.
111     var tuple = SC.Object.tupleForPropertyPath(this.get('to')) ;
112     if (tuple) tuple = this._walkTuple(tuple) ;
113     if (tuple) {
114 
115       // transform the value
116       var transformFunc = this.transform ;
117       if (transformFunc) value = transformFunc('to', key, value) ;
118       this._lastToValue = value ;
119 
120       // apply placeholder settings.
121       var pholder ;
122 
123       // handle empty placeholder.
124       if (value && (value == []) && (pholder = this.get('emptyPlaceholder'))) {
125         value = pholder ;
126 
127       // handle multiple value placeholder
128       } else if (value && (value instanceof Array) && (pholder = this.get('multiplePlaceholder'))) {
129         value = (value.length == 1) ? value[0] : pholder ;
130       }
131 
132       // now handle null placeholder.  This is used if one of these other 
133       // transforms results in a null value.
134       if ((value == null) && (pholder = this.get('nullPlaceholder') || this.get('emptyPlaceholder'))) {
135         value = pholder ;
136       }
137 
138       tuple[0].set(tuple[1],value) ;      
139       this._lastToPropertyRevision = tuple[0].propertyRevision ;
140       
141     }
142   },
143 
144   _toObserver: function(target,key,value, propertyRevision) {
145     // try to get the to object.
146     if (this.get('oneWay')) return ; // block.
147 
148     // to avoid echos, check against last toProperty revision.
149     if (propertyRevision <= this._lastToPropertyRevision) return ;
150     this._lastToPropertyRevision = propertyRevision ;
151     
152     // no need to forward values if they haven't actually changed.
153     if (!this._didChange(this._lastToValue,value)) return ;
154     this._lastToValue = value ;
155     
156     var tuple = SC.Object.tupleForPropertyPath(this.get('from')) ;
157     if (tuple) tuple = this._walkTuple(tuple) ;
158     if (tuple) {
159       // transform the value
160       var transformFunc = this.get('transform') ;
161       if (transformFunc) value = transformFunc('from', key, value) ;
162       this._lastFromValue = value ;
163       
164       // send along to the 'from' source.
165       tuple[0].set(tuple[1],value) ;
166       var result = tuple[0].get(tuple[1]);
167       if (result) this._lastFromPropertyRevision = result.propertyRevision ;
168       
169       // now that it has been set, the FROM object might not allow some 
170       // changes.  If that is the case, then we need to set this back on the
171       // sender.
172       if (result != value) {
173         target.set(key,result) ;
174         this._lastToPropertyRevision = target.propertyRevision ;
175       }
176       
177     } 
178   },
179   
180   _didChange: function(lastValue, newValue) {
181     if (newValue && lastValue) {
182       if (typeof(newValue) == typeof(lastValue)) {
183         if (lastValue == newValue) return false ;
184       }
185     } else if (((newValue === null) && (lastValue === null)) || ((newValue === undefined) && (lastValue === undefined))) return false ;
186     return true ;
187   },
188   
189   _lastToPropertyRevision: 0,
190   _lastFromPropertyRevision: 0,
191   
192   _walkTuple: function(tuple) {
193     var parts = tuple[1].split('.') ;
194     if (parts.length > 1) {
195       tuple = tuple.slice() ; // duplicate to avoid an error.
196       var obj = tuple[0] ;
197       tuple[1] = parts.pop() ;
198       for(var loc=0;(obj && (loc<parts.length));loc++) {
199         obj = obj.get(parts[loc]) ;
200       }
201       tuple[0] = obj ;
202     }
203     return (tuple[0] && tuple[1]) ? tuple : null ;
204   }
205   
206 }) ;

211   NULL_PLACEHOLDER: '@@NULL@@',
212   EMPTY_PLACEHOLDER: '@@EMPTY@@'

215 // This is the basic method you can use to create a builder function for
216 // different types of relays.  The first param is either a string or a set
217 // of properties.  The second param is a set of properties.  Both are opt.
218 SC.Binding.From = function(from,opts) {
219   if (!opts) opts = {} ;
220   if (($type(from) == T_STRING) || ($type(from) == T_ARRAY)) {
221     opts.from = from ;
222   } else Object.extend(opts,from) ;
223   var ret = SC.Binding.extend(opts) ; 
224   return ret ;
225 } ;
226 
227 SC.Binding.build = function(tr) {
228   return function(from) { 
229     return SC.Binding.From(from,{ transform: tr }) ; 
230   } ;
231 } ;

235 // This binding will return errors as null.
236 SC.Binding.NoError = SC.Binding.build(function(dir, key, value) {
237   return ($type(value) == T_ERROR) ? null : value ;  
238 }) ;
239 
240 SC.Binding.NoError.ext = function(bindFunc) {
241   return function(d,k,v) {
242     return ($type(value) == T_ERROR) ? null : bindFunc(d,k,v) ;  
243   } ;
244 } ;

246 // This binding only allows single, null, or error values.  If an array is 
247 // passed, it will be mapped like so:
248 //
249 // [] => null
250 // [x] => x
251 // [x,x,x] => MULTIPLE_PLACEHOLDER
252 //
253 SC.Binding.Single = SC.Binding.build(function(d,k,v) {
254   if ($type(v) == T_ARRAY) {
255     switch(v.length) {
256       case 0:
257         v = null ;
258         break ;
259       case 1:
260         v = v[0] ;
261         break ;
262       default:
263         v = SC.Binding.MULTIPLE_PLACEHOLDER ;
264     }
265   }
266   return v;
267 });
268 
269 // This binding works just like Single except if you pass a multiple value
270 // it will be converted to NULL.
271 SC.Binding.SingleNull = SC.Binding.build(function(d,k,v) {
272   if ($type(v) == T_ARRAY) {
273     switch(v.length) {
274       case 0:
275         v = null ;
276         break ;
277       case 1:
278         v = v[0] ;
279         break ;
280       default:
281         v = null ;
282     }
283   }
284   return v;
285 });

287 // NoError versions.
288 SC.Binding.SingleNoError = SC.Binding.NoError.ext(SC.Binding.Single);
289 SC.Binding.SingleNullNoError = SC.Binding.NoError.ext(SC.Binding.SingleNull);

291 // This requires the value to be a multiple.  null => [], x => [x]
292 SC.Binding.Multiple = SC.Binding.build(function(d,k,v) {
293   var t = $type(v) ;
294   if (t != T_ARRAY) {
295     if (t == null) {
296       v = [] ;
297     } else if (t != T_ERROR) {
298       v = [v] ;
299     }
300   }
301   return v ;
302 }) ;
303 
304 SC.Binding.MultipleNoError = SC.Binding.NoError.ext(SC.Binding.Multiple);
305 
306 // Converts value to a bool.  true if: not null, not empty array, not 0, '',
307 // etc.
308 SC.Binding.Bool = SC.Binding.build(function(d,k,v) {
309   return ($type(v) == T_ARRAY) ? (v.length > 0) : !!v ;
310 }) ;
311 
312 // Converts value to a bool, but its only true if not null
313 SC.Binding.NotNull = SC.Binding.build(function(d,k,v) { 
314   return (v != null) ;
315 }) ;
316 
317 // Converts inverse of bool.
318 SC.Binding.Not = SC.Binding.build(function(d,k,v) {
319   return !(($type(v) == T_ARRAY) ? (v.length > 0) : !!v) ;
320 }) ;
321 
322 // Converts value to a bool, but its only true if not null
323 SC.Binding.IsNull = SC.Binding.build(function(d,k,v) {
324   return (v == null) ;
325 }) ;

336 // This relay forces all values to be multiple values
337 SC.Binding.Multiple = function(from) {
338   return SC.Binding.From(from,{
339     transform: function(dir,key,value) { 
340       return (value) ? (SC.isArray(value) ? value : [value]) : value;
341     }
342   }) ;
343 } ;
344 
345 // This relay forces all values to be multiple values, null values are not
346 // allowed.
347 SC.Binding.MultipleNotEmpty = function(from) {
348   return SC.Binding.From(from,{
349     transform: function(dir,key,value) { 
350       return (value) ? (SC.isArray(value) ? value : [value]) : [];
351     }
352   }) ;
353 } ;

355 // This relay allows single values and no null/empty values.
356 SC.Binding.SingleNotEmpty = function(from) {
357   return SC.Binding.From(from,{
358     multiplePlaceholder: SC.Binding.MULTIPLE_PLACEHOLDER,
359     emptyPlaceholder: SC.Binding.EMPTY_PLACEHOLDER,
360     nullPlaceholder: SC.Binding.NULL_PLACEHOLDER
361   }) ;
362 } ;

364 // This relay allows any values, but only one way flow.
365 SC.Binding.OneWay = function(from) {
366   return SC.Binding.From(from, { oneWay: true }) ;
367 } ;

369 // This relay forces the value to be true or false.  Empty and null values
370 // represent false.
371 SC.Binding.Flag = function(from) {
372   return SC.Binding.From(from, { 
373     transform: function(dir,key,value) {
374       return (value && (value instanceof Array)) ? (value.length == 0) : !!value ;
375     }
376   }) ;
377 } ;
378 
379 SC.Binding.OneWayFlag = function(from) {
380   var ret = SC.Binding.Flag(from) ;
381   ret.oneWay = true ;
382   return ret ;  
383 } ;
