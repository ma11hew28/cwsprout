  9 /** @class
 10 
 11   An InputManager knows how to convert incoming keyboard events and convert
 12   them into actions on a responder.  The default version should provide the
 13   correct behavior for most people.  However, if you want some special
 14   behavior, you can always write your own.
 15 
 16   An instance of this input manager is created as a property of
 17   SC.Responder.  To overide the inputManager used for all responders, replace
 18   this property with your own instance.  To override the inputManager used 
 19   only for one responder, set the inputManager property on your specific 
 20   instance.
 21 
 22   Note that generally you do not want to write your own inputManager.  They
 23   are tricky to get right.  Instead, you should implement the various handler
 24   methods on the responder.
 25   
 26   @extends SC.Object
 27   @author Charles Jolley
 28 */
 29 SC.InputManager = SC.Object.extend(
 30 /** @scope SC.InputManager.prototype */ {
 31   
 32   /**
 33     The is the primary entry point for the inputManager.  If you override
 34     the input manager, have this method process the key event and invoke
 35     methods on the responder. 
 36   */
 37   interpretKeyEvents: function(event, responder)
 38   {
 39     var codes = this.codesForEvent(event) ;
 40     var cmd = codes[0]; var chr = codes[1];
 41     if (!cmd && !chr) return false ;  //nothing to do.
 42 
 43     // if this is a command key, try to do something about it.
 44     if (cmd)
 45     {
 46       var methodName = SC.MODIFIED_COMMAND_MAP[cmd] || SC.BASE_COMMAND_MAP[cmd.split('_').last()];
 47       if (methodName && responder.respondsTo(methodName))
 48       {
 49         // the responder has a method matching the keybinding... call it.
 50         return responder[methodName]( event );
 51       }
 52     } 
 53     // if(cmd == "space") chr = " ";
 54     if ( chr && responder.respondsTo('insertText'))
 55     {
 56       // if we haven't returned yet and there is plain text, then do an insert of the text.
 57       return responder.insertText(chr);
 58     }
 59 
 60     return false ; //nothing to do.
 61   },
 62   
 63   /**
 64     this will get a standardized command code for the event.  It returns
 65     null if the event is plain text, not a command code.
 66   */  
 67   codesForEvent: function(e) {
 68     var code = e.keyCode ;
 69     var ret = null ; var key = null ;
 70     var modifiers = '' ;
 71     
 72     // handle function keys.
 73     if (code) {
 74       ret = SC.FUNCTION_KEYS[code] ;
 75       if (!ret && (e.altKey || e.ctrlKey)) ret = SC.PRINTABLE_KEYS[code] ;
 76       if (ret) {
 77         if (e.altKey) modifiers += 'alt_' ;
 78         if (e.ctrlKey) modifiers += 'ctrl_' ;
 79         if (e.shiftKey) modifiers += 'shift_' ;
 80       }
 81     }
 82 
 83     // otherwise just go get the right key.
 84     if (!ret) {
 85       var code = e.charCode || e.keyCode ;
 86       key = ret = String.fromCharCode(code) ;
 87       var lowercase = ret.toLowerCase() ;
 88       if (ret != lowercase) {
 89         modifiers = 'shift_' ;
 90         ret = lowercase ;
 91       } else ret = null ;
 92     }
 93 
 94     if (ret) ret = modifiers + ret ;
 95     return [ret, key] ;
 96   }
 97     
 98 });

152 // make reverse keycode lookup for using in unit tests...
153 SC.KEY_CODES = {};
154 for (var i=0, n=256; i < n; i++)
155 {
156   if (SC.MODIFIER_KEYS[i] !== undefined) {
157     SC.KEY_CODES[ SC.MODIFIER_KEYS[i] ] = i;
158   } else if (SC.FUNCTION_KEYS[i] !== undefined) {
159     SC.KEY_CODES[ SC.FUNCTION_KEYS[i] ] = i;

161     SC.KEY_CODES[ SC.PRINTABLE_KEYS[i] ] = i;
162   }
