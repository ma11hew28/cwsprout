  8 /**
  9   @class
 10 
 11   The Store is where you can find all of your records.  You should also
 12   use this to define your various types of records, since this will be
 13   used to automatically update from data coming from the server.
 14 
 15   You should create a store for each application.  This allows the records
 16   for apps to be kept separate, even if they live in the same page.
 17 
 18   @extends SC.Object
 19   @static
 20   @since SproutCore 1.0
 21 */
 22 SC.Store = SC.Object.create(
 23 /** @scope SC.Store.prototype */ {
 24   
 25   /**
 26     Pushes updated data to all the named records.  
 27   
 28     This method is often called from a server to update the store with the 
 29     included record objects.
 30   
 31     You can use this method yourself to mass update the store whenever you 
 32     retrieve new records from the server.  The first parameter should contain
 33     an array of JSON-compatible hashes.  The hashes can have any properties 
 34     you want but they should at least contain the following two keys:
 35   
 36     - guid: This is a unique identifier for the record. 
 37     - type: The name of the record type.  I.e. "Contact" or "Photo"
 38   
 39     @param dataHashes {Array} array of hash records.  See discussion.
 40     @param dataSource {Object} the data source.  Usually a server object.
 41     @param recordType {SC.Record} optional record type, used if type is not 
 42       found in the data hashes itself.
 43     @param isLoaded {Boolean} YES if the data hashes represent the full set of 
 44       data loaded from the server.  NO otherwise.
 45 
 46     @returns {Array} Array of records that were actually created/updated.
 47   */
 48   updateRecords: function(dataHashes, dataSource, recordType, isLoaded) {
 49     
 50     this.set('updateRecordsInProgress',true) ;
 51     var store = this ; 
 52     var ret = [] ;
 53     if (!recordType) recordType = SC.Record ;
 54     this.beginPropertyChanges() ;
 55 
 56     //SC.Benchmark._bench(function() { 
 57     dataHashes.each(function(data) {
 58       var rt = data.recordType || recordType; 
 59       if (data.recordType !== undefined) delete data.recordType;
 60       
 61       var pkValue = data[rt.primaryKey()] ;      
 62       var rec = store.getRecordFor(pkValue,rt,true) ;
 63       rec.dataSource = dataSource ;
 64       if (data.refreshURL) rec.refreshURL = data.refreshURL;
 65       if (data.updateURL) rec.updateURL = data.updateURL;
 66       if (data.destroyURL) rec.destroyURL = data.destroyURL;
 67       rec.updateAttributes(data, isLoaded, isLoaded) ;
 68       if (rec.needsAddToStore) store.addRecord(rec) ;
 69       ret.push(rec) ;
 70     });
 71 //    },'dataHashes') ;  
 72 
 73     this.endPropertyChanges() ;
 74     this.set('updateRecordsInProgress',false) ;
 75     return ret ;
 76   },

 83   
 84   createRecords: function(recs) {
 85     recs.invoke('set','newRecord','false') ;
 86     this.commitRecords(recs); 
 87   },
 88   
 89   commitRecords: function(recs) { recs.invoke('set','isDirty','false'); },

 91   destroyRecords: function(recs) {
 92     var store = this ;
 93     recs.each(function(r) { 
 94       r.set('isDeleted',true); store.removeRecord(r);
 95     });
 96   },

102   /**
103     Add a record instance to the store.  The record will now be monitored for
104     changes.
105   */
106   addRecord: function(rec) {
107     // save record in a cache
108     rec.needsAddToStore = false;
109     var guid = rec._storeKey();
110     var records = this._records[guid] || [];
111     records.push(rec);
112     this._records[guid] = records; 
113 
114     // global record cache
115     if (!this._quickCache) this._quickCache = {};
116 
117     // records are cached by Class type
118     var records = this._quickCache[guid] || {};
119     var pkey = rec.get(rec.primaryKey);
120     records[pkey] = rec;
121     this._quickCache[guid] = records;
122 
123     // and start observing it.
124     rec.addObserver('*',this._boundRecordObserver) ;
125     this.recordDidChange(rec) ;
126   },
127 
128   /**
129     remove a record instance from the store.  The record will no longer be
130     monitored for changes and may be deleted.
131   */  
132   removeRecord: function(rec) {
133     // remove from cache
134     var guid = rec._storeKey();
135     var records = this._records[guid] || [];
136     records = records.without(rec);
137     this._records[guid] = records;
138     
139     // remove from quick cache
140     if (this._quickCache)
141     {
142       var records = this._quickCache[guid] || {};
143       var pkey = rec.get(rec.primaryKey);
144       delete records[pkey];
145       this._quickCache[guid] = records;
146     }
147 
148     // and stop observing it.
149     rec.removeObserver('*',this._boundRecordObserver) ;
150     this.recordDidChange(rec) ; // this will remove from cols since destroyed.
151   },
152 
153   /**
154     Since records are cached by primaryKey, whenever that key changes we need 
155     to re-cache it in the proper place
156     
157     @param {string} oldkey Previous primary key
158     @param {string} newkey New primary key
159     @param {SC.Record} rec The object to relocate
160     @returns {SC.Record} The record passed in
161   **/
162   relocateRecord: function( oldkey, newkey, rec )
163   {
164     if (!this._quickCache) return rec;
165     
166     var classKey = rec._storeKey();
167     var records  = this._quickCache[classKey] || {};
168 
169     records[newkey] = rec;
170     delete records[oldkey];
171     
172     this._quickCache[classKey] = records;
173     
174     return rec;
175   },
176 
177 
178   /**
179     You can pass any number of condition hashes to this, ending with a
180     recordType.  It will AND the results of each condition hash.
181   */  
182   findRecords: function() {
183     var allConditions = SC.$A(arguments) ;
184     var recordType = allConditions.pop() ;
185     var guid = recordType._storeKey() ;
186 
187     // initial set of records.
188     var records = this._records[guid] ;
189     
190     while(allConditions.length > 0) {
191       var conditions = allConditions.pop() ;
192       var ret = [] ; var loc = (records) ? records.length : 0;
193       while(--loc >= 0) {
194         var rec = records[loc] ;
195         if ((rec._type == recordType) || (rec._type.coreRecordType == recordType)) {
196           if (rec.matchConditions(conditions)) ret.push(rec) ;
197         }
198       }
199       records = ret ;
200     }
201 
202     // clone records...
203     return SC.$A(records) ;
204   },
205   
206   // private method used by Record and Store. Returns null if the record does not exist.
207   _getRecordFor: function(pkValue,recordType) {
208     var guid = recordType._storeKey() ;
209     var records = (this._quickCache) ? this._quickCache[guid] : null;
210     var ret = (records) ? records[pkValue] : null ;
211     return ret ;
212   },
213   
214   /**
215     finds the record with the primary key value.  If the record does not 
216     exist, creates it.
217   */
218   getRecordFor: function(pkValue,recordType,dontAutoaddRecord) {
219     var ret = this._getRecordFor(pkValue,recordType) ;
220     if (!ret) {
221       var opts = {}; opts[recordType.primaryKey()] = pkValue;
222       ret = recordType.create(opts) ;
223       if (dontAutoaddRecord) {
224         ret.needsAddToStore = true ;
225       } else this.addRecord(ret) ;
226     }
227     return ret ;
228   },
229   
230   /**
231     Returns an array of all records in the store.  Mostly used for storing.
232   */
233   records: function() {
234     var ret = [] ;
235     if (this._quickCache) {
236       for(var key in this._quickCache) {
237         var recs = this._quickCache[key] ;
238         for(var recKey in recs) {
239           ret.push(recs[recKey]) ;
240         }
241       }
242     }
243     return ret ;
244   }.property(),

250   addCollection: function(collection) {
251     var guid = collection.recordType._storeKey() ;
252     var collections = this._collections[guid] || [] ;
253     collections.push(collection) ;
254     this._collections[guid] = collections ;
255   },
256   
257   removeCollection: function(collection) {
258     var guid = collection.recordType._storeKey() ;
259     var collections = this._collections[guid] || [] ;
260     collections = collections.without(collection) ;
261     this._collections[guid] = collections ;
262   },
263   
264   listFor: function(opts) {
265     var conditions = opts.conditions || {} ;
266     var order = opts.order || ['guid'] ;
267     var records = this.findRecords(conditions,opts.recordType) ;
268     var count = records.length ;
269 
270     // sort
271     records = records.sort(function(a,b) { return a.compareTo(b,order); }) ;
272     
273     // slice if needed.
274     if (opts.limit && (opts.limit > 0)) {
275       var start = (opts.offset) ? opts.offset : 0 ;
276       var end = start + opts.limit ;
277       records = records.slice(start,end) ;      
278     }
279     
280     // now run callback.
281     if (opts.callback) opts.callback(records,count) ; 
282   },

285   // PRIVATE

287   _records: {}, _changedRecords: null, _collections: {},
288   
289   /** @private
290     called whenever properties on a record change.
291   */  
292   recordDidChange: function(rec) {
293     // add to changed records.  This will eventually notify collections.
294     var guid = rec._storeKey() ;
295     changed = this.get('_changedRecords') || {};
296     records = changed[guid] || {} ;
297     records[rec._guid] = rec ;
298 
299     changed[guid] = records ;    
300     this.set('_changedRecords',changed) ;
301   },
302   
303   // invoked whenever the changedRecords hash is updated. This will notify
304   // collections.
305   _changedRecordsObserver: function() { 
306     // process changedRecords to notify collections.
307     for(var guid in this._changedRecords) {
308       var collections = this._collections[guid] ;
309       if (collections && collections.length>0) {
310         
311         // collect records into array.  Faster than using uniq.
312         var records = [] ;
313         for(var key in this._changedRecords[guid]) {
314           records.push(this._changedRecords[guid][key]) ;
315         }
316         var cloc = collections.length ;
317         while(--cloc >= 0) {
318           
319           // for each collection watching this type of record, notify.
320           var col = collections[cloc] ;
321           col.beginPropertyChanges() ;
322           try {
323             // for each record...
324             var rloc = records.length ;
325             while(--rloc >= 0) {
326               var record = records[rloc] ;
327               // notify only if record type matches.
328               if (col.recordType == record._type) {
329                 col.recordDidChange(record) ;
330               }
331             }
332           }
333           catch (e) {
334             console.log('EXCEPTION: While notifying collection') ;
335           }
336           col.endPropertyChanges() ;
337           
338         }
339       }
340     }
341     
342     // then clear changed records to start again.
343     this._changedRecords = {} ;
344     
345   }.observes('_changedRecords'),
346   
347   init: function() {
348     arguments.callee.base.call(this) ;
349     this._boundRecordObserver = this.recordDidChange.bind(this) ;
350   }
