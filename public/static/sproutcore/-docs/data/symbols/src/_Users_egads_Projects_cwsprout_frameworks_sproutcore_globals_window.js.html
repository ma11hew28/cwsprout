 12 // The window global object is automatically setup for each window that you 
 13 // load.  Window listens for mouse and keyboard events and routes them to the
 14 // first responder.  Using the firstResponder method you can control who gets
 15 // to respond to keyboard events.

 18   // This finds the first view responding to the event.
 19   firstViewForEvent: function(evt) {
 20     var el = Event.element(evt) ;
 21     while(el && (el != document) && (!el._configured)) el = el.parentNode ;
 22     if (el) el = SC.View.findViewForElement(el) ;
 23     if (el == this) el = null ;
 24   	return el ;
 25   },

 51     if (!this._size) {
 52       if (window.innerHeight) {
 53         this._size = { 
 54           width: window.innerWidth, 
 55           height: window.innerHeight
 56         };
 57       } else if (document.documentElement && document.documentElement.clientHeight) {
 58         this._size = { 
 59           width: document.documentElement.clientWidth, 
 60           height: document.documentElement.clientHeight
 61         };
 62       } else if (document.body) {
 63         this._size = { 
 64           width: document.body.clientWidth, 
 65           height: document.body.clientHeight
 66         };
 67       }
 68     }
 69     
 70     return this._size ;

 92   _lastModifiers: null,
 93   _handleModifierChanges: function(evt)
 94   {
 95     //console.debug('_lastModifiers: %o', this._lastModifiers);
 96     // if the modifier keys have changed, then notify the first responder.
 97     var m = this._lastModifiers = this._lastModifiers || { alt: false, ctrl: false, shift: false };
 98 
 99     var changed = false;
100     if (evt.altKey != m.alt) { m.alt = evt.altKey; changed=true; }
101     if (evt.ctrlKey != m.ctrl) { m.ctrl = evt.ctrlKey; changed=true; }
102     if (evt.shiftKey != m.shift) { m.shift = evt.shiftKey; changed=true;}
103 
104     if (changed) 
105     {
106       evt._type = 'flagsChanged';
107       evt._modifiers = m;
108       SC.app.sendEvent( evt );
109     }
110   },
111   
112   // this one gets called once when the key is down.  We use this to handle
113   // function and non-printable keys. (i.e. if a modifier key is pressed or
114   // we can map to some non-printable set of keycodes.)
115   _onkeydown: function(evt)
116   {
117     // Firefox does NOT handle delete here...
118     if (SC.Platform.Firefox > 0 && (evt.which === 8)) {
119       return true ;
120     }
121     
122     // modifier keys are handled separately by the 'flagsChanged' event
123     this._handleModifierChanges(evt);
124     if (this._isModifierKey(evt)) return false;
125     
126     // let normal browser processing do its thing.
127     if (!this._isFunctionOrNonPrintableKey(evt)) return true;  
128     var ret = this._sendEvent('keyDown', evt);
129     return ret ;
130   },
131 
132   // this one gets used for all the other keys not handled by key down.
133   // FireFox also needs to handle delete here...
134   _onkeypress: function(evt)
135   {
136 
137     // handled in _onkeydown
138     if (SC.Platform.Firefox > 0 && (evt.which === 8)) {
139       var ret = this._sendEvent('keyDown', evt);
140 
141     } else {
142       if (this._isFunctionOrNonPrintableKey(evt)) return true; 
143       if (evt.charCode != undefined && evt.charCode == 0) return true;
144       var ret = this._sendEvent('keyDown', evt);
145     }
146     return ret ;
147   },
148   
149   _onkeyup: function(evt)
150   {
151     // modifier keys are handled separately by the 'flagsChanged' event
152     this._handleModifierChanges(evt);
153     if (this._isModifierKey(evt)) return;
154     return this._sendEvent('keyUp', evt);
155   },
156 
157   _sendEvent: function( sctype, evt )
158   {
159     SC.runLoop.beginRunLoop();
160     
161     evt._type = sctype;
162     evt._stopWhenHandled = (evt._stopWhenHandled !== undefined) ? evt._stopWhenHandled : true;
163     
164     var handler = SC.app.sendEvent( evt );
165     
166     var ret = true;
167     if (handler && evt._stopWhenHandled) {
168       Event.stop(evt);
169       ret = false;
170     }
171     
172     SC.runLoop.endRunLoop();
173     return ret ; 
174   },
175 
176   // util code factored out of keypress and keydown handlers
177   _isFunctionOrNonPrintableKey: function( evt )
178   {
179     return !!(evt.altKey || evt.ctrlKey || SC.FUNCTION_KEYS[evt.keyCode]);
180   },
181   _isModifierKey: function( evt )
182   {
183     return !!SC.MODIFIER_KEYS[evt.keyCode];
184   },

189   // Mouse Handling
190   //
191   // Mouse clicks are routed to the target view.
192   //
193   _mouseDownView: null,
194   _clickCount: 0,
195   _lastMouseUpAt: null,
196   
197   dragDidStart: function(drag) {
198     this._mouseDownView = drag ;  
199   },
200   
201   _onmousedown: function(evt)
202   {
203     SC.runLoop.beginRunLoop();
204 
205     // make sure the view gets focus no matter what.  FF is inconsistant 
206     // about this.
207     this._onfocus(); 
208     
209     // first, save the click count.  Click count resets if your down is
210     // more than 125msec after you last click up.
211     this._clickCount = this._clickCount + 1 ;
212     if (!this._lastMouseUpAt || ((Date.now() - this._lastMouseUpAt) > 200)) {
213       this._clickCount = 1 ; 
214     }
215     evt.clickCount = this._clickCount ;
216     
217     evt._type = 'mouseDown';
218     evt._stopWhenHandled = (evt._stopWhenHandled !== undefined) ? evt._stopWhenHandled : true;
219 
220     this._mouseDownView = SC.app.sendEvent( evt );
221     var ret = true ;
222     if (this._mouseDownView && evt._stopWhenHandled)
223     {
224       Event.stop(evt);
225       ret = false ;
226       if(this._mouseDownView.mouseDragged && $type(this._mouseDownView.mouseDragged) == T_FUNCTION) {
227         //IE7: if the mouseDownView handles the mouseDragged event, set mouseCanDrag
228         //to true.  TODO: make an optional "allowBrowserSelect" property that
229         // would do the same thing; for now let's just prevent it in this case
230         this._mouseCanDrag = true;
231       }
232     }
233 
234     SC.runLoop.endRunLoop();
235     return ret ;
236   },
237   
238   // mouseUp only gets delivered to the view that handled the mouseDown evt.
239   // we also handle click and double click notifications through here to 
240   // ensure consistant delivery.  Note that if mouseDownView is not
241   // implemented, then no mouseUp event will be sent, but a click will be 
242   // send.
243   _onmouseup: function(evt)
244   {
245     SC.runLoop.beginRunLoop();
246     var handler = null;
247 
248     this._lastMouseUpAt = Date.now();
249 
250     // record click count.
251     evt.clickCount = this._clickCount ;
252     
253     // attempt the mouseup call only if there's a target.
254     // don't want a mouseup going to anyone unless they handled the mousedown...
255     if (this._mouseDownView)
256     {
257       evt._type = 'mouseUp';
258       handler = SC.app.sendEvent(evt, this._mouseDownView);
259     }
260     // no one handled the mouseup... try doubleclick...
261     if (!handler && (this._clickCount == 2))
262     {
263       evt._type = 'doubleClick';
264       handler = SC.app.sendEvent(evt, this._mouseDownView);
265     }
266     // no one handled the doubleclick... try click...
267     if (!handler)
268     {
269       evt._type = 'click';
270       handler = SC.app.sendEvent(evt, this._mouseDownView);
271     }
272     this._mouseCanDrag = false;
273     this._mouseDownView = null;
274     SC.runLoop.endRunLoop() ;
275   },
276 
277   _ondblclick: function(evt){
278     if(SC.isIE())
279     {
280       this._clickCount = 2;
281       this._onmouseup(evt);
282     }
283   },

285   _lastHovered: null,
286 
287   // this will sent mouseOver, mouseOut, and mouseMoved to the views you
288   // hover over.  To receive these events, you must implement the method.
289   // If any subviews implement them and return true, then you won't receive
290   // any notices.
291   //
292   // if there is a target mouseDown view, then mouse moved events will also
293   // trigger calls to mouseDragged.
294   //
295   _onmousemove: function(evt) {
296 
297     SC.runLoop.beginRunLoop();
298 
299     // make sure the view gets focus no matter what.  FF is inconsistant 
300     // about this.
301     this._onfocus(); 
302     
303 
304     var lh = this._lastHovered || [] ;
305     var nh = [] ;
306     var view = this.firstViewForEvent(evt) ;
307     
308     // work up the view chain.  Notify of mouse entered and
309     // mouseMoved if implemented.
310     while(view && (view != this)) {
311       var entered = view.mouseOver || view.didMouseOver || view.mouseEntered;
312       var moved = view.mouseMoved || view.mouseDidMove  ;
313 
314       if (lh.include(view)) {
315         if (moved) moved.call(view,evt) ;
316         nh.push(view) ;
317       } else {
318         if (entered) entered.call(view,evt) ;
319         nh.push(view) ;
320       }
321       
322       view = view.get('nextResponder');
323     }
324 
325     // now find those views last hovered over that were no longer found 
326     // in this chain and notify of mouseExited.
327     for(var loc=0; loc < lh.length; loc++) {
328       view = lh[loc] ;
329       var exited = view.mouseOut || view.didMouseOut || view.mouseExited ;
330       if (exited && !nh.include(view)) exited.call(view, evt) ;
331     }
332     
333     this._lastHovered = nh; 
334     
335     // also, if a mouseDownView exists, call the mouseDragged action, if it 
336     // exists.
337     if (this._mouseDownView && this._mouseDownView.mouseDragged) {
338       this._mouseDownView.mouseDragged(evt) ;
339     }
340     
341     SC.runLoop.endRunLoop();
342   },

367   /** @private */
368   // these methods are used to prevent unnecessary text-selection in IE,

389   _EVTS: ['mousedown', 'mouseup', 'click', 'dblclick', 'keydown', 'keyup', 'keypress', 'mouseover', 'mouseout', 'mousemove', 'resize', 'unload', 'focus', 'blur','drag','selectstart'],
390 
391   _listenerCache: [],
392   
393   setup: function()
394   {
395     // setup event listeners for window.
396     var win = this ;
397     win._EVTS.each(function(e)
398     {
399       var func = win['_on' + e] ;
400       var target = (e != 'resize') ? document : window ;
401       if (func)
402       {
403         var f = func.bindAsEventListener(win) ;
404         if (e === 'keypress' && SC.CAPTURE_BACKSPACE_KEY && SC.Platform.Firefox > 0)
405         {
406           document.onkeypress = f ;
407         }
408         else if (e === 'selectstart' && SC.Platform.IE > 0)
409         {
410           //capture onselectstart events in IE (proprietary)
411           document.body.onselectstart = f;
412         }
413         else if (e === 'drag' && SC.Platform.IE > 0)
414         {
415           document.body.ondrag = f;
416         }
417         else
418         {
419           Event.observe(target, e, f) ;
420         }
421         win._listenerCache.push([target, e, f]) ;
422       }
423     });
424     this.get('size') ;
425     // fetch the size from the window and save it.
426     this.set('isVisibleInWindow', true) ;
427     this._onfocus() ;
428   }
429   }).viewFor($tag('body')) ;

432 // events:
433 // window.onfocus --
434 //  notify all views that they now have window focus.
435 
436 // window.onblur --
437 //  notify all views that they no longer have window focus.
438 
439 // element.onfocus/element.onblur --
440 //  when a field gains focus, it should become first responder.  Other
441 //  elements should become first responder when you click on them.
442 
443 // window.onresize --
444 //  call the resize code on the views to tell them their parent views resized.
445 //  -- we should have a root view that is the window.
446 
447 // window.onscroll --
448 //  call the scroll code on the root view.
449 
450 // element.onscroll --
451 //  call the scroll code on the element.
452 
453 // window.onunload --
454 //  call this on all views? to let them unload....
455 
456 // document.onclick --
457 // document.ondblclick --
458 // these events are ignored (actually, they are eaten) by the window object.
459 // click and double click events are handled by the mouseUp handler so that
460 // we can avoid calling click if mouseUp is handled by someone.
461 
462 // document.onmousedown --
463 // document.onmouseup --
464 //  find the target element and work your way up the DOM hierarchy until you
465 //  find Views to handle the click.
466 
467 // document.onmousemove --
468 // document.onmouseover --
469 // document.onmouseout --
470 //  use these events to simulate mouseover and mouseout on views.  Find the 
471 //  target element and work you way up the hierarchy until you find a view
472 //  that responds to these events.
473 //
474 //  also we should have some generic handlers in the window object that you
475 //  can use to be notified when the mouse enters and leaves the window in
476 //  general.
477 
478 // formelement.onchange
479 //   use to trigger a value change on the form field element.
480 
481 // form.onreset 
482 //   um, do nothing?  button will handle action directly.
483 
484 // form.onsubmit
485 //   disable so that AJAX code can takeover.
486 
487 // document.onkeydown --
488 // document.onkeypress --
489 // document.onkeyup --
490 //  notify the firstResponder and work up the chain until someone handles the
491 //  keypress.  You can have a defaultResponder, which gets the keypress if 
492 //  no one else does.  First normalize key info.
493 
494 // img.onerror --
495 //  this should call some handler function in the ImageView.
