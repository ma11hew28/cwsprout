  6 // A menu is not a view exactly, but you can use it to bundle together 
  7 // buttons and then validate them as a whole.
  8 SC.Toolbar = SC.View.extend({
  9   
 10   // override with an array of outlet functions.
 11   buttons: [],
 12   
 13   // to to false an all the buttons will be disabled also.
 14   isEnabled: true,
 15   
 16   init: function() {
 17     arguments.callee.base.call(this) ;
 18     var toolbar = this ;
 19     this.buttons = this.buttons.map(function(button) {
 20       return button(toolbar) ; // get outlet.
 21     }) ;
 22   },
 23   
 24   isEnabledObserver: function() {
 25     var e = this.get('isEnabled') ;
 26     this.get('buttons').each(function(button) { button.set('isEnabled',e); }) ;
 27   }.observes('isEnabled')
 28   
 29 }) ;
