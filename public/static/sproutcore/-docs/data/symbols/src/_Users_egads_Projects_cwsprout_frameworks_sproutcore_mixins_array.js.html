  8 SC.OUT_OF_RANGE_EXCEPTION = "Index out of range" ;

 10 /**
 11   @namespace
 12   
 13   This module implements Observer-friendly Array-like behavior.  This mixin is 
 14   picked up by the Array class as well as other controllers, etc. that want to  
 15   appear to be arrays.
 16   
 17   Unlike SC.Enumerable, this mixin defines methods specifically for 
 18   collections that provide index-ordered access to their contents.  When you
 19   are designing code that needs to accept any kind of Array-like object, you
 20   should use these methods instead of Array primitives because these will 
 21   properly notify observers of changes to the array. 
 22   
 23   Although these methods are efficient, they do add a layer of indirection to
 24   your application so it is a good idea to use them only when you need the 
 25   flexibility of using both true JavaScript arrays and "virtual" arrays such
 26   as controllers and collections.
 27   
 28   You can use the methods defined in this module to access and modify array 
 29   contents in a KVO-friendly way.  You can also be notified whenever the 
 30   membership if an array changes by observing the "[]" property.
 31 
 32   To support SC.Array in your own class, you must override two
 33   primitives to use it: replace() and objectAt().  
 34 
 35   Note that the SC.Array mixin also incorporates the SC.Enumerable mixin.  All
 36   SC.Array-like objects are also enumerable.
 37 
 38   @extends SC.Enumerable
 39   @since SproutCore 0.9.0
 40 */
 41 SC.Array = {
 42 
 43 /**
 44   @field {Number} length
 45   
 46   Your array must support the length property.  your replace methods should
 47   set this property whenever it changes.
 48 */
 49   // length: 0,
 50   
 51 /**
 52   This is one of the primitves you must implement to support SC.Array.  You 
 53   should replace amt objects started at idx with the objects in the passed 
 54   array.  You should also call this.enumerableContentDidChange() ;
 55   
 56   @param {Number} idx 
 57     Starting index in the array to replace.  If idx >= length, then append to 
 58     the end of the array.
 59 
 60   @param {Number} amt 
 61     Number of elements that should be removed from the array, starting at 
 62     *idx*.
 63 
 64   @param {Array} objects 
 65     An array of zero or more objects that should be inserted into the array at 
 66     *idx* 
 67 */
 68   replace: function(idx, amt, objects) {
 69     throw "replace() must be implemented to support SC.Array" ;
 70   },
 71 
 72 /**
 73   This is one of the primitives you must implement to support SC.Array.  
 74   Returns the object at the named index.  If your object supports retrieving 
 75   the value of an array item using get() (i.e. myArray.get(0)), then you do
 76   not need to implement this method yourself.
 77   
 78   @param {Number} idx
 79     The index of the item to return.  If idx exceeds the current length, 
 80     return null.
 81 */
 82   objectAt: function(idx)
 83   {
 84     if (idx < 0) return undefined ;
 85     if (idx >= this.get('length')) return undefined;
 86     return this.get(idx);
 87   },
 88   
 89   /**
 90     @field []
 91 
 92     This is the handler for the special array content property.  If you get
 93     this property, it will return this.  If you set this property it a new 
 94     array, it will replace the current content.
 95     
 96     This property overrides the default property defined in SC.Enumerable.
 97   */
 98   '[]': function(key, value) {
 99     if (value !== undefined) {
100       this.replace(0, this.get('length'), value) ;
101     }  
102     return this ;
103   }.property(),
104   
105 /**  
106   This will use the primitive replace() method to insert an object at the 
107   specified index.
108   
109   @param {Number} idx index of insert the object at.
110   @param {Object} object object to insert
111 */
112   insertAt: function(idx, object) {
113     if (idx > this.get('length')) throw SC.OUT_OF_RANGE_EXCEPTION ;
114     this.replace(idx,0,[object]) ;
115     return this ;
116   },
117   
118   /**
119     Remove an object at the specified index using the replace() primitive method.
120   
121     @param {Number} idx index of object to remove
122   */
123   removeAt: function(idx) {
124     if ((idx < 0) || (idx >= this.get('length'))) throw SC.OUT_OF_RANGE_EXCEPTION;
125     var ret = this.objectAt(idx) ;
126     this.replace(idx,1,[]);
127     return ret ;
128   },
129   
130   /**
131     Search the array of this object, removing any occurrences of it.
132     @param {object} obj object to remove
133   */
134   removeObject: function(obj) {
135     var loc = this.get('length') || 0;
136     while(--loc >= 0) {
137       var curObject = this.objectAt(loc) ;
138       if (curObject == obj) this.removeAt(loc) ;
139     }
140     return this ;
141   },
142   
143   /**
144     Push the object onto the end of the array.  Works just like push() but it 
145     is KVO-compliant.
146   */
147   pushObject: function(obj) {
148     this.insertAt(this.get('length'), obj) ;
149     return obj ;
150   },
151   
152   /**
153     Pop object from array or nil if none are left.  Works just like pop() but 
154     it is KVO-compliant.
155   */
156   popObject: function() {
157     var len = this.get('length') ;
158     if (len == 0) return null ;
159     
160     var ret = this.objectAt(len-1) ;
161     this.removeAt(len-1) ;
162     return ret ;
163   },
164   
165   /**
166     Shift an object from start of array or nil if none are left.  Works just 
167     like shift() but it is KVO-compliant.
168   */
169   shiftObject: function() {
170     if (this.get('length') == 0) return null ;
171     var ret = this.objectAt(0) ;
172     this.removeAt(0) ;
173     return ret ;
174   },
175   
176   /**
177     Unshift an object to start of array.  Works just like unshift() but it is 
178     KVO-compliant.
179   */
180   unshiftObject: function(obj) {
181     this.insertAt(0, obj) ;
182     return obj ;
183   },
184   
185   /**  
186     Compares each item in the array.  Returns true if they are equal.
187   */
188   isEqual: function(ary) {
189     if (!ary) return false ;
190     if (ary == this) return true;
191     
192     var loc = ary.get('length') ;
193     if (loc != this.get('length')) return false ;
194 
195     while(--loc >= 0) {
196       if (!SC.isEqual(ary.objectAt(loc), this.objectAt(loc))) return false ;
197     }
198     return true ;
199   }
200     
201 } ;

209 /**
210   Returns a new array that is a slice of the receiver.  This implementation
211   uses the observable array methods to retrieve the objects for the new 
212   slice.
213   
214   @param beginIndex {Integer} (Optional) index to begin slicing from.     
215   @param endIndex {Integer} (Optional) index to end the slice at.
216   @returns {Array} New array with specified slice
217 */
218 SC.Array.slice = function(beginIndex, endIndex) {
219   var ret = []; 
220   var length = this.get('length') ;
221   if (beginIndex == null) beginIndex = 0 ;
222   if ((endIndex == null) || (endIndex > length)) endIndex = length ;
223   while(beginIndex < endIndex) ret[ret.length] = this.objectAt(beginIndex++) ;
224   return ret ;
225 }  ;

234   SC.mixin(Array.prototype, {

236     // primitive for array support.
237     replace: function(idx, amt, objects) {
238       if (!objects || objects.length == 0) {
239         this.splice(idx, amt) ;
240       } else {
241         var args = [idx, amt].concat(objects) ;
242         this.splice.apply(this,args) ;
243       }
244       this.enumerableContentDidChange() ;
245       return this ;
246     },
247   
248     // If you ask for an unknown property, then try to collect the value
249     // from member items.
250     unknownProperty: function(key, value) {
251       var ret = this.reducedProperty(key, value) ;
252       if (ret === undefined) {
253         ret = (value === undefined) ? this.invoke('get', key) : null ;
254       }
255       return ret ;
256     }
