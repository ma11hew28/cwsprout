1 // ==========================================================================
  2 // Not Empty Validator
  3 // Author: Charles Jolley
  4 //
  5 // Force a field to be not empty.
  6 //
  7 // ==========================================================================
  8 
  9 require('validators/validator') ;
 10 
 11 /**
 12   Requires some content in field, but does not check the specific content.
 13   
 14   @class
 15   @extends SC.Validator
 16   @author Charles Jolley
 17   @version 1.0
 18 */
 19 SC.Validator.NotEmpty = SC.Validator.extend(
 20 /** @scope SC.Validator.NotEmpty.prototype */ {
 21   
 22   validate: function(form, field) {
 23     var value = field.get('fieldValue'); 
 24     var ret = !!value ;
 25     if (ret && value.length) ret = value.length > 0 ;
 26     return ret ;
 27   },
 28   
 29   validateError: function(form, field) {
 30     var label = field.get('errorLabel') || 'Field' ;
 31     return $error("Invalid.NotEmpty(%@)".loc(label.capitalize()), field.get('errorLabel'));
 32   }
 33     
 34 }) ;
