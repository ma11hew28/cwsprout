  6 /**
  7   @namespace
  8 
  9   Any views you implement that are scrollable should include this mixin to
 10   provide basic support for scrolling actions.  You can also override any 
 11   of these methods as needed for your own specific behaviors.
 12   
 13   Often times instead of adding SC.Scrollable to your view, you should
 14   place your view inside of an SC.ScrollView.  See that class for more 
 15   info.
 16   
 17   Note that isScrollable must always be true.
 18 
 19 */
 20 SC.Scrollable = {
 21   
 22   /** Informs the view system that the receiver is scrollable.
 23   
 24     Must always be true.
 25   */
 26   isScrollable: true,
 27   
 28   /** 
 29     Amount to scroll one vertical line.
 30   
 31     Used by the default implementation of scrollDownLine() and scrollUpLine().  Defaults
 32     to 20px.
 33   */
 34   verticalLineScroll: 20,
 35   
 36   /**
 37     Amount to scroll one horizontal line.
 38   
 39     Used by the default implementation of scrollLeftLine() and scrollRightLine(). Defaults
 40     to 20px.
 41   */
 42   horizontalLineScroll: 20,
 43   
 44   /**
 45     Amount to scroll one vertical page.
 46     
 47     Used by the default implementation of scrollUpPage() and scrollDownPage(). Defaults to
 48     current innerFrame height.
 49   */
 50   verticalPageScroll: function() {
 51     return this.get('innerFrame').height ;
 52   }.property('innerFrame'),
 53   
 54   /**
 55     Amount to scroll one horizontal page.
 56     
 57     Used by the default implementation of scrollLeftPage() and scrollRightPage().  Defaults
 58     to current innerFrame width.
 59   */
 60   horizontalPageScroll: function() {
 61     return this.get('innerFrame').width ;  
 62   }.property('innerFrame'),
 63   
 64   /**
 65     Returns true if the receiver has enough vertical content to require 
 66     scrolling.
 67     
 68     If you do not want to allow vertical scrolling, override this to be false 
 69     and set the appropriate CSS.
 70     
 71   */
 72   hasVerticalScroller: function() {
 73     return this.get('scrollFrame').height > this.get('innerFrame').height ;
 74   }.property('scrollFrame'),
 75   
 76   /**
 77     Returns true if the receiver has enough horizontal content to require 
 78     scrolling.
 79     
 80     If you do not want to allow horizontal scrolling, override this to be 
 81     false and set the appropriate CSS.
 82     
 83   */
 84   hasHorizontalScroller: function() {
 85     return this.get('scrollFrame').width > this.get('innerFrame').width ;
 86   }.property('scrollFrame'),
 87 
 88   /**
 89     Scrolls the receiver in the horizontal and vertical directions by the 
 90     amount specified, if allowed.
 91     
 92     @param {Point} amount the amount to scroll.  Must include x, y or both
 93     @returns {Point} the actual amount scrolled.
 94   */
 95   scrollBy: function(amount) {
 96     var sf = this.get('scrollFrame') ;
 97     var f = this.get('innerFrame') ;
 98 
 99     if (!this.get('hasVerticalScroller')) amount.y = 0 ;
100     if (sf.height <= f.height) amount.y = 0 ;
101     
102     if (!this.get('hasHorizontalScroller')) amount.x = 0 ; 
103     if (sf.width <= f.width) amount.x = 0 ;
104 
105     // compute new sf
106     var newSf = { x: sf.x - (amount.x || 0), y: sf.y - (amount.y || 0) } ;
107     this.set('scrollFrame', newSf) ;
108     newSf = this.get('scrollFrame') ;
109     
110     return { x: newSf.x - sf.x, y: newSf.y - sf.y }; 
111   },
112 
113   /**
114     Scrolls the receiver to the specified x,y coordinate
115   */
116   scrollTo: function(x,y) {
117     this.set('scrollFrame', { x: 0-x, y: 0-y }) ;  
118   },
119   
120   /**
121     Scroll the view to make the passed frame visible.
122     
123     Frame must be relative to the receiver's offsetParent.
124     
125     @param {SC.View} view the view you want to make visible
126   */
127   scrollToVisible: function(view) {
128 
129     // get frames and convert them to proper offsets
130     var f = this.get('innerFrame') ;
131     var sf = this.get('scrollFrame') ;
132     
133     // frame of the view, relative to the top of the scroll frame
134     var vf = this.convertFrameFromView(view.get('frame'), view) ;
135     vf.x -= (f.x + sf.x); vf.y -= (f.y + sf.y);
136     
137     // first visible origin
138     var vo = { 
139       x: 0-sf.x, 
140       y: 0-sf.y, 
141       width: f.width, 
142       height: f.height 
143     };
144 
145     // if top edge is not visible, shift origin
146     vo.y -= Math.max(0, SC.minY(vo) - SC.minY(vf)) ;
147     vo.x -= Math.max(0, SC.minX(vo) - SC.minX(vf)) ;
148 
149     // if bottom edge is not visible, shift origin
150     vo.y += Math.max(0, SC.maxY(vf) - SC.maxY(vo)) ;
151     vo.x += Math.max(0, SC.maxX(vf) - SC.maxX(vo)) ;
152 
153     // scroll to that origin.
154     this.scrollTo(vo.x, vo.y) ;
155   },
156   
157   /**
158     Scrolls the receiver down one line if allowed.
159     
160     @param {Number} lines number of lines to scroll
161     @returns {Number} the amount actually scrolled.
162   */
163   scrollDownLine: function(lines) {
164     if (lines === undefined) lines = 1 ;
165     return this.scrollBy({ y: this.get('verticalLineScroll')*lines }).y ;
166   },
167 
168   /**
169     Scrolls the receiver down up line if allowed.
170     
171     @param {Number} lines number of lines to scroll
172     @returns {Number} the amount actually scrolled.
173   */
174   scrollUpLine: function(lines) {
175     if (lines === undefined) lines = 1 ;
176     return 0-this.scrollBy({ y: 0-this.get('verticalLineScroll')*lines }).y ;
177   },
178 
179   /**
180     Scrolls the receiver right one line if allowed.
181     
182     @param {Number} lines number of lines to scroll
183     @returns {Number} the amount actually scrolled.
184   */
185   scrollRightLine: function(lines) {
186     if (lines === undefined) lines = 1 ;
187     return this.scrollTo({ y: this.get('horizontalLineScroll')*lines }).x ;
188   },
189 
190   /**
191     Scrolls the receiver left one line if allowed.
192     
193     @param {Number} lines number of lines to scroll
194     @returns {Number} the amount actually scrolled.
195   */
196   scrollLeftLine: function(lines) {
197     if (lines === undefined) lines = 1 ;
198     return 0-this.scrollTo({ y: 0-this.get('horizontalLineScroll')*lines }).x ;
199   },
200 
201   /**
202     Scrolls the receiver down one page if allowed.
203     
204     @param {Number} pages number of pages to scroll
205     @returns {Number} the amount actually scrolled.
206   */
207   scrollDownPage: function(pages) {
208     if (pages === undefined) pages = 1 ;
209     return this.scrollBy({ y: this.get('verticalPageScroll')*pages }).y ;
210   },
211 
212   /**
213     Scrolls the receiver down up page if allowed.
214     
215     @param {Number} pages number of pages to scroll
216     @returns {Number} the amount actually scrolled.
217   */
218   scrollUpPage: function(pages) {
219     if (pages === undefined) pages = 1 ;
220     return 0-this.scrollBy({ y: 0-this.get('verticalPageScroll')*pages }).y ;
221   },
222 
223   /**
224     Scrolls the receiver right one page if allowed.
225     
226     @param {Number} pages number of pages to scroll
227     @returns {Number} the amount actually scrolled.
228   */
229   scrollRightPage: function(pages) {
230     if (pages === undefined) pages = 1 ;
231     return this.scrollTo({ y: this.get('horizontalPageScroll')*pages }).x ;
232   },
233 
234   /**
235     Scrolls the receiver left one page if allowed.
236     
237     @param {Number} pages number of pages to scroll
238     @returns {Number} the amount actually scrolled.
239   */
240   scrollLeftPage: function(pages) {
241     if (pages === undefined) pages = 1 ;
242     return 0-this.scrollTo({ y: 0-this.get('horizontalPageScroll')*pages }).x ;
243   }
244   
245 } ;
