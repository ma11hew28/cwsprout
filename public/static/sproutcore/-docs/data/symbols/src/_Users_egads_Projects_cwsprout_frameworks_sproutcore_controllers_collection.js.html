  9 /** @class
 10 
 11   A CollectionController works just like an ObjectController except that
 12   it includes support for a few extra items including the arrangedObjects
 13   property (which may have a filter applied), and a selection.
 14 
 15   This expects the content object to be a Collection object.
 16 
 17   @extends SC.ObjectController
 18 */
 19 SC.CollectionController = SC.ObjectController.extend(SC.SelectionSupport,
 20 /** @scope SC.CollectionController.prototype */ {
 21 
 22   // ...................................
 23   // PROPERTIES
 24   //
 25 
 26   /**
 27     This is the current set of objects for the UI.
 28     
 29     @type Array
 30   */
 31   arrangedObjects: [],
 32   
 33 
 34   /**
 35     If true, new, add, remove will work.
 36   
 37     @type bool
 38   */
 39   canEditCollection: false,
 40 
 41   /**
 42     Set to the total number of items to show on a single page.  If set to
 43     zero, then no pagination will be performed.
 44   
 45     @type number
 46   */
 47   pageSize: 0,
 48   
 49   /**
 50     [RO] read only property with the current total number of pages.
 51   */
 52   pageCount: function() {
 53     var pageSize = this.get('pageSize') ;
 54     if (pageSize <= 0) return 1 ;
 55     
 56     var content = this.get('content') ;
 57     var count = (content && content.get) ? content.get('count') : 0 ;
 58     if (count === null) count = 0 ;
 59     return Math.ceil(count / pageSize) ;
 60   }.property(),
 61   
 62   /**
 63     Set to the current page.  This will change the offset and limit shown in
 64     the collection.  If you try to set the page to a number greater than the
 65     maximum, then it will be set to the last page.
 66   */
 67   currentPage: function(key, value) {
 68     if (value !== undefined) {
 69       // constrain pages.
 70       if (this._currentPage != value) {
 71         var pc = Math.max(this.get('pageCount')-1,0);
 72         if (value > pc) value = pc ;
 73         if (value < 0) value = 0 ;
 74         this._currentPage = value ;
 75       }
 76     }
 77     return this._currentPage || 0 ;  
 78   }.property(),
 79   
 80   // ...................................
 81   // METHODS
 82   //
 83 
 84   // adds a new object to the current collection, if allowed, and sets it
 85   // as the current selection.
 86   newObject: function(settings) {
 87     var content = this.get('content') ;
 88     if (!content || !this.get('canEditCollection')) return ; // only if allowed 
 89     try {
 90       if (content.newRecord) { 
 91         var rec = content.newRecord(settings) ;
 92         var t = function() {
 93           this.set('selection',(rec) ? [rec] : []) ;
 94           this._editingNewRecord = rec ;
 95         }.invokeLater(this, 1) ;
 96         return rec;
 97       }
 98     }   
 99     catch (e) {
100       // DO SOMETHING
101     }
102   },
103 
104   addObjects: function(objects) {
105     var content = this.get('content') ;
106     if (!content || !this.get('canEditCollection')) return ; // only if allowed 
107     try {
108       objects = SC.$A(arguments).flatten() ;
109       if (content.addRecords) { 
110         content.addRecords(objects) ;
111         this.set('selection',(objects) ? objects : []) ;
112       }
113     }   
114     catch (e) {
115       // DO SOMETHING
116     }
117   },
118   
119   addSelection: function() { 
120     return this.addObjects(this.get('selection'));
121   },
122   
123   // adds a new object to the current collection, if allowed, and sets it
124   // as the current selection.
125   removeObjects: function(objects) {
126     var content = this.get('content') ;
127     if (!content || !this.get('canEditCollection')) return ; // only if allowed 
128     try {
129       objects = SC.$A(arguments).flatten() ;
130       if (content.removeRecords) { 
131         var rec = content.removeRecords(objects) ;
132         var sel = (this.get('selection') || []).without(objects) ;
133         this.set('selection',(sel) ? sel : []) ;
134       }
135     }   
136     catch (e) {
137       // DO SOMETHING
138     }
139   },
140 
141   removeSelection: function() {
142     return this.removeObjects(this.get('selection')) ;
143   },
144 
145   // this method is called if a new object was created through the controller
146   // and then the selection was changed from the new record without the 
147   // record being saved first.  By default, this will remove the object, but
148   // you could override it to just do a commit.
149   newObjectDidLoseFocus: function(rec) { rec.destroy() ; },
150   
151   // ...................................
152   // PRIVATE
153   //
154   
155   _newRecordDidLoseFocus: function(rec) {
156     if (rec.get('newRecord')) this.newObjectDidLoseFocus(rec) ;
157   },
158   
159   // Update the current page.
160   _pageObserver: function() {
161     // get content -- nothing to do if no content.
162     var content = this.get('content') ; 
163     if (content instanceof Array) content = content[0] ;
164     if (!content) return ; 
165     
166     var curOffset = content.get('offset') || 0 ;
167     var curLimit = content.get('limit') || 0 ;
168     var count = content.get('count') || 0 ;
169 
170     // calculate the offset and limit.
171     var currentPage = this.get('currentPage') ;
172     var pageSize = this.get('pageSize') ;
173     var newOffset, newLimit ;
174     if (pageSize == 0) {
175       newOffset = 0; newLimit = 0 ;
176     } else {
177       newOffset = currentPage * pageSize ;
178       newLimit = pageSize ;      
179     }
180     
181     // set new page info.
182     if ((newOffset != curOffset) || (newLimit != curLimit)) {
183       content.beginPropertyChanges() ;
184       content.set('offset',newOffset) ;
185       content.set('limit',newLimit) ;
186       content.endPropertyChanges() ;
187     }    
188   }.observes('currentPage','pageCount','pageSize'),
189   
190   // invoked whenever the list changes.  Updated the arrangedObjects and
191   // potentially the selection.
192   _recordsObserver: function(target,key,value) {
193     var old = this.get('arrangedObjects') ;
194     value = Array.from(target.get(key)) ;
195     
196     this.set('arrangedObjects',value.slice()) ;
