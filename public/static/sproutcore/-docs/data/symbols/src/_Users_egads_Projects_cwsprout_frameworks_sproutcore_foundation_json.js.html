  7     This file adds these methods to JavaScript:
  8 
  9         array.toJSONString()
 10         boolean.toJSONString()
 11         date.toJSONString()
 12         number.toJSONString()
 13         object.toJSONString()
 14         string.toJSONString()
 15             These methods produce a JSON text from a JavaScript value.
 16             It must not contain any cyclical references. Illegal values
 17             will be excluded.
 18 
 19             The default conversion for dates is to an ISO string. You can
 20             add a toJSONString method to any date object to get a different
 21             representation.
 22 
 23         string.parseJSON(filter)
 24             This method parses a JSON text to produce an object or
 25             array. It can throw a SyntaxError exception.
 26 
 27             The optional filter parameter is a function which can filter and
 28             transform the results. It receives each of the keys and values, and
 29             its return value is used instead of the original value. If it
 30             returns what it received, then structure is not modified. If it
 31             returns undefined then the member is deleted.
 32 
 33             Example:
 34 
 35             // Parse the text. If a key contains the string 'date' then
 36             // convert the value to a date.
 37 
 38             myData = text.parseJSON(function (key, value) {
 39                 return key.indexOf('date') >= 0 ? new Date(value) : value;
 40             });

 60     Array.prototype.toJSONString = function () {
 61         var a = ['['],  // The array holding the text fragments.
 62             b,          // A boolean indicating that a comma is required.
 63             i,          // Loop counter.
 64             l = this.length,
 65             v;          // The value to be stringified.
 66 
 67         function p(s) {
 68 
 69 // p accumulates text fragments in an array. It inserts a comma before all
 70 // except the first fragment.
 71 
 72             if (b) {
 73                 a.push(',');
 74             }
 75             a.push(s);
 76             b = true;
 77         }
 78 
 79 // For each value in this array...
 80 
 81         for (i = 0; i < l; i += 1) {
 82             v = this[i];
 83             switch (typeof v) {
 84 
 85 // Serialize a JavaScript object value. Ignore objects thats lack the
 86 // toJSONString method. Due to a specification error in ECMAScript,
 87 // typeof null is 'object', so watch out for that case.
 88 
 89             case 'object':
 90                 if (v) {
 91                     if (typeof v.toJSONString === 'function') {
 92                         p(v.toJSONString());
 93                     } else p(Object.toJSONString(v)) ;
 94                 } else {
 95                     p("null");
 96                 }
 97                 break;
 98 
 99 // Otherwise, serialize the value.
100 
101             case 'string':
102             case 'number':
103             case 'boolean':
104                 p(v.toJSONString());
105 
106 // Values without a JSON representation are ignored.
107 
108             }
109         }
110 
111 // Join all of the fragments together and return.
112 
113         a.push(']');
114         return a.join('');
115     };

151     Object.toJSONString = function (object) {
152         var a = ['{'],  // The array holding the text fragments.
153             b,          // A boolean indicating that a comma is required.
154             k,          // The current key.
155             v;          // The current value.
156 
157         function p(s) {
158 
159 // p accumulates text fragment pairs in an array. It inserts a comma before all
160 // except the first fragment pair.
161 
162             if (b) {
163                 a.push(',');
164             }
165             a.push(k.toJSONString(), ':', s);
166             b = true;
167         }
168 
169 // Iterate through all of the keys in the object, ignoring the proto chain.
170 
171         for (k in object) {
172             if (object.hasOwnProperty(k)) {
173                 v = object[k];
174                 switch (typeof v) {
175 
176 // Serialize a JavaScript object value. Ignore objects that lack the
177 // toJSONString method. Due to a specification error in ECMAScript,
178 // typeof null is 'object', so watch out for that case.
179 
180                 case 'object':
181                     if (v) {
182                         if (typeof v.toJSONString === 'function') {
183                             p(v.toJSONString());
184                         } else p(Object.toJSONString(v)) ;
185                     } else {
186                         p("null");
187                     }
188                     break;
189 
190             case 'string':
191             case 'number':
192             case 'boolean':
193                     p(v.toJSONString());
194 
195 // Values without a JSON representation are ignored.
196 
197                 }
198             }
199         }
200 
201 // Join all of the fragments together and return.
202 
203         a.push('}');
204         return a.join('');
205     };

210 // Augment String.prototype. We do this in an immediate anonymous function to
211 // avoid defining global variables.

213 // m is a table of character substitutions.
214 
215         var m = {
216             '\b': '\\b',
217             '\t': '\\t',
218             '\n': '\\n',
219             '\f': '\\f',
220             '\r': '\\r',
221             '"' : '\\"',

226         s.parseJSON = function (filter) {
227 
228 // Parsing happens in three stages. In the first stage, we run the text against
229 // a regular expression which looks for non-JSON characters. We are especially
230 // concerned with '()' and 'new' because they can cause invocation, and '='
231 // because it can cause mutation. But just to be safe, we will reject all
232 // unexpected characters.
233 
234             try {
235                 if (/^("(\\.|[^"\\\n\r])*?"|[,:{}\[\]0-9.\-+Eaeflnr-u \n\r\t])+?$/.
236                         test(this)) {
237 
238 // In the second stage we use the eval function to compile the text into a
239 // JavaScript structure. The '{' operator is subject to a syntactic ambiguity
240 // in JavaScript: it can begin a block or an object literal. We wrap the text
241 // in parens to eliminate the ambiguity.
242 
243                     var j = eval('(' + this + ')');
244 
245 // In the optional third stage, we recursively walk the new structure, passing
246 // each name/value pair to a filter function for possible transformation.
247 
248                     if (typeof filter === 'function') {
249 
250                         function walk(k, v) {
251                             if (v && typeof v === 'object') {
252                                 for (var i in v) {
253                                     if (v.hasOwnProperty(i)) {
254                                         v[i] = walk(i, v[i]);
255                                     }
256                                 }
257                             }
258                             return filter(k, v);
259                         }
260 
261                         j = walk('', j);
262                     }
263                     return j;
264                 }
265             } catch (e) {
266 
267 // Fall through if the regexp test fails.
268 
269             }
270             throw new SyntaxError("parseJSON");
271         };
