  6 require('validators/validator') ;

  8 /**
  9   Validate a field value as a credit card number. 
 10   
 11   This validator will perform a basic check to ensure the credit card number
 12   is mathematically valid.  It will also accept numbers with spaces, dashes
 13   or other punctuation.  
 14   
 15   Converted credit card numbers are broken into units of 4.
 16   
 17   Basic credit card validation courtesy David Leppek 
 18   (https://www.azcode.com/Mod10)
 19 
 20   @class
 21   @extends SC.Validator
 22   @author Charles Jolley
 23   @version 1.0
 24 */
 25 SC.Validator.CreditCard = SC.Validator.extend(
 26 /** @scope SC.Validator.CreditCard.prototype */ {
 27 
 28   /**
 29     Expects a string of 16 digits.  Will split into groups of 4 for display.
 30   */
 31   fieldValueForObject: function(object, form, field) {
 32     if (typeof(object) == "string" && object.length == 16) {
 33       object = [object.slice(0,4),object.slice(4,8),object.slice(8,12),object.slice(12,16)].join(' ') ;
 34     }
 35     return object ;
 36   },
 37 
 38   /**
 39     Removes all whitespace or dashes to make a single string.
 40   */
 41   objectForFieldValue: function(value, form, field) {
 42     return value.replace(/[\s-\.\:]/g,'') ;
 43   },
 44   
 45   validate: function(form, field) { 
 46     return this.checkNumber(field.get('fieldValue')) ; 
 47   },
 48   
 49   validateError: function(form, field) {
 50     var label = field.get('errorLabel') || 'Field' ;
 51     return $error("Invalid.CreditCard(%@)".loc(label), label);
 52   },
 53   
 54   checkNumber: function(ccNumb) {
 55     var valid = "0123456789";  // Valid digits in a credit card number
 56     var len = ccNumb.length;  // The length of the submitted cc number
 57     var iCCN = parseInt(ccNumb,0);  // integer of ccNumb
 58     var sCCN = ccNumb.toString();  // string of ccNumb
 59     sCCN = sCCN.replace (/^\s+|\s+$/g,'');  // strip spaces
 60     var iTotal = 0;  // integer total set at zero
 61     var bNum = true;  // by default assume it is a number
 62     var bResult = false;  // by default assume it is NOT a valid cc
 63     var temp;  // temp variable for parsing string
 64     var calc;  // used for calculation of each digit
 65 
 66     // Determine if the ccNumb is in fact all numbers
 67     for (var j=0; j<len; j++) {
 68       temp = "" + sCCN.substring(j, j+1);
 69       if (valid.indexOf(temp) == "-1"){bNum = false;}
 70     }
 71 
 72     // if it is NOT a number, you can either alert to the fact, 
 73     // or just pass a failure
 74     if(!bNum) bResult = false;
 75 
 76     // Determine if it is the proper length 
 77     if((len == 0)&&(bResult)){  // nothing, field is blank AND passed above # check
 78       bResult = false;
 79     } else{  // ccNumb is a number and the proper length - let's see if it is a valid card number
 80       if(len >= 15){  // 15 or 16 for Amex or V/MC
 81         for(var i=len;i>0;i--){  // LOOP throught the digits of the card
 82           calc = parseInt(iCCN,0) % 10;  // right most digit
 83           calc = parseInt(calc,0);  // assure it is an integer
 84           iTotal += calc;  // running total of the card number as we loop - Do Nothing to first digit
 85           i--;  // decrement the count - move to the next digit in the card
 86           iCCN = iCCN / 10;                               // subtracts right most digit from ccNumb
 87           calc = parseInt(iCCN,0) % 10 ;    // NEXT right most digit
 88           calc = calc *2;                                 // multiply the digit by two
 89           // Instead of some screwy method of converting 16 to a string and then parsing 1 and 6 and then adding them to make 7,
 90           // I use a simple switch statement to change the value of calc2 to 7 if 16 is the multiple.
 91           switch(calc){
 92             case 10: calc = 1; break;       //5*2=10 & 1+0 = 1
 93             case 12: calc = 3; break;       //6*2=12 & 1+2 = 3
 94             case 14: calc = 5; break;       //7*2=14 & 1+4 = 5
 95             case 16: calc = 7; break;       //8*2=16 & 1+6 = 7
 96             case 18: calc = 9; break;       //9*2=18 & 1+8 = 9
 97             default: calc = calc;           //4*2= 8 &   8 = 8  -same for all lower numbers
 98           }                                               
 99         iCCN = iCCN / 10;  // subtracts right most digit from ccNum
100         iTotal += calc;  // running total of the card number as we loop
101       }  // END OF LOOP
102       if ((iTotal%10)==0){  // check to see if the sum Mod 10 is zero
103         bResult = true;  // This IS (or could be) a valid credit card number.
104       } else {
105         bResult = false;  // This could NOT be a valid credit card number
106         }
107       }
108     }
109     return bResult; // Return the results
110   }
111     
112 }) ;
