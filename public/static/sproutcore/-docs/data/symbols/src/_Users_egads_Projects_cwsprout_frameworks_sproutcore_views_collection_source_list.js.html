152   // whenever updateChildren is called with a deep method, flush the
153   // cached group rows to make sure we get an accurate count.
154   updateChildren: function(deep) {
155     if (deep) this._groupRows = null ;
156     return sc_super() ;

159   // determines if the group at the specified content index is visible or
160   // not.  This will look either at a property on the group or on the
161   // SourceListGroupView.
162   groupAtContentIndexIsVisible: function(contentIndex) {
163     
164     if (!this.get('groupBy')) return YES; // no grouping
165 
166     // get the group value and try to find a matching view, which may
167     // or may not exist yet.
168     var groupValue = this.groupValueAtContentIndex(contentIndex) ;
169     var groupView = this.groupViewForGroupValue(groupValue) ;
170     
171     // if the groupView exists, use that.  The visible state is stored here
172     // in case the group does not actually support storing its own visibility.
173     // ignore groupView if it does not support isGroupVisible
174     var ret = YES ;
175     if (groupView) ret = groupView.get('isGroupVisible') ;
176 
177     // otherwise try to get from the group itself.
178     if (((ret === undefined) || (ret === null) || !groupView) && groupValue && groupValue.get) {
179       var key = this.get('groupVisibleKey') ;
180       if (key) ret = !!groupValue.get(key) ;
181     }
182     
183     // if the above methods failed for some reason, just leave the group visible
184     if ((ret === undefined) || (ret === null)) ret = YES ;
185     
186     return ret ;
187   },
188     
189   // calculates the number of rows consumed by each group.  stores a hash of
190   // contentIndexes and rows. 
191   computedGroupRows: function() {
192     if (this._groupRows) return this._groupRows;
193     
194     var loc = 0 ; 
195     var content = Array.from(this.get('content')) ;  
196     var max = content.get('length') ;
197     
198     var ret = {} ;
199     while(loc < max) {
200       var range = this.groupRangeForContentIndex(loc) ;
201       var isVisible = this.groupAtContentIndexIsVisible(range.start) ;
202       ret[range.start] = (isVisible) ? range.length : 0 ;
203 
204       // add a header row space if neede
205       var groupValue = this.groupValueAtContentIndex(range.start) ;
206       if (groupValue != null) ret[range.start]++ ;
207       
208       loc = (range.length <= 0) ? max : SC.maxRange(range) ;
209     }
210     
211     return this._groupRows = ret ;
212   },
213 
214   // Returns the number of rows in the specified range.
215   countRowsInRange: function(range) {
216     var groupRows = this.computedGroupRows() ;
217     var max = SC.maxRange(range) ;
218     var loc = SC.minRange(range) ;
219     var ret = 0 ;
220     
221     while(loc < max) {
222       var range = this.groupRangeForContentIndex(loc) ;
223       loc = (range.length <= 0) ? max : SC.maxRange(range) ;
224       ret += groupRows[range.start] || (range+1);
225     }
226     return ret ;
227   },
228   
229   computeFrame: function() {
230     var content = this.get('content') ;
231     var rowHeight = this.get('rowHeight') || 20 ;
232 
233     // find number of groups.  
234     var rows = this.countRowsInRange({ start: 0, length: content.get('length') });
235     if (rows <= 0) rows = 0 ;
236 
237     // get parent width
238     var parent = this.get('parentNode') ;
239     var f = (parent) ? parent.get('innerFrame') : { width: 100, height: 100 };
240     f.x = f.y = 0;
241     f.height = Math.max(f.height, rows * rowHeight) ;
242     return f ;
243   },
244 
245   
246   // disable incremental rendering for now
247   contentRangeInFrame: function(frame) {
248     var content =this.get('content') ;
249     var len = (content) ? content.get('length') : 0 ;
250     var ret = { start: 0, length: len } ;
251     return ret ;
252   },

255   /** @private */
256   layoutItemView: function(itemView, contentIndex, firstLayout) {
257     if (SC.BENCHMARK_SOURCE_LIST_VIEW) {
258       SC.Benchmark.start('SC.SourceListView.layoutItemViewsFor') ;
259     }
260     
261     // if itemView's group is not visible, then just set to invisible.
262     if (!this.groupAtContentIndexIsVisible(contentIndex)) {
263       itemView.set('isVisible', false) ;
264     } else {
265 
266       // if item was not visible, make it visible.  Also force layout.
267       if (!itemView.get('isVisible')) {
268         firstLayout = YES ;        
269         itemView.set('isVisible', true) ;
270       }
271 
272       var rowHeight = this.get('rowHeight') || 0 ;
273 
274 
275       // layout relative to top of group.  Leave open row for title
276       if(this.get("groupBy"))
277       {
278         
279         var range = this.groupRangeForContentIndex(contentIndex) ;
280         contentIndex = (contentIndex - range.start) ;
281 
282         var groupValue = this.groupValueAtContentIndex(range.start) ;
283         if (groupValue != null) contentIndex++ ;
284       }
285 
286       var f = { 
287         x: 0, 
288         y: contentIndex*rowHeight,
289         height: rowHeight, 
290         width: this.get('innerFrame').width 
291       } ;
292 
293       if (firstLayout || !SC.rectsEqual(itemView.get('frame'), f)) {
294         itemView.set('frame', f) ;      
295       }
296 
297     }
298     
299     if (SC.BENCHMARK_SOURCE_LIST_VIEW) {
300       SC.Benchmark.end('SC.SourceListView.layoutItemViewsFor') ;
301     }
302   },
303   
304   layoutGroupView: function(groupView, groupValue, contentIndexHint, firstLayout) {
305     
306     if (SC.BENCHMARK_SOURCE_LIST_VIEW) {
307       SC.Benchmark.start('SC.SourceListView.layoutGroupView') ;
308     }
309     
310     //console.log('layoutGroupView', groupValue) ;
311     
312     // find the range this group will belong to
313     var range = this.groupRangeForContentIndex(contentIndexHint) ;
314     var isVisible = this.groupAtContentIndexIsVisible(range.start) ;
315     
316     var priorRows = this.countRowsInRange({ start: 0, length: range.start }) ;
317     var rowHeight = this.get('rowHeight') || 0 ;
318     var parentView = groupView.get('parentView') || this ;
319     var rows = (isVisible) ? range.length : 0 ;
320     if (groupValue != null) rows++ ;
321 
322     var f = { 
323       x: 0, 
324       y: priorRows*rowHeight,
325       height: rowHeight * rows, 
326       width: (parentView || this).get('innerFrame').width 
327     } ;
328     
329     if (firstLayout || !SC.rectsEqual(groupView.get('frame'), f)) {
330       groupView.set('frame', f) ;      
331     }
332 
333     if (SC.BENCHMARK_SOURCE_LIST_VIEW) {
334       SC.Benchmark.end('SC.SourceListView.layoutGroupView') ;    
335     }

348   showInsertionPoint: function(itemView, dropOperation) {
349     if (!itemView) return ;
350 
351     // if drop on, then just add a class...
352     if (dropOperation === SC.DROP_ON) {
353       if (itemView !== this._dropOnInsertionPoint) {
354         this.hideInsertionPoint() ;
355         itemView.addClassName('drop-target') ;
356         this._dropOnInsertionPoint = itemView ;
357       }
358       
359     } else {
360       
361       if (this._dropOnInsertionPoint) {
362         this._dropOnInsertionPoint.removeClassName('drop-target') ;
363         this._dropOnInsertionPoint = null ;
364       }
365     
366       if (!this._insertionPointView) {
367         this._insertionPointView = this.insertionPointClass.create() ;
368       } ;
369     
370       var insertionPoint = this._insertionPointView ;
371       var f = this.calculateInsertionPointFrame(itemView);
372       insertionPoint.set('frame', f) ;
373 
374       if (insertionPoint.parentNode != itemView.parentNode) {
375         itemView.parentNode.appendChild(insertionPoint) ;
376       }
377     }
378     
379   },

381   /**
382     This is the default frame for the insertion point.  Override this method 
383     if your insertion point's styling needs to be customized, or if you need 
384     more control of the insertion point's positioning (i.e., heirarchical 
385     placement)
386   */
387   calculateInsertionPointFrame: function(itemView) {
388     return { height: 0, x: 8, y: itemView.get('frame').y, width: itemView.owner.get('frame').width };
389   },
390 
391   hideInsertionPoint: function() {
392     var insertionPoint = this._insertionPointView ;
393     if (insertionPoint) insertionPoint.removeFromParent() ;
394 
395     if (this._dropOnInsertionPoint) {
396       this._dropOnInsertionPoint.removeClassName('drop-target') ;
397       this._dropOnInsertionPoint = null ;
398     }
399   },

401   // We can do this much faster programatically using the rowHeight
402   insertionIndexForLocation: function(loc, dropOperation) {  
403     var f = this.get('innerFrame') ;
404     var sf = this.get('scrollFrame') ;
405     var rowHeight = this.get('rowHeight') || 0 ;
406     var headerRowCount = (this.get("groupBy")) ? 1 : 0;
407     
408     // find the offset to work with.
409     var offset = loc.y - f.y - sf.y ;
410     
411     var ret = -1; // the return value
412     var retOp = SC.DROP_BEFORE ;
413 
414     // search groups until we find one that matches
415     var top = 0 ;
416     var idx = 0 ;
417     while((ret<0) && (range = this.groupRangeForContentIndex(idx)).length>0){
418       var max = top + ((range.length+headerRowCount) * rowHeight) ;
419       
420       // the offset is within the group, find the row in the group.  Remember
421       // that the top row is actually the label, so we should return -1 if 
422       // we hit there.
423       if (max >= offset) {
424         offset -= top ;
425           
426         ret = Math.floor(offset / rowHeight) ;
427 
428         // find the percent through the row...
429         var percentage = (offset / rowHeight) - ret ;
430         
431         // if the dropOperation is SC.DROP_ON and we are in the center 60%
432         // then return the current item.
433         if (dropOperation === SC.DROP_ON) {
434           if (percentage > 0.80) ret++ ;
435           if ((percentage >= 0.20) && (percentage <= 0.80)) {
436             retOp = SC.DROP_ON;
437           }
438         } else {
439           if (percentage > 0.45) ret++ ;
440         }
441         
442         // handle dropping on top row...
443         if (ret < headerRowCount) return [-1, SC.DROP_BEFORE] ; // top row!
444         
445         // convert to index
446         ret = (ret - headerRowCount) + idx ;
447 
448       // we are not yet within the group, go on to the next group.
449       } else {
450         idx += range.length ;
451         top = max ;
452       }
453     }
454     
455     return [ret, retOp] ;
456   }
