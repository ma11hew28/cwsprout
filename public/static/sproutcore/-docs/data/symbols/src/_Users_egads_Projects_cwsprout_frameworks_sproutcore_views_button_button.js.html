 16 /** @class
 17 
 18   A button handles simple link functions.  It can be set to a selected,
 19   enabled or disabled state.
 20

 27 SC.ButtonView = SC.View.extend(SC.Control,
 28 /** @scope SC.ButtonView.prototype */ {
 29   
 30   emptyElement: '<a href="javascript:;" class="sc-button-view regular"><span class="button-inner"><span class="label"></span></span></a>',
 31   
 32   // PROPERTIES
 33   
 34   /**
 35     Used to automatically update the state of the button view for toggle style
 36     buttons.
 37     
 38     for toggle style buttons, you can set the value and it will be used to
 39     automatically update the state of the button view.  The value will also
 40     change as the user selects to deselects.  The button will make its best
 41     effort to convert this value into a reasonable selection state:
 42   
 43     null, false, 0 -> isSelected = false
 44     any other single value -> isSelected = true
 45     array -> if all values are the same state: that state.  otherwise MIXED.
 46     
 47     @type Object
 48 
 49   */  
 50   value: false,
 51   
 52   /**
 53     Value of a selected toggle button.
 54   
 55     for a toggle button, set this to any object value you want.  The button
 56     will be selected if the value property equals the targetValue.  If the
 57     value is an array of multiple items that contains the targetValue, then
 58     the button will be set to a mixed state.
 59 
 60     default is false
 61     
 62     @type Object
 63   */
 64   toggleOnValue: true,
 65 
 66   /**
 67     Value of an unselected toggle button.
 68   
 69     For a toggle button, set this to any object value you want.  When the
 70     user toggle's the button off, the value of the button will be set to this
 71     value.
 72   
 73     default is false 
 74   
 75     @type Object
 76 
 77   */
 78   toggleOffValue: false,
 79   
 80   /**
 81     optionally set this to the theme you want this button to have.  
 82     
 83     This is used to determine the type of button this is.  You generally 
 84     should set a class name on the HTML with the same value to allow CSS 
 85     styling.
 86     
 87     The default SproutCore theme supports "regular", "checkbox", and "radio"
 88   */
 89   theme: 'regular',
 90   
 91   /**
 92     Optionally set the behavioral mode of this button.  
 93   
 94     Possible values are:
 95 
 96     {{{
 97      SC.PUSH_BEHAVIOR: ('push')
 98        Pressing the button will trigger an action tied to the button. Does
 99        not change the value of the button.
100   
101      SC.TOGGLE_BEHAVIOR: ('toggle')
102        Pressing the button will invert the current value of the button. If
103        the button has a mixed value, it will be set to true.
104   
105      SC.TOGGLE_ON_BEHAVIOR: ('on')
106        Pressing the button will set the current state to true no matter the
107        previous value.
108   
109      SC.TOGGLE_OFF_BEHAVIOR: ('off')
110        Pressing the button will set the current state to false no matter the
111        previous value.
112     }}}
113   
114   */  
115   buttonBehavior: SC.PUSH_BEHAVIOR,
116   
117   /**
118     If NO the button will be disabled. 
119     
120     @type Bool
121   */  
122   isEnabled: YES,
123   
124   /**
125     button's selection state.  Returns YES, NO, or SC.MIXED_STATE
126   */
127   isSelected: NO,
128 
129   /**
130     If YES, then this button will be triggered when you hit return.
131     
132     This is the same as setting the keyEquivalent to 'return'.  This will also
133     apply the "def" classname to the button.
134   */
135   isDefault: NO,
136   isDefaultBindingDefault: SC.Binding.OneWayBool,
137   
138   /**
139     If YES, then this button will be triggered when you hit escape.
140     
141     This is the same as setting the keyEquivalent to 'escape'.
142   */  
143   isCancel: NO,
144   isCancelBindingDefault: SC.Binding.OneWayBool,
145   
146   /**
147     If YES, then the title will be localized.
148   */
149   localize: NO,
150   
151   /**
152     The selector path to the element that contains the button title. 
153     
154     This property is only used if you try to get or set the title property.
155   */
156   titleSelector: '.label',
157   
158   /**
159     The button title.
160     
161     This property is observable and bindable.
162     
163     @field {String}
164   */
165   title: function(key, value) {
166     
167     // set the value of the label text.  Possibly localize and set innerHTML.
168     if (value !== undefined) {
169       if (this._title != value) {
170         var text = this._title = value ;
171         var lsel = this.get('titleSelector') ;
172         var el = (lsel) ? this.$sel(lsel) : this.rootElement ;
173 
174         if (this.get('localize')) text = text.loc() ;
175         el.innerHTML = text ;
176       }
177     }
178 
179     // lazily fetch the label text.
180     if (!this._title) {
181       var el = this.$sel(this.get('titleSelector')) ;
182       this._title = (el) ? el.innerHTML : '' ;
183     }
184     return this._title ;
185   }.property(),
186   
187   /**
188     The button href value.
189       
190     This property is observable and bindable. Can be used to create localized button href values.
191   */
192   href: function(key, value) {
193     var el = this.rootElement;
194     if (value !== undefined) {
195       if (el) {
196         el.setAttribute('href', value);
197       }
198     }
199     return (el) ? el.getAttribute('href') : null;
200   }.property(),
201 
202   /**
203     The name of the action you want triggered when the button is pressed.  
204     
205     This property is used in conjunction with the target property to execute
206     a method when a regular button is pressed.  These properties are not 
207     relevant when the button is used in toggle mode.
208     
209     If you do not set a target, then pressing a button will cause the
210     responder chain to search for a view that implements the action you name
211     here.  If you set a target, then the button will try to call the method
212     on the target itself.
213     
214     For legacy support, you can also set the action property to a function.  
215     Doing so will cause the function itself to be called when the button is
216     clicked.  It is generally better to use the target/action approach and 
217     to implement your code in a controller of some type.
218     
219     @type String
220   */
221   action: null,
222   
223   /**
224     The target object to invoke the action on when the button is pressed.
225     
226     If you set this target, the action will be called on the target object
227     directly when the button is clicked.  If you leave this property set to
228     null, then the button will search the responder chain for a view that 
229     implements the action when the button is pressed instead.
230     
231     @type Object
232   */
233   target: null,
234 
235   /**
236     The key equivalent that should trigger this button on the page.
237   */
238   keyEquivalent: null,
239   
240   /** @private {String} used to store a previously defined key equiv */
241   _defaultKeyEquivalent: null,
242   
243   performKeyEquivalent: function( keystring, evt )
244   {
245     if (!this.get('isEnabled')) return false;
246     var keyEquivalent = this.get('keyEquivalent');
247     if (keyEquivalent && (keyEquivalent == keystring))
248     {
249       // button has defined a keyEquivalent and it matches!
250       // if triggering succeeded, true will be returned and the operation will be handeled 
251       // (i.e performKeyEquivalent will cease crawling the view tree)
252       return this.triggerAction(evt);
253     }
254     return false;
255   },
256   
257   /**
258     fakes a click... evt is optional.  
259     
260     Temporarily highlights the button to show that it is being triggered.  
261     Does nothing if the button is disabled. 
262     
263     @returns {bool} success/failure of the request
264   */  
265   triggerAction: function(evt) {  
266     if (!this.get('isEnabled')) return false;
267     this.setClassName('active', true);
268     this.didTriggerAction();
269     this._action(evt);
270     this.invokeLater('setClassName', 200, 'active', false) ;
271     return true;
272   },
273   
274   didTriggerAction: function() {},
275 
276   // ................................................................
277   // INTERNAL SUPPORT
278   
279   /** @private */
280   init: function() {
281     arguments.callee.base.call(this) ;
282     
283     //cache the key equivalent
284     if(this.get("keyEquivalent")) this._defaultKeyEquivalent = this.get("keyEquivalent"); 
285     // setup initial CSS clases
286     this._isDefaultOrCancelObserver() ;
287     
288     // If we need to localze, handle it...
289     var el ;
290     var sel = this.get('titleSelector') ;
291     if (this.get('localize') && sel && (el = this.$sel(sel))) {
292       this._title = (el.innerHTML || '').strip() ;
293       el.innerHTML = this._title.loc() ;
294     }
295   },
296 
297   // determines the target selected state
298   _selectedStateFromValue: function(value) {
299     var targetValue = this.get('toggleOnValue') ;
300     var state ;
301     
302     if ($type(value) == T_ARRAY) {
303       if (value.length == 1) {
304         state = (value[0] == targetValue) ;
305       } else {
306         state = (value.indexOf(targetValue) >= 0) ? SC.MIXED_STATE : false ;
307       }
308     } else {
309       state = (value == targetValue) ;
310     }
311     return state ;
312   },
313   
314   /** @private */
315   propertyObserver: function(observing,target,key,value) {
316     if (target != this) return ;
317 
318     // handle changes to the value
319     switch(key) {
320 
321       // determine the new selection state.
322       case 'value':
323         value = this.get('value') ;
324         if (value == this._value) return ; // process value one time.
325         this._value = value ;
326       
327         // set the new selected state if it does not match
328         var state = this._selectedStateFromValue(value) ;
329         this.setIfChanged('isSelected', state) ;
330         break ;
331 
332       // forward to value if needed.
333       case 'isSelected':
334         var newState = this.get('isSelected') ;
335         var curState = this._selectedStateFromValue(this.get('value')) ;
336         if (curState != newState) {
337           var valueKey = (newState) ? 'toggleOnValue' : 'toggleOffValue' ;
338           this.set('value', this.get(valueKey)) ;
339         }
340         break ;
341         
342       // otherwise do nothing
343       default:
344         break ;
345     }
346   },
347   
348   _isDefaultOrCancelObserver: function() {
349     var isDef = !!this.get('isDefault') ;
350     var isCancel = !isDef && this.get('isCancel') ;
351     
352     if(this.didChangeFor('defaultCancelChanged','isDefault','isCancel')) {
353       this.setClassName('def', isDef) ;
354       if (isDef) {
355         this.setIfChanged('keyEquivalent', 'return');
356       } 
357       else if (isCancel)
358       {
359         this.setIfChanged('keyEquivalent', 'escape') ;
360       }
361       else
362       {
363         //restore the default key equivalent
364         this.set("keyEquivalent",this._defaultKeyEquivalent);
365       }
366     }
367       
368   }.observes('isDefault', 'isCancel'),
369     
370     
371   isMouseDown: false, 
372     
373   // on mouse down, set active only if enabled.  
374   /** @private */
375   mouseDown: function(evt) {
376     this.setClassName('active',this.get('isEnabled')) ;
377     this._isMouseDown = true;
378     return true ;
379   },
380   
381   // remove the active class on mouse down as well
382   /** @private */
383   
384   mouseOut: function(evt)
385   {
386     this.setClassName('active', false);
387     return true;
388   },
389   
390   // add the active class name if the mouse is down
391   // this covers a scenario where the user drags out and back on to a button
392   mouseOver: function(evt)
393   {
394     this.setClassName('active', this._isMouseDown) ;
395     return true;
396   },
397   
398   // on mouse up, trigger the action only if we are enabled and the mouse
399   // was released inside the view.
400   /** @private */
401   mouseUp: function(evt) {
402     this.setClassName('active', false) ;
403     this._isMouseDown = false;
404     var tgt = Event.element(evt) ;
405     var inside = false ;
406     while(tgt && (tgt != this.rootElement)) tgt = tgt.parentNode;
407     if (tgt == this.rootElement) inside = true ;
408     
409     if (inside && this.get('isEnabled')) this._action(evt) ;
410     return true ;
411   },
412 
413   // perform action depending on the behavior of the button.
414   _action: function(evt) {
415     switch(this.get('buttonBehavior')) {
416       
417     // When toggling, try to invert like values. i.e. 1 => 0, etc.
418     case SC.TOGGLE_BEHAVIOR:
419       var sel = this.get('isSelected') ;
420       if (sel == true) {
421         this.set('value', this.get('toggleOffValue')) ;
422       } else {
423         this.set('value', this.get('toggleOnValue')) ;
424       }
425       break ;
426       
427     // set value to on.  change 0 => 1.
428     case SC.TOGGLE_ON_BEHAVIOR:
429       this.set('value', this.get('toggleOnValue')) ;
430       break ;
431       
432     // set the value to false. change 1 => 0
433     case SC.TOGGLE_OFF_BEHAVIOR:
434       this.set('value', this.get('toggleOffValue')) ;
435       break ;
436       
437     // otherwise, just trigger an action if there is one.
438     default:
439       //if (this.action) this.action(evt);
440       var action = this.get('action');
441       var target = this.get('target') || null;
442       if (action)
443       {
444         if (this._hasLegacyActionHandler()) {
445           // old school... 
446           this._triggerLegacyActionHandler(evt);
447         } else {
448           // newer action method + optional target syntax...
449           SC.app.sendAction(action, target, this);
450         }
451       }
452     }
453   },
454   
455   /** @private */
456   _hasLegacyActionHandler: function()
457   {
458     var action = this.get('action');
459     if (action && ($type(action) == T_FUNCTION)) return true;
460     if (action && ($type(action) == T_STRING) && (action.indexOf('.') != -1)) return true;
461     return false;
462   },
463 
464   /** @private */
465   _triggerLegacyActionHandler: function( evt )
466   {
467     if (!this._hasLegacyActionHandler()) return false;
468     
469     var action = this.get('action');
470     if ($type(action) == T_FUNCTION) this.action(evt);
471     if ($type(action) == T_STRING)
472     {
473       eval("this.action = function(e) { return "+ action +"(this, e); };");
474       this.action(evt);
475     }
476   }
