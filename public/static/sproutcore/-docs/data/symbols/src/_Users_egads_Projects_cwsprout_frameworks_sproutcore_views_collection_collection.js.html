490     Find the first content item view for the passed event.
491     
492     This method will go up the view chain, starting with the view that was the 
493     target of the passed event, looking for a child item.  This will become 
494     the view that is selected by the mouse event.
495     
496     This method only works for mouseDown & mouseUp events.  mouseMoved events 
497     do not have a target.
498     
499     @param {Event} evt An event
500     
501   */
502   itemViewForEvent: function(evt)
503   {
504     var view = SC.window.firstViewForEvent( evt );
505     // work up the view hierarchy to find a match...
506     do {
507       // item clicked was the ContainerView itself... i.e. the user clicked outside the child items
508       // nothing to return...
509       if ( view == this ) return null;
510       
511       // sweet!... the view is not only in the collection, but it says we can hit it.
512       // hit it and quit it... 
513       if ( this.hasItemView(view) && (!view.hitTest || view.hitTest(evt)) ) return view;
514     } while ( view = view.get('parentNode') );
515     
516     // nothing was found... 
517     return null;
518   },

568   /**
569     Expands the index into a range of content objects that have the same
570     group value.
571     
572     This method searches backward and forward through your content array for  
573     objects that have the same group value as the object at the index you 
574     pass in.  You can use this method when implementing layoutGroupView to 
575     determine the range of the content that belongs to the group.  
576     
577     Since this method simply searches through the content array, it is really
578     only suitable for content arrays of a few hundred items or less.  If you
579     expect to have a larger size of content array, then you may need to do
580     something custom in your data model to calculate this range in less time.
581     
582     @param {Number} contentIndex index of a content object
583     @returns {Range} a range of objects
584   */
585   groupRangeForContentIndex: function(contentIndex) {
586     var content = Array.from(this.get('content')) ;
587     var len = content.get('length') ;
588     var groupBy = this.get('groupBy') ;
589     if (!groupBy) return { start: 0, length: len } ;
590 
591     var min = contentIndex, max = contentIndex ;
592     var cur = content.objectAt(contentIndex) ;
593     var groupValue = (cur) ? cur.get(groupBy) : null ;
594     
595     // find first item at bottom that does not match.  add one to get start
596     while(--min >= 0) {
597       var cur = content.objectAt(min) ;
598       var curGroupValue = (cur) ? cur.get(groupBy) : null ;
599       if (curGroupValue !== groupValue) break ;
600     }
601     min++ ;
602     
603     // find first item at top that does not match.  keep value to calc range
604     while(++max < len) {
605       var cur = content.objectAt(max) ;
606       var curGroupValue = (cur) ? cur.get(groupBy) : null ;
607       if (curGroupValue !== groupValue) break ;
608     }
609     
610     return { start: min, length: max-min } ;
611   },

624   /**
625     Update the itemViews in the receiver to match the currently visible 
626     content objects.  Normally this method assumes the content objects 
627     themselves have not changed and only updates the views if the range of 
628     visible content has changed.  If you pass true to the fullUpdate property, 
629     then the entire set of itemViews will be revalidated in case any content 
630     objects have changed.
631     
632     @param {Bool} fullUpdate (Optional) if set to true, assumes content has
633       changed and will perform a full update.
634     
635   */
636   updateChildren: function(fullUpdate) {
637     var f ;
638 
639     // if the collection is not presently visible in the window, then there is 
640     // really nothing to do here.  Just mark the view as dirty and return.
641     if (!this.get('isVisibleInWindow')) {
642       this.set('isDirty', true) ;
643       this._needsFullUpdate = this._needsFullUpdate || fullUpdate ;
644       return; 
645     }
646 
647     if (SC.BENCHMARK_UPDATE_CHILDREN) {
648       var bkey = '%@.updateChildren(%@)'.fmt(this, (fullUpdate) ? 'FULL' : 'FAST') ;
649       SC.Benchmark.start(bkey);
650     }
651 
652     this.beginPropertyChanges() ; // avoid sending notifications
653     
654     // STEP 1: Update frame size if needed.  Required to compute the 
655     // clippingFrame.
656     var f ;
657     if ((f = this.computeFrame()) && !SC.rectsEqual(f, this.get('frame'))) {
658       var parent = this.get('parentNode') ;
659       if (parent) parent.viewFrameWillChange() ;
660       this.set('frame', f) ;
661       if (parent) parent.viewFrameDidChange() ;
662       
663       if ((f = this.computeFrame()) && !SC.rectsEqual(f, this.get('frame'))) {
664         this.set('frame', f) ;
665       } 
666     } 
667 
668     // Save the current clipping frame.  If the frame methods are called again
669     // later but the frame has not actually changed, we don't want to run
670     // updateChildren again.
671     var clippingFrame = this._lastClippingFrame = this.get('clippingFrame') ;
672 	
673     // STEP 2: Calculate the new range of content to display in 
674     // the clipping frame.  Determine if we need to do a full update or
675     // not.
676    
677     var range = this.contentRangeInFrame(clippingFrame) ;
678     var content = Array.from(this.get('content'));
679      
680     //make sure the range isn't greater than the content length 
681     //this will prevent trying to render items that aren't really there.
682     range.length = Math.min(SC.maxRange(range), content.get('length')) - range.start ;
683 
684     var nowShowingRange = this.get('nowShowingRange') ;
685     fullUpdate = fullUpdate || (SC.intersectRanges(range, nowShowingRange).length <= 0) ;
686     this.set('nowShowingRange', range) ;
687 
688     // STEP 3: Update item views.
689     var groupBy = this.get('groupBy') ;
690     var didChange = false ;
691     
692     // If this is a fullUpdate, then rebuild the itemViewsByContent hash
693     // from scratch.  This is necessary if the content or the visible range
694     // might have changed.
695     if (fullUpdate) {
696      
697       var itemViewsByContent = {} ; // this will replace the current hash.
698       
699       // iterate through all of the views and insert them.  If the view 
700       // already exists, it will simply be reused.
701       var idx = SC.maxRange(range) ;
702       while(--idx >= range.start) {
703         var c = content.objectAt(idx) ;
704         var key = SC.guidFor(c) ;
705         var itemView = this._insertItemViewFor(c, groupBy, idx) ;
706 
707         if(itemView)
708         {
709           // add item view to new hash and remove from old hash.
710           itemViewsByContent[key] = itemView;
711 
712           delete this._itemViewsByContent[key];
713         }
714       }
715       
716       // Now iterate through the old hash.  Any left over item views should
717       // be removed.
718       for(var key in this._itemViewsByContent) {
719         if (!this._itemViewsByContent.hasOwnProperty(key)) continue ;
720         var itemView = this._itemViewsByContent[key] ;
721         this._removeItemView(itemView, groupBy) ;
722       } ;
723       
724       // Swap out remaining content items.
725       this._itemViewsByContent = itemViewsByContent ;
726       didChange = true;
727       
728     // If a fullUpdate is not required, then we assume no content has changed
729     // and we just need to add or remove some views to bring the ranges up
730     // to date.
731     } else {

1705     // Instead, we just need to save a bunch of state about the mouse down
1706     // so we can choose the right thing to do later.

1709     this._mouseDownEvent = ev ;

1721     // hitTest() on item views so they can implement non-square detection

1872     var content = this.get('content');

1875     var contentLowerBounds = 0;

1904       if (currentMouseDownIndex === previousMouseDownIndex) {

1907       } else if (currentMouseDownIndex > previousMouseDownIndex) {
