1 // ==========================================================================
  2 // Not Empty Validator
  3 // Author: Charles Jolley
  4 //
  5 // This requires all password fields in the form to match.  The normal
  6 // validation only tries to match if all password fields are not empty.
  7 // submit validation requires a match no matter what.
  8 //
  9 // ==========================================================================
 10 
 11 require('validators/validator') ;
 12 
 13 /**
 14   Ensures all fields with the Password validator attached in the same form
 15   contain the same value.
 16   
 17   @class
 18   @extends SC.Validator
 19   @author Charles Jolley
 20   @version 1.0
 21 */
 22 SC.Validator.Password = SC.Validator.extend(
 23 /** @scope SC.Validator.Password.prototype */ {
 24 
 25   attachTo: function(form,field) {
 26     argments.callee.base.call(this,form,field) ;
 27     if (!this.fields) this.fields = [] ;
 28     this.fields.push(field) ;
 29   },
 30 
 31   validate: function(force) {
 32     if (!this.fields || this.fields.length == 0) return true ;
 33     
 34     var empty = false ;
 35     var notEmpty = false ;
 36     var ret = true ;
 37     var value = this.fields[0].get('fieldValue') ;
 38     this.fields.each(function(field) {
 39       var curValue = field.get('fieldValue') ;
 40       if (curValue != value) ret= false ;
 41       if (!curValue || curValue.length == 0) empty = true ;
 42       if (curValue && curValue.length > 0) notEmpty = true ;
 43     }) ;
 44 
 45     // if forces, valid OK if there was an empty.  If not forced, valid OK 
 46     // only if all fields match AND they are not all empty.
 47     if (force) {
 48       return (notEmpty == false) ? false : ret ;
 49     } else {
 50       return (empty == true) ? true : ret ;
 51     }
 52   },
 53   
 54   // update field states
 55   updateFields: function(form,valid) {
 56     if (!this.fields || this.fields.length == 0) return true ;
 57     var err = "Invalid.Password".loc();
 58     var topField = this._field ;
 59     this.fields.each(function(f) {
 60       var msg = (valid) ? null : ((f == topField) ? err : '') ;
 61       form.setErrorFor(f,msg) ;
 62     }) ;
 63     return (valid) ? SC.Validator.OK : err ;
 64   },
 65   
 66   validateChange: function(form, field, oldValue) { 
 67     return this.updateFields(form, this.validate(false)) ;
 68   },
 69 
 70   // this method is called just before the form is submitted.
 71   // field: the field toe validate.
 72   validateSubmit: function(form, field) { 
 73     return this.updateFields(form, this.validate(true)) ;
 74   },
 75 
 76   // this method gets called 1ms after the user types a key (if a change is
 77   // allowed).  You can use this validate the new partial string and return 
 78   // an error if needed.
 79   //
 80   // The default will validate a partial only if there was already an error.
 81   // this allows the user to try to get it right before you bug them.
 82   validatePartial: function(form, field) {
 83     var isInvalid = !this._field.get('isValid') ;
 84     if (isInvalid) {
 85       return this.updateFields(form, this.validate(false)) ;
 86     } else return SC.Validator.NO_CHANGE ;
 87   }
 88     
 89 }) ;
