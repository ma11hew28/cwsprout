  9 /**
 10   @class
 11   @extends SC.Object
 12   @since   0.9.14
 13 
 14   @author Charles Jolley
 15   
 16   _IMPORTANT: SC.Collection currently has several known performance issues.  It 
 17   is suitable for use in existing code, but it is possible that the API will 
 18   change before SproutCore 1.0
 19   
 20   A collection holds a set of records matching the specified conditions. You
 21   can set the data source used to find the objects matching the conditions
 22   to create collections pulled from the local set of objects or pulled 
 23   from the server.
 24   
 25   Collection's automatically update their contents based on the conditions
 26   settings you provide as the state of loaded records change.
 27 */
 28 SC.Collection = SC.Object.extend(
 29 /** @scope SC.Collection.prototype */ {
 30   
 31   // ........................................
 32   // CONFIGURABLE PROPERTIES
 33   //
 34   // Set these properties to control the records that will be in this list.
 35   
 36   /**
 37     Indicates the keys to use to order the records.  If you want the records
 38     ordered by descending order, use a string like 'guid DESC'.
 39     
 40     @type {Array}
 41   */
 42   orderBy: ['guid'],
 43 
 44   /**
 45     Set this to the range of records you are interested in seeing or null if
 46     you want to see all records.  Note that collection actually retrieves
 47     a list of records slightly larger than what you pass here to allow for
 48     members to be added and removed.
 49     
 50     @see SC.Collection.offset
 51     
 52     @type {Number}
 53   */
 54   limit: 0,
 55   
 56   /**
 57     Set this to the offset within the range of records you want to retrieve
 58     this.get('limit') records from.
 59     
 60     @see SC.Collection.limit
 61     
 62     @type {Number}
 63   */
 64   offset: 0,
 65   
 66   /**
 67     Set this to a hash with conditions options. e.g. { active: true }.  If you
 68     don't set this property, then all records of the given type will be
 69     used.
 70     
 71     Note also that if you are matching a property that contains an array, the
 72     condition will match if the key you specify is found in the array.  For 
 73     example  { groups: someGroup } would match any record where someGroup is
 74     included in the groups array.
 75     
 76     @type {Object}
 77   */
 78   conditions: {},
 79   
 80   /**
 81     @property
 82   
 83     This is the actual array of records in the current collection. This
 84     property will change anytime the record members change (but not when
 85     the member record properties change).
 86   
 87     @type {Array}
 88   */
 89   records: function() {
 90     if (this._changedRecords) this._flushChangedRecords() ;
 91     return this._records ;
 92   }.property(),
 93   
 94   /**
 95     @property
 96     
 97     The total count of records matching the conditions settings. The contents
 98     of the records array will be clipped to the range value.
 99     
100     @type {Number}
101   */
102   count: function(key, value) {
103     if (value !== undefined) {
104       this._count = value ;
105     } else if (this._changedRecords) this._flushChangedRecords() ;
106     return this._count || 0 ;
107   }.property(),
108 
109   /**
110     Set to true when the collection is destroyed
111     
112     @type {Boolean}
113   */
114   isDeleted: false, // RO
115   
116   /**
117     Set this to the data source you want to use to get the records.  Use
118     either SC.Store or SC.Server.  If you use SC.Server, your recordType
119     must have a resourceURL property.
120     
121     This should be set when the collection is created and not changed later.
122     
123     @type {SC.Store or SC.Server}
124   */
125   dataSource: SC.Store, // NC
126   
127   /**
128     Set this to the type of record you want in the collection.
129     
130     This should be set when the collection is created and not changed later.
131     
132     @type {SC.Record}
133   */
134   recordType: SC.Record, // NC
135 
136   /**
137     Set to true while a refresh is in progress.
138     
139     @type {Boolean}
140   */
141   isLoading: false, // RO
142   
143   /**
144     Set to true if records have changed in a way that might leave the records out
145     of date.
146     
147     @type {Boolean}
148   */
149   isDirty: false, // RO
150   
151   // ........................................
152   // ACTIONS
153   //
154 
155   /**
156     Call this to force the list to refresh.  The refresh may not happen
157     right away, depending on the dataSource.
158   */
159   refresh: function() {
160     var recordType = this.get('recordType') || SC.Record ;
161     var offset = (this._limit > 0) ? this._offset : 0 ;
162     
163     if (!this._boundRefreshFunc) {
164       this._boundRefreshFunc = this._refreshDidComplete.bind(this) ;
165     }    
166     
167     // start refresh
168     if (!this.dataSource) throw "collection does not have dataSource" ;
169     this.beginPropertyChanges();
170     if (!this.isLoading) this.set('isLoading',true) ;
171     this._refreshing = true ;
172     
173     var order = this.get('orderBy') ;
174     if (order && !(order instanceof Array)) order = [order] ;
175     this.dataSource.listFor({
176       recordType: recordType,
177       offset: offset, 
178       limit: this._limit,
179       conditions: this.get('conditions'), 
180       order: order,
181       callback: this._boundRefreshFunc,
182       cacheCode: this._cacheCode
183     }) ;
184     this.endPropertyChanges() ;
185     return this; 
186   },
187   
188   /**
189     Call this method when you are done with a collection.  This will remove
190     it as an observer to changes in the SC.Store so that it can be reclaimed.
191     isDeleted will also be set to true.
192   */
193   destroy: function() { SC.Store.removeCollection(this); return this; },
194   
195   /**
196     TODO: Needs documentation.
197   */
198   newRecord: function(settings) {
199     if (!settings) settings = {} ;
200     settings.newRecord = true ;
201     settings.dataSource = this.get('dataSource') ;
202     var ret = this.recordType.create(settings);
203     SC.Store.addRecord(ret) ; // this will add the record to the collection.
204     return ret;
205   },
206   
207   // ........................................
208   // INTERNAL
209   //
210   
211   // this is the real offset, limit, and records used by the collection.
212   // this can be a superset of the actual limits seen by outsiders.
213   _offset: 0, _limit: 0, _records: null, _members: null, _store: null,
214   
215   init: function() {
216     arguments.callee.base.call(this) ;
217     SC.Store.addCollection(this) ; // get notified of changes.
218     this._computeInteralOffsetAndLimit() ;
219   },
220   
221   // This is the callback executed when the data source has found the records
222   // matching the passed parameters.  The count is the total count of records
223   // matching the conditions, ignoring the offset and limit.
224   _refreshDidComplete: function(records,count,cacheCode) {
225     if (cacheCode) this._cacheCode = cacheCode;
226 
227     if (records) {
228       this.beginPropertyChanges() ;
229 
230       // update count
231       if (this.get('count') != count) this.set('count',count) ;
232 
233       // update the record store and reslice.
234       this.propertyWillChange('records') ;
235       records = this._store = records.slice() ;
236       this._reslice() ;
237       this.propertyDidChange('records') ;
238 
239       this.endPropertyChanges();
240     }
241     
242     this._refreshing = false ; // only one refresh at a time.
243   },
244 
245   // called by SC.Store whenever an locally stored record has changed state.
246   // This method simply indicates that the records property has changed and saved
247   // the record for later processing.
248   recordDidChange: function(rec) {
249     if (!rec && !rec._guid) return ; // probably an error, but recover anyway.
250     if (!this._changedRecords) this._changedRecords = {} ;
251     this._changedRecords[rec._guid] = rec ;
252     this.propertyWillChange('records') ;
253     this.propertyDidChange('records') ;
254     this.propertyWillChange('count') ;
255     this.propertyDidChange('count') ;
256   },
257 
258   // this method gets called just before the records property is returned.  If 
259   // there are any changed records queued up, the record changes will be 
260   // integrated.
261   _flushChangedRecords: function() {
262     if (!this._changedRecords) return ; // nothing to do.
263 
264     if (this.dataSource != SC.Store) throw "non-local data source is not supported" ;
265 
266     var current = this._store || [] ;
267     var order = this.get('orderBy') || [this.recordType.primaryKey()] ;
268     if (!(order instanceof Array)) order = [order] ;
269     var conditions = this.get('conditions') ;
270 
271     // get the sorted set of changed records, both sorted and as a hash.
272     var records = [] ;
273     var changed = this._changedRecords ;
274     for(var guid in changed) {
275       if (!changed.hasOwnProperty(guid)) continue ;
276       records.push(changed[guid]) ;
277     }
278     records = records.sort(function(a,b){ return a.compareTo(b,order); }) ;    
279     this._changedRecords = null ;
280 
281     // step through the current set of records.  Interpolate changed records, remove records 
282     // that don't belong.
283     var loc = 0 ;
284     while(loc < current.length)  {
285       var working = current[loc] ;
286       var compareToPrev, compareToNext ;
287       
288       // is this record one of the changed?
289       // if so, then the record may need to be removed from its current position if it is deleted,
290       // no longer belongs to the group, or is out of order.
291       if (changed[working._guid]) {
292 
293         var belongs = (!working.get('isDeleted')) && working.matchConditions(conditions) ;
294         if (belongs) {
295           
296           // comes after prev?
297           if (loc>0) {
298             belongs = (working.compareTo(current[loc-1],order) >= 0) ;
299           }
300           
301           // comes before next?
302           if (belongs && (loc+1 < current.length)) {
303             belongs = (working.compareTo(current[loc+1],order) <= 0) ;
304           }
305         }
306         if (!belongs) { current.splice(loc,1); continue; }
307       }
308       
309       // if we get here, then the current record belongs where it is. next see if any of the changed
310       // records need to be inserted here.
311       var goAgain = true ;
312       while ((records.length > 0) && goAgain) {
313         var rec = records[0] ;
314         if ((rec != working) && !rec.get('isDeleted') && rec.matchConditions(conditions)) {
315           if ((rec == working) || (rec.compareTo(working,order) <= 0)) {
316             if (rec != working) current.splice(loc,0,rec) ; // insert only if not the same. 
317             loc++ ;
318           } else goAgain = false ;
319         }
320         if (goAgain) records.shift() ;
321       }
322       
323       // move on to the next record.
324       loc++ ;
325     }
326 
327     // if we get to the end and there are changed records left to process,
328     // they probably need to be added.
329     while(records.length > 0) {
330       var rec = records.shift() ;
331       if (!rec.get('isDeleted') && rec.matchConditions(conditions)) {
332         current.push(rec) ;
333       }
334     }
335         
336 //    if ((this == window.tphotos) && (current.length != this._count)) debugger ;
337     
338     // update the count as well.
339     this._store = current ;
340     this._count = current.length ;
341     this._reslice() ;
342   },
343   
344   // Anytime the properties used to filter the collection change, reslice if possible
345   // then refresh from the data source.
346   propertyObserver: function(observing,target,key,value) {
347     if (target != this) return ;
348 
349     // update the internal properties then refresh.
350     var needsRefresh = false ; 
351     var nv ;
352     value = this.get(key) ; 
353     switch(key) {
354       case 'offset':
355       case 'limit':
356         var oldOffset = this._offset ;
357         var oldLimit = this._limit ;
358         this._computeInteralOffsetAndLimit() ;
359         if ((this._offset == oldOffset) && (this._limit == oldLimit)) {
360           this.propertyWillChange('records') ;
361           this._reslice();
362           this.propertyDidChange('records') ;
363         } else needsRefresh = true ;
364         break ;
365         
366       case 'conditions':
367       case 'orderBy':
368         needsRefresh = true ;
369         break ;
370         
371       default:
372         break ;
373     }
374 
375     // refresh only once per loop. We don't want to refresh multiple times
376     // if the user makes changes to multiple settings.
377     if (needsRefresh && !this._refreshing) {
378       this._refreshing = true ;
379       this._cacheCode = null ;
380       this.set('isLoading',true) ; 
381       this.invokeLater(this.refresh) ;
382     }
383   },
384   
385   // using the current offset and limit properties, compute the internal offset and limit used for
386   // records.
387   _computeInteralOffsetAndLimit: function() {
388     if (this.dataSource != SC.Store) {
389       var v ;
390       this._offset = ((v=this.get('offset')) > this.MARGIN) ? (v-this.MARGIN) : 0 ;
391       this._limit = ((v=this.get('limit')) > 0) ? (v + this.MARGIN) : 0 ; 
392     } else this._offset = this._limit = 0 ;
393   },
394   
395   // This method will slice the _store records based on the current offset and limit.
396   // Is it used internally and thus does not indicate that the computed records
397   // property has changed.
398   _reslice: function() {
399     var offset = this.get('offset') ; 
400     var limit = this.get('limit') ;
401     if ((offset > 0) || (limit > 0)) {
402       var start = offset - this._offset ;
403       var end = start + ((limit <= 0) ? (this._store || []).length : limit) ;
404       this._records = this._store.slice(start,end) ;
405     } else this._records = this._store ;
406   },
407   
408   MARGIN: 10
409   
410 }) ;
