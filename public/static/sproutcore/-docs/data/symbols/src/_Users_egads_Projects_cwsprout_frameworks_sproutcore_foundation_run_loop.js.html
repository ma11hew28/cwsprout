  8 /**
  9   @class
 10 
 11   The run loop provides a universal system for coordinating events within
 12   your application.  The run loop processes timers as well as pending 
 13   observer notifications within your application.
 14   
 15   Typically you will not work with a run loop directly but instead user 
 16   SC.Timer objects and property observing to indirectly trigger actions on
 17   the run loop.  The only time you may need to work with the run loop is if
 18   you implement a setTimeout or event handler yourself.  In these cases, you
 19   should begin and end your function handler with a call to beginRunLoop()
 20   and endRunLoop().  This will give the run loop a chance to process any
 21   pending events on your application.
 22   
 23   h2. Using the Loop Start Time
 24   
 25   Sometimes you need to schedule events such as timers and you want to make
 26   sure all of the events you schedule occur at the same time.  If you want to
 27   keep items in sync you can't use Date.now() because that value will 
 28   increment as your code executes.  Instead, you should get the loop start
 29   time and use that.  For example, if you want to schedule three timers to
 30   repeat until 1 second, 2 seconds and 3 seconds from now you might do:
 31   
 32   {{{
 33     var t1 = SC.Timer.schedule({ 
 34       interval: 100,
 35       action: 'timer1', 
 36       repeats: YES, 
 37       until: SC.runLoop.get('startTime') + 1000 }) ;
 38 
 39     var t2 = SC.Timer.schedule({ 
 40       interval: 100,
 41       action: 'timer1', 
 42       repeats: YES, 
 43       until: SC.runLoop.get('startTime') + 2000 }) ;
 44 
 45     var t3 = SC.Timer.schedule({ 
 46       interval: 100,
 47       action: 'timer1', 
 48       repeats: YES, 
 49       until: SC.runLoop.get('startTime') + 3000 }) ;
 50   }}}
 51   
 52   This will ensure that each timer uses the exact same start time when 
 53   scheduling.  This is critical if you want to keep animations in sync.
 54   
 55   @extends SC.Object
 56   @author Charles Jolley
 57   @version 1.0
 58   @since SproutCore 1.0
 59 */
 60 SC.runLoop = SC.Object.create({
 61   
 62   /**
 63     Maximum time we allow things to run before taking a break.
 64   */
 65   maxRunTime: 3000,
 66   
 67   /**
 68     Called by an observable object to schedule an observer to be notified.
 69     
 70     @param target {Object} the target object
 71     @param action {Function} the method to call
 72     @param args {Array} array of arguments to pass on
 73     @returns {void}
 74   */
 75   notifyObserver: function(target, action, args) {
 76     if (!this._notifications) this._notifications = [] ;
 77     this._notifications.push({ target: target, action: action, args: args });
 78   },
 79   
 80   /**
 81     Tries to flush observer notifications.  The idea is that this will
 82     continue to remove observers until all of the notifications have flushed
 83     or until the max execution time has passed.  Called automatically from
 84     endRunLoop().
 85     
 86     The way this works, it does not unshift pending notifications but
 87     instead just sets them to null until the entire queue is flushed.  Then
 88     it will reset the queue to a new array.  This is faster than shifting().
 89     
 90     @returns {void}
 91   */
 92   deliverNotifications: function() {
 93     if (!this._notifications || this._notifications.length <=0) return;
 94     if (this._notifying) return ; // do not allow recursion
 95     this._notifying = YES ;
 96     
 97     var start = this.get('startTime') ;
 98     var max = start + this.get('maxRunTime') ;
 99     var loc = 0;
100     while((Date.now() < max) && (loc < this._notifications.length)) {
101       var notify = this._notifications[loc] ;
102       this._notifications[loc] = null;
103       loc++ ; // IMPORTANT: Make sure you increment so that we always go on.
104 
105       if (notify) {
106         var args = notify.args;
107         notify.action.call(notify.target, args[0], args[1], args[2], args[3]);
108       }
109     }
110     
111     // if we made it through and the queue is empty, reset it
112     // otherwise, leave the queue in place and schedule another runloop.
113     if (loc >= this._notifications.length) {
114       this._notifications = [];
115     } else {
116       this.invokeLater(this.deliverNotifications) ;
117     }
118     
119     this._notifying = NO ;
120   },
121   
122   /**
123     Call this method whenver you begin executing code.  
124   
125     This is typically invoked automatically for you from event handlers and 
126     the timeout handler.  If you call setTimeout() or setInterval() yourself, 
127     you may need to invoke this yourself.
128   
129     @returns {void}
130   */
131   beginRunLoop: function() {
132     this._start = Date.now() ;  
133   },
134 
135   /**
136     Call this method whenever you are done executing code.
137     
138     This is typically invoked automatically for you from event handlers and
139     the timeout handler.  If you call setTimeout() or setInterval() yourself
140     you may need to invoke this yourself.
141     
142     @returns {void}
143   */
144   endRunLoop: function() {
145     // send any pending notifications.
146     //this.deliverNotifications() ;
147     
148     // flush any expired timers, possibly cancelling the timeout.
149     this._flushExpiredTimers() ;
150     this._start = null ;
151   },
152 
153   /**
154     The time the current run loop began executing.
155     
156     All timers scheduled during this run loop will begin executing as if 
157     they were scheduled at this time.
158   
159     @type {Number}
160     @field
161   */
162   startTime: function() {
163     if (!this._start) this._start = Date.now();
164     return this._start ;  
165   }.property(),
166 
167   // timers are stored in the hash with a double linked list to keep them
168   // in order.
169   scheduleTimer: function(timer, runTime) {
170     
171     if (!timer) throw "scheduleTimer requires a timer" ; // nothing to do
172     
173     if (!this._timers) this._timers = {} ;
174     var guid = SC.guidFor(timer) ;
175 
176     // either remove the timer record or create a new one and add it.
177     var t = this._timers[guid];
178     if (t) {
179       if (t.prev) t.prev.next = t.next ;
180       if (t.next) t.next.prev = t.prev ;
181       t.next = t.prev = null ;
182       t.at = runTime ;
183     } else {
184       t = this._timers[guid] = { 
185         timer: timer, 
186         at: runTime, 
187         guid: guid,
188         next: null, prev: null 
189       } ;
190     }
191 
192     // now walk the chain to figure out where to insert the timer.  If the
193     // timer goes at the front, also reschedule the next timeout.
194     var cur = this._next ;
195     if (!cur || cur.at > runTime) {
196       this._next = t ;
197       t.next = cur ;
198       if (cur) cur.prev = t.next ;
199       this._rescheduleTimeout() ;
200       
201     } else {
202       // find the item to insert after
203       while(cur.next && cur.next.at <= runTime) cur = cur.next ;
204       t.next = cur.next ;
205       if (cur.next) cur.next.prev = t ;
206       cur.next = t ;
207       t.prev = cur ;
208     }
209   },
210   
211   cancelTimer: function(timer) {
212     
213     if (!timer) return ; //nothing to do
214     
215     if (!this._timers) this._timers = {} ;
216     var guid = SC.guidFor(timer) ;
217     var t = this._timers[guid] ;
218     
219     // if a timer record was found, remove it from the list.  
220     // if timer was at the front, reschedule the timeout.
221     if (t) {
222       if (t.next) t.next.prev = t.prev ;
223       if (t.prev) t.prev.next = t.next ;
224       if (this._next === t) {
225         this._next = t.next ;
226         this._rescheduleTimeout() ;
227       }
228       t.next = t.prev = t.timer = null ; // clear objects
229     }
230   },
231 
232   timerPausedStateDidChange: function(timer) {
233     this._rescheduleTimeout() ;
234   },
235   
236   // determines the next time the timeout needs to trigger and reschedules
237   // if necessary.  If you pass in the next timer to use, then it will be 
238   // scheduled instead of searching the timers.
239   _rescheduleTimeout: function() {
240     
241     // if we are currently flushing, don't do this since it will happen
242     // later anyway
243     if (this._flushing) return ;
244     
245     if (!this._timers) this._timers = {} ;
246     
247     // find next timer to trigger.  If the first unpaused timer.
248     var rec = this._next ;
249     while(rec && (!rec.timer || rec.timer.get('isPaused')) ) rec = rec.next ;
250 
251     // if no next timer was found, then cancel any timer.
252     if (!rec) {
253       this._timeoutAt = 0 ;
254       if (this._timeout) clearTimeout(this._timeout) ;
255       this._timeout = null ;
256 
257     // determine if we need to reschedule
258     } else if ((this._timeoutAt === 0) || (rec.at !== this._timeoutAt)) {
259       if (this._timeout) clearTimeout(this._timeout) ;
260 
261       var delay = Math.max(rec.at - Date.now(),0) ;
262       this._timeout = setTimeout(this._timeoutAction, delay) ;
263       this.timeoutAt = rec.at ;
264     }
265   },
266   
267   // called when the timeout is executed.  Find any timers that have 
268   // expired and call them.  
269   _timeoutAction: function() {
270     var rl = SC.runLoop;
271     rl._timeout = null; rl._timeoutAt = 0 ;
272     rl.beginRunLoop() ;
273     rl._flushExpiredTimers() ;
274     rl.endRunLoop() ;
275   },
276   
277   // finds any timers that might have expired.  This will also find the
278   // next timer to execute and reschedule it if needed.
279   _flushExpiredTimers: function() {
280     if (!this._timers) this._timers = {} ;
281     var now = this.get('startTime') ;
282     var max = now + this.get('maxRunTime');  // max time we are allowed to run timers
283 
284     this._flushing = YES ;
285     
286     // work down the list, do not fire a timer more than once per loop.
287     var fired = {} ;
288     var rec = this._next ;
289     while(rec && (rec.at <= now) && (Date.now() < max)) {
290       
291       // if rec has been fired, go on to next one.
292       var guid = SC.guidFor(rec.timer) ;
293       if (fired[guid]) {
294         rec = rec.next;
295         
296       // otherwise, remove rec from list and then fire it.
297       } else {
298         var next = rec.next;
299         
300         if (this._next === rec) this._next = rec.next ;
301         if (rec.next) rec.next.prev = rec.prev ;
302         if (rec.prev) rec.prev.next = rec.next ;
303         delete this._timers[rec.guid] ;
304         
305         fired[guid] = YES ; 
306         if (rec.timer) rec.timer.fire() ;
307         
308         // finish clean up.
309         rec.next = rec.prev = rec.timer = null ;
310         rec = next ;
311       }
312     }
313     
314     this._flushing = NO ;
315     
316     // schedule next timer if needed.
317     this._rescheduleTimeout() ;
318   }
319   
320   
321 }) ;
