  9 SC.STRING_TITLEIZE_REGEXP = (/([\s|\-|\_|\n])([^\s|\-|\_|\n]?)/g);

 11 /**
 12   @namespace
 13   
 14   SproutCore implements a variety of enhancements to the built-in String 
 15   object that make it easy to perform common substitutions and conversions.
 16   
 17   Most of the utility methods defined here mirror those found in Prototype
 18   1.6.
 19   
 20   @since SproutCore 1.0
 21 */
 22 SC.String = {
 23   
 24   // Interpolate string. looks for %@ or %@1; to control the order of params.
 25   /**
 26     Apply formatting options to the string.  This will look for occurrences
 27     of %@ in your string and substitute them with the arguments you pass into
 28     this method.  If you want to control the specific order of replacement, 
 29     you can add a number after the key as well to indicate which argument 
 30     you want to insert.  
 31 
 32     Ordered insertions are most useful when building loc strings where values
 33     you need to insert may appear in different orders.
 34 
 35     h3. Examples
 36     
 37     {{{
 38       "Hello %@ %@".fmt('John', 'Doe') => "Hello John Doe"
 39       "Hello %@2, %@1".fmt('John', 'Doe') => "Hello Doe, John"
 40     }}}
 41     
 42     @param args {Object...} optional arguments
 43     @returns {String} formatted string
 44   */
 45   fmt: function() {
 46     // first, replace any ORDERED replacements.
 47     var str = this.gsub(/%@([0-9]+)/, function(m) {
 48       return (arguments[parseInt(m[1],0)-1] || '').toString(); 
 49     }) ;
 50 
 51     // now, replace any remaining %@ items.  Use this indexOf() method b/c
 52     // it is faster than split().
 53     var ret = [] ;
 54     var idx = -1 ;
 55     var loc = 0 ;
 56     var argIdx = 0;
 57     while((idx = str.indexOf("%@",loc)) >= 0) {
 58      // slice off initial part of string and push into ret. update loc.
 59      ret.push(str.slice(loc,idx)) ;
 60      loc = idx + 2 ; // 2 to skip '%@'.
 61      
 62      // add in replacement.
 63      var value = arguments[argIdx++] ;
 64      if (value && value.toString) value = value.toString() ;
 65      ret.push(value) ;
 66     }
 67     
 68     // include any remaining bits of the string.
 69     if (loc < str.length) {
 70       ret.push(str.slice(loc,str.length)) ;
 71     }
 72     
 73     // join return value.
 74     return (ret.length > 1) ? ret.join('') : ret[0] ;
 75   },
 76 
 77   /**
 78     Localizes the string.  This will look up the reciever string as a key 
 79     in the current Strings hash.  If the key matches, the loc'd value will be
 80     used.  The resulting string will also be passed through fmt() to insert
 81     any variables.
 82     
 83     @param args {Object...} optional arguments to interpolate also
 84     @returns {String} the localized and formatted string.
 85   */
 86   loc: function() {
 87     // NB: This could be implemented as a wrapper to locWithDefault() but
 88     // it would add some overhead to deal with the arguments and adds stack
 89     // frames, so we are keeping the implementation separate.
 90     
 91     var kit = String[String.currentLanguage()];
 92     var str = kit[this] ;
 93     if (!str) str = String.English[this] || this ;
 94     return str.fmt.apply(str,arguments) ;
 95   },
 96 
 97   /**
 98     Works just like loc() except that it will return the passed default 
 99     string if a matching key is not found.
100     
101     @param def {String} the default to return
102     @param args {Object...} optional formatting arguments
103     @returns {String} localized and formatted string
104   */
105   locWithDefault: function(def) {
106     var kit = String[String.currentLanguage()];
107     var str = kit[this] ;
108     if (!str) str = String.English[this] || def ;
109     var args = SC.$A(arguments) ;
110     args.shift() ; // escape def.
111     return str.fmt.apply(str, args) ;
112   },
113   
114   /** 
115     Capitalizes a string.
116     @return {String} capitalized string
117   */
118   capitalize: function() {
119     return this.charAt(0).toUpperCase() + this.substring(1) ;
120   },
121 
122   /**
123     Capitalized every word in the string, separated by spaces or dashes.
124     @return {String} titleized string.
125   */
126   titleize: function() {
127     return this.replace(SC.STRING_TITLEIZE_REGEXP, 
128       function(str,separater,character) { 
129         return (character) ? (' ' + character.toUpperCase()) : ' ';
130       }).capitalize() ;
131   },
132   
133   /**
134     Camelizes a string.  This will take any words separated by spaces, dashes
135     or underscores and convert them into camelCase.
136     
137     h3. Examples
138     
139     {{{
140       "my favorite items".camelize() => "myFavoriteItems"
141       "css-class-name".camelize() => "cssClassName"  
142       "action_name".camelize() => "actionName" 
143     }}}
144     
145     @returns {String} camelized string
146   */
147   camelize: function() {
148     var parts = this.split('-'), len = parts.length;
149     if (len == 1) return parts[0];
150 
151     var camelized = this.charAt(0) == '-'
152       ? parts[0].charAt(0).toUpperCase() + parts[0].substring(1)
153       : parts[0];
154 
155     for (var i = 1; i < len; i++)
156       camelized += parts[i].charAt(0).toUpperCase() + parts[i].substring(1);
157 
158     return camelized;
159   },
160 
161   
162   /**
163     Converts the string into a class name.  This method will camelize your 
164     string and then capitalize the first letter.
165   */
166   classify: function() {
167     return this.camelize().capitalize() ;
168   },
169   
170   /**
171     Converts a camelized string into all lower case separated by underscores.
172     
173     @returns {String} the decamelized string.
174   */
175   decamelize: function() { 
176     return this.replace(/([a-z])([A-Z])/g,'$1_$2').toLowerCase();
177   },
178 
179   /**
180     Converts a camelized string or a string with spaces or underscores into
181     a string with components separated by dashes.
182     
183     @returns {String} the dasherized string.
184   */
185   dasherize: function() {
186     return this.decamelize().replace(/[ _]/g,'-') ;  
187   },
188   
189   /**
190     Converts a camelized string or a string with dashes or underscores into
191     a string with components separated by spaces.
192     
193     @returns {String} the humanized string.
194   */
195   humanize: function() {
196     return this.decamelize().replace(/[-_]/g,' ') ;
197   },
198   
199   /**
200     Removes any extra whitespace from the edges of the strings.
201     
202     This method is also aliased as strip().
203     
204     @returns {String} the trimmed string
205   */
206   trim: function () {
207     return this.replace(/^\s+|\s+$/g,"");
208   },
209 
210   /**
211     Explodes a string into an array of characters.
212     
213     @returns {Array}
214   */
215   toArray: function() {
216     return this.split('');
217   }
218   
219 } ;

228 // Add strings for various languages to this collection.  String.loc()
229 // method will try to localize the string passed using the current language.
230 // if the language is not available, it will use English.
231 Object.extend(String,
232 /** @scope String @static */ {
233 
234   /**
235     The current browser language as a two letter code.
236   */
237   browserLanguage: ((navigator.language || navigator.browserLanguage).split('-', 1)[0]),
238   
239   /**
240     If YES, localization will favor the detected language instead of the
241     preferred one.
242   */
243   useAutodetectedLanguage: NO,
244   
245   /**
246     This property is set by the build tools to the current build language.
247   */
248   preferredLanguage: null,
249   
250   /**
251     Returns the hash key to use for loc strings.  The default implementation
252     will autodetect the browser language and look for a loc string to 
253     match.  If it can't find one then it will introspect to find loc strings
254     that are defined and use those instead.
255   */
256   currentLanguage: function () {
257     var ret = (this.useAutodetectedLanguage) ? (this.browserLanguage || this.preferredLanguage || 'en') : (this.preferredLanguage || this.browserLanguage || 'en') ;
258 
259     // then try a couple of normalized forms...
260     if (!this[ret]) ret = this.normalizedLanguage(ret);
261     return ret ;
262   },
263   
264   /**
265     Returns a normalized language string for the two letter country code.
266   */
267   normalizedLanguage: function(ret) {
268     switch(ret) {
269       case 'fr':
270         ret = 'French'; 
271         break ;
272       case 'de':
273         ret = 'German'; 
274         break ;
275       case 'ja':
276       case 'jp':
277         ret = 'Japanese'; 
278         break ;
279       case 'en':
280         ret = 'English' ;
281         break ;
282       
283       case 'es':
284         ret = 'Spanish' ;
285         break;
286         
287       default:
288         ret = "English";
289         break ;
290     }
291     return ret;
292   },
293   
294   /**
295     Adds loc strings for the named language.  This method takes care of 
296     creating the localized string hash if it does not already exist.
297     The language can be one of the following or any two-letter country code.
298     
299     English, French, German, Japanese, Spanish
300     
301     @param language {String} the language code
302     @param strings {Hash} hash of loc strings.
303     @returns {this}
304   */
305   addStringsFor: function(language, strings) {    
306     // convert language to a normalized name...
307     language = String.normalizedLanguage(language) ;
308     if (!String[language]) String[language] = {} ;
309     Object.extend(String[language], strings || {}); 
310     return this;
311   }
312 
313 });
