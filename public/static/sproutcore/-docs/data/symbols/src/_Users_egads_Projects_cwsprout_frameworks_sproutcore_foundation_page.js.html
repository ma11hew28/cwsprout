  6 require('foundation/path_module') ;
  7 
  8 /**
  9   @class SC.Page
 10   
 11   The Page object is used to store a set of views that can be lazily 
 12   configured on page load.  Page assumes that all of its properties are
 13   basically outlets and will try to configure them as such.
 14 
 15   @extends SC.Object
 16 */
 17 SC.Page = SC.Object.extend(
 18   /** @scope SC.Page.prototype */
 19   {
 20   
 21   get: function(key) {
 22     var value = this[key] ;
 23     if (value && (value instanceof Function) && (value.isOutlet)) {
 24       var ret = this.outlet(key) ;
 25       if (SC.window && !ret.parentNode) {
 26         SC.window._insertBefore(ret, null, false) ;
 27         SC.window._rebuildChildNodes();
 28       }
 29       ret.awake() ;
 30       return ret ;
 31     } else return sc_super() ;
 32   },
 33   
 34   // in addition to activating bindings, calling awake on the page object
 35   // will cause any outlet properties to be loaded.
 36   awake: function() {
 37     arguments.callee.base.call(this) ;
 38     for(var key in this) {
 39       if (this.hasOwnProperty(key) && this[key] && this[key].isOutlet) {
 40         this.get(key) ; // convert to outlet.
 41       } 
 42     }
 43   },
 44   
 45   init: function() {
 46     sc_super() ;
 47     var el = this.rootElement = $('resources') ;
 48     SC.callOnLoad(function() {
 49       if (el && el.parentNode) el.parentNode.removeChild(el) ;
 50 			el = null;
 51     }) ;
 52   },
 53   
 54   // returns the property, but only if it is already configured.
 55   getIfConfigured: function(key) {
 56     var value = this[key] ;
 57     if (value && (value instanceof Function) && (value.isOutlet)) {
 58       return null ;
 59     } else return value ;
 60   },
 61   
 62   _insertBefore: function() {},
 63   _rebuildChildNodes: function() {}
 64   
 65 }) ;
 66 
 67 Object.extend(SC.Page.prototype, SC.PathModule) ;
