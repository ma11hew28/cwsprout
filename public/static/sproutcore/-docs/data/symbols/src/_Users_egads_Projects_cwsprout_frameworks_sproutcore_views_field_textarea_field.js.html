  6 require('views/field/text_field') ;

  8 /**
  9   @class
 10   
 11   Manages a text area field.
 12   
 13   @extends SC.TextFieldView
 14   @author Skip Baney
 15   @version 1.0
 16 */
 17 SC.TextareaFieldView = SC.TextFieldView.extend(
 18 /** @scope SC.TextareaFieldView.prototype */ {
 19   
 20   emptyElement: '<textarea></textarea>',
 21   didBecomeFirstResponder: function() {
 22     if (this.get('isVisibleInWindow')) {
 23       this.rootElement.focus();
 24 	    this.invokeLater(this._selectRootElement, 1) ;
 25     }
 26     // hide the hint text if it is showing.
 27     this._updateFieldHint() ;
 28   },
 29 
 30   insertNewline: function(evt) { evt._stopWhenHandled = false; return true; }
 31 
 32 });
