1 // ==========================================================================
  2 // SC.PathModule
  3 // Author: Charles Jolley
  4 //
  5 // When creating a web app, you often need to find elements in your page to
  6 // configure them on load.  These elements are often marked with a class or
  7 // id.  General purpose methods such as $$() or getElementsByClassName are
  8 // not very good for this because they search entire trees of nodes.
  9 //
 10 // These methods are much faster because they make some assumptions about 
 11 // how your elements are structured.
 12 //
 13 // ==========================================================================

 22 /**
 23   PathModule is high-performance API for locating DOM nodes in your HTML
 24   document.  It's designed to be used by searching elements from a parent
 25   node.  
 26 
 27   PathModule is patched into Prototypes Element.Methods, which means that 
 28   you can automatically access this API for any DOM element retrieved via
 29   the $() operator.  You can also mixin the PathModule api to your own 
 30   objects.  Be sure to also have a property called rootElement that points to 
 31   the root DOM element you want to search.
 32 */

109 // This version of PathModule can be added to Element.Methods among others.
110 // It is the same API except that it passes the element as the first parameter.
111 SC._PathModule = {
112 
113   // this is the core method that needs to be implemented for each object.
114   // this method will search the receiver (up to _levels_ deep) for _max_
115   // elements for which the function func() returns true. 
116   //
117   // el:     (req) the root element.  If you pass null, document will be used. 
118   // func:   (req) the function to be used to check each element.
119   // levels: (opt) the max levels deep in nested elements. defaults: NO_LIMIT
120   // max:    (opt) the maximum number of elements to return. defaults: 
121   //         NO_LIMIT
122   // nest:   (opt) if true, then the children of matched elements will also be
123   //         searched.  if your element matches will never be nested, setting
124   //         this to false can be much faster.  defaults: false
125   $$func: function(el, func, levels, max, nest) {
126     // defaults
127     levels = levels || NO_LIMIT ; max = max || NO_LIMIT; nest = nest || false;
128     
129     // function to search this node.
130     var searchNode = function(node, depth, remain, includeThisNode) {
131       var ret = [] ;
132       
133       // does this node match?
134       // we extend any returned elements so you can chain these methods. 
135       // on browsers that support HTMLElement.prototype, this will do nothing.
136       var match = (includeThisNode) ? func(node) : false ;
137       if (match) { ret.push(node); remain--; }
138 
139       // can we search children?
140       depth-- ;
141       if ((match && !nest) || (remain<=0) || (depth<=0)) return ret;
142 
143       // search children.
144       node = node.firstChild ;
145       while(node && (remain > 0)) {
146         var found = searchNode(node,depth,remain,true) ;
147         remain -= found.length ;
148         ret = ret.concat(found) ;
149         node = node.nextSibling ;
150       }
151       return ret ;
152     } ;
153     
154     return searchNode(el || document, levels+1, max, false) ;
155   },
156   
157   // finds all nodes with class name.
158   $$C: function(el, className, levels, max, nest) {
159     return SC._PathModule.$$func(el, function(node) {
160       return Element.hasClassName(node, className) ; 
161     }, levels, max, nest) ;
162   },
163   
164   // find all nodes with tag name
165   $$T: function(el, tagName, levels, max, nest) {
166     tagName = tagName.toUpperCase() ;
167     return SC._PathModule.$$func(el, function(node) { 
168       return node.tagName == tagName; 
169     }, levels, max, nest) ;
170   },
171 
172   // find all nodes with an attribute matching then named value.
173   $$P: function(el, property, value, levels, max, nest) {
174     return SC._PathModule.$$func(el, function(node) { 
175       var pvalue = (node.getAttribute) ? node.getAttribute(property) : node[property] ;
176       return pvalue == value; 
177     }, levels, max, nest) ;
178   },
179 
180   // this will parse the selector and then execute the selectors to return
181   // the result.  this understands a very simple form for selectors. the 
182   // parts are split by spaces. and treated like so:
183   // '.classname' = class name
184   // 'tag' = tag
185   // '#id' = an id. (this will bypass the normal selector methods) 
186   // 'attr=value' = find an attr with the value.
187   //
188   // understands the following qualifiers:
189   // .selector:3   => find .selector at most 3 levels deep
190   // .selector:3:4 => find at most 4 .selectors at most 3 levels deep
191   // .selector?    => find at most 1 .selector, any depth.
192   // .selector?3   => find at most 3 .selectors, any depth
193   // .selector:*:4 => find at more 4 .selectors at any depth.
194   // 
195   // You can also directly access child nodes by doing:
196   // .selector[0][1][3] -> get child 3 of child 1 of child 0.
197   // or:
198   // [0][2][3] -> search base.
199   //
200   $$S: function(el,selector, levels, max, nest) {
201     var parts = selector.split(' ') ;
202     var ret = [el] ; var nextMax = null ; var bits ;
203     var blevels; var bmax; var q; var indicies ;
204 
205     parts.each(function(part) {
206       // this is included for compatibility with earlier changes I made to
207       // prototype.  the preferred way to do this in the future is with with
208       // the colon format.
209       if (part == '?') { nextMax = 1; return ; }
210 
211       // get defaults
212       blevels = levels; bmax = (nextMax) ? nextMax : max ;
213       nextMax = null ;
214 
215       // if there are indexes, find those and save them.
216       if (part.slice(part.length-1,part.length) == ']') {
217         bits = part.split('[') ;
218         part = bits.shift();
219         indicies = bits.map(function(b) {return parseInt(b.slice(0,-1),0);});
220       } else indicies = null ;
221 
222       // find found modifiers
223       bits = part.split('?') ;
224       part = bits[0] ;
225       if (bits.length > 1) {
226         bmax = (bits[1].length == 0) ? 1 : parseInt(bits[1],0) ;
227       }
228 
229       // find level modifiers
230       bits = part.split(':') ;
231       part = bits[0] ;
232       if (bits.length > 1) blevels = (bits[1] == '*') ? '*' : parseInt(bits[1],0) ;
233       if (bits.length > 2) bmax = (bits[2] == '*') ? '*' : parseInt(bits[2],0) ;
234 
235       // convert star to NO_LIMIT
236       if (blevels == '*') blevels = NO_LIMIT ;
237       if (bmax == '*') bmax = NO_LIMIT ;
238 
239       // prepare the selector info.
240       // q tells us how to process the first part.
241       // part is the first part of the selector
242       // bits are any extra css selectors.
243       if (part && part.length > 0) {
244         q = part.slice(0,1) ; 
245         if (['.','#'].indexOf(q) >= 0) { 
246           part = part.slice(1,part.length) ; 
247         }
248         
249         bits = part.split('.') ;
250         part = bits[0] ;
251         bits = bits.slice(1,bits.length) ;
252 
253         // optimization for # types.
254         var fret = null ;
255         if (q == '#') {
256           fret = $(part) ;
257           fret = (fret) ? [fret] : null ;
258           if (fret) ret = fret ;
259         }
260         
261         // otherwise pass this off to $$func to actually search.
262         if (fret == null) {
263           fret = [] ;
264           for(var i=0;i<ret.length;i++) {
265 
266             var found = SC._PathModule.$$func(ret[i],function(node) {
267               var match ;
268               switch(q) {
269                 case '.':
270                   NODE = node ;
271                   var elementClassName = node.className;
272                   if (elementClassName && (elementClassName.length > 0)) {
273                     match = (elementClassName == part ||
274                       elementClassName.match( new RegExp("(^|\\s)" + part + 
275                         "(\\s|$)"))) ;
276                   } else match = false ;
277                   break ;
278                 case '#':
279                   match = node.id == part ;
280                   break ;
281                 default:
282                   if (node.tagName) {
283                     match = (node.tagName.toLowerCase() == part.toLowerCase()) ;
284                   } else { match = false; }
285               }
286 
287               // if the first item matched, verify that the others match as well.
288               var ilen = bits.length ;
289               for(var i=0; match && i < ilen; i++) {
290                 if (!Element.hasClassName(node,bits[i])) match = false ; 
291               }
292               return match ; 
293             }, blevels, bmax, nest) ;
294             fret = fret.concat(found) ;
295           }
296           ret = fret ;
297         }
298       } else if ($type(ret) != T_ARRAY) ret = [ret] ;
299 
300       // now follow indicies, if there are any
301       if (indicies && indicies.length > 0 && ret) {
302         ret = ret.map(function(el){
303           var iloclen = indicies.length ;
304           for(var iloc=0; el && (iloc < iloclen); iloc++) {
305             el = el.childNodes[indicies[iloc]] ;
306           }
307           return el;
308         }) ;
309       }
310     }) ;
311     return ret ;
312   },
313 
314   // finds the first node for which func returns true.
315   $func: function(el,func, levels) {
316     var ret = SC._PathModule.$$func(el,func,levels,1,false) ;
317     return (ret.length>0) ? ret[0] : null ;
318   },
319 
320   // finds the first node with class name.  returns element, not array.
321   $C: function(el, className, levels) {
322     var ret = SC._PathModule.$$C(el, className,levels,1,false) ;
323     return (ret.length>0) ? ret[0] : null ;
324   },
325 
326   // finds the first node with tag name.  returns element, not array.
327   $T: function(el, tagName, levels) {
328     var ret = SC._PathModule.$$T(el, tagName,levels,1,false) ;
329     return (ret.length>0) ? ret[0] : null ;
330   },
331 
332   // find first node with an attribute matching then named value.
333   $P: function(el, attr, value, levels) {
334     var ret = SC._PathModule.$$P(el, attr, value,levels,1,false) ;
335     return (ret.length>0) ? ret[0] : null ;
336   },
337 
338   // find first node matching the specified class selector.
339   $S: function(el, selector, levels) {
340     var ret = SC._PathModule.$$S(el, selector,levels,1,false) ;
341     return (ret.length>0) ? ret[0] : null ;
342   },
343   
344   // find an element and configure it as the named view if it is not already
345   // configured.
346   $$view: function(el, selector, viewClass, levels, max, nest) {
347     //if (!viewClass) viewClass = SC.View;
348     var ret ;
349     if (selector && (typeof(selector) != 'string')) {
350       ret = [selector] ;
351     } else {
352       ret = SC._PathModule.$$S(el,selector,levels,max,nest) ;
353     }
354     if (ret) ret = ret.map(function(x) { return (viewClass) ? viewClass.viewFor(x) : SC.View.findViewForElement(x); }) ;
355     return ret ;
356   },
357 
358   // find the first element matching the selector and create a view if it is
359   // not already configured.
360   $view: function(el, selector, viewClass, levels) {
361     var ret = SC._PathModule.$$view(el, selector, viewClass, levels, 1, false) ;
362     return (ret.length>0) ? ret[0] : null ;
363   }
