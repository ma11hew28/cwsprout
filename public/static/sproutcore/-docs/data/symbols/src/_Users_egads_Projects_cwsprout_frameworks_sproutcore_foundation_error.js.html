  8 /**
  9   @class
 10   
 11   An error, used to represent an error state.
 12   
 13   Many API's within SproutCore will return an instance of this object whenever
 14   they have an error occur.  An error includes an error code, description,
 15   and optional human readable label that indicates the item that failed. 
 16   
 17   Depending on the error, other properties may also be added to the object
 18   to help you recover from the failure.
 19   
 20   You can pass error objects to various UI elements to display the error in
 21   the interface. You can easily determine if the value returned by some API is 
 22   an error or not using the helper SC.$ok(value).
 23   
 24   @extends SC.Object
 25   @since SproutCore 1.0
 26 */
 27 SC.Error = SC.Object.extend(
 28 /** @scope SC.Error.prototype */ {
 29   
 30   /**
 31     error code.  Used to designate the error type.
 32   */
 33   code: -1,
 34   
 35   /**
 36     Human readable description of the error.  This can also be a non-localized
 37     key.
 38   */
 39   description: '',
 40   
 41   /**
 42     Human readable name of the item with the error.
 43   */
 44   label: null
 45 }) ;
 46 
 47 /**
 48   Creates a new SC.Error instance with the passed description, label, and
 49   code.  All parameters are optional.
 50   
 51   @param description {String} human readable description of the error
 52   @param label {String} human readable name of the item with the error
 53   @param code {Number} an error code to use for testing.
 54   @returns {SC.Error} new error instance.
 55 */
 56 SC.Error.desc = function(description, label, code) {
 57   var opts = { description: description } ;
 58   if (label !== undefined) opts.label = label ;
 59   if (code !== undefined) opts.code = code ;
 60   return this.create(opts) ;
 61 } ;
 62 
 63 /**
 64   Shorthand form of the SC.Error.desc method.
 65 */
 66 SC.$error = function(description, label, c) { 
 67   return SC.Error.desc(description,label, c); 
 68 } ;
 69 var $error = SC.$error ; // export globally
 70 
 71 /**
 72   Returns YES if the passed value is an error object, otherwise NO.
 73 */
 74 SC.$ok = function(ret) {
 75   return (ret !== false) && ($type(ret) != T_ERROR) ;
 76 };
 77 var $ok = SC.$ok ; // export globally.
