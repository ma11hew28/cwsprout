  6 /**
  7   @namespace
  8   
  9   The inline editor delegate receives notifications from the inline text
 10   editor before, during, and after the user completes inline editing.
 11   
 12   The inline editor delegate is used by views that work with the inline
 13   editor.  You may need to implement this protocol if you want to
 14   use the inline editor in your own custom views.
 15   
 16   @since SproutCore 1.0
 17 */
 18 SC.InlineEditorDelegate = {
 19   
 20     /**
 21      This is a  classname you can apply to the inline editor field
 22      to configure it's styling, in addition to the the editor's 
 23      default style-cloning behavior.
 24       
 25       @property inlineEditorClassName {String} A class name to use with the inline editor.
 26     */
 27     inlineEditorClassName: "",
 28   
 29   
 30     /**
 31       Called just before the inline edit displays itself but after it has been 
 32       configured for display.  
 33       
 34       You can use this method to make last minute changes to the display of 
 35       the inline editor or to collect its value.
 36       
 37       @param inlineEditor {SC.InlineTextFieldView} The inline editor.
 38       @returns {void}
 39     */
 40     inlineEditorWillBeginEditing: function(inlineEditor) {},
 41 
 42     /**
 43       Called just after an inline editor displays itself.
 44       
 45       You can use this method to perform any hiding or other view changes
 46       you need to perform on your own view to make room for the new editor.
 47       
 48       Note tht editors are placed over the top of views in the page, not 
 49       inside of them from a DOM perspective.
 50       
 51       @param inlineEditor {SC.InlineTextFieldView} The inline editor.
 52       @returns {void}
 53     */
 54     inlineEditorDidBeginEditing: function(inlineEditor) {},
 55     
 56     /**
 57       Called just before an inline editor tries to end editing and hide 
 58       itself.
 59       
 60       You can use this method to control whether the inline editor will
 61       actually be allowed to end editing.  For example, you might disallow
 62       the editor to end editing if the new value fails validation.
 63       
 64       @param inlineEditor {SC.InlineTextFieldView} the inline editor
 65       @param finalValue {Object} the final value
 66       @returns {Boolean} YES to allow the editor to end editing.
 67     */
 68     inlineEditorShouldEndEditing: function(inlineEditor, finalValue) {
 69       return YES ;
 70     },
 71     
 72     /**
 73       Called just after the inline editor has ended editing. You can use this 
 74       method to save the final value of the inline editor and to perform any 
 75       other cleanup you need to do.
 76       
 77       @param inlineEditor {SC.InlineTextFieldView} the inline editor
 78       @param finalValue {Object} the final value
 79       @returns {void}
 80     */
 81     inlineEditorDidEndEditing: function(inlineEditor, finalValue) {}
 82 };
