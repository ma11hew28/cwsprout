 18   // set this property to your current app lication
 19   location: function(key,value) {
 20     if (value !== undefined) {
 21       if (value === null) value = '' ;
 22       
 23       // convert an object hash to a string, if it was passed.
 24       if (typeof(value) == "object") {
 25 
 26         // get the original route and any params
 27         var parts = (value.route) ? value.route.split('&') : [''] ;
 28         var route = parts.shift() ;
 29         var params = {} ;
 30         parts.each(function(p) {
 31           var bits = p.split('=') ;
 32           params[bits[0]] = bits[1] ;
 33         }) ;
 34 
 35         // overlay any params passed in the object.
 36         for(var key in value) {
 37           if (!value.hasOwnProperty(key)) continue ;
 38           if (key != 'route') {
 39             params[key] = encodeURIComponent(''+value[key]) ;
 40           }
 41         }
 42         
 43         // now build params.
 44         parts = [route] ;
 45         for(var key in params) {
 46           if (!params.hasOwnProperty(key)) continue ;
 47           parts.push([key,params[key]].join('=')) ;          
 48         }
 49         
 50         // and combine.
 51         value = parts.join('&') ;
 52       }
 53       
 54       if (this._location != value) {
 55         this._location = value ;
 56         this._setWindowLocation(value) ;
 57         //this.gotoRoute(value) ;
 58       }
 59     }
 60     return this._location ;
 61   }.property(),
 62   
 63   // ensures we are at the current route location.
 64   ping: function() { 
 65     if (!this._didSetupHistory) {
 66       this._didSetupHistory = true ;
 67       this._setupHistory() ;
 68     }
 69     this._checkWindowLocation();

 72   // register a route here.  Routes have the following format:
 73   // static/route/path -- matches this path only.
 74   // static/route/:path -- matches any static/route, :path passed as param.
 75   // static/*route -- matches any static, route gets rest of URL.
 76   //
 77   // parameters can also be passed using &.
 78   // static/route¶m1=value¶m2=value2
 79   //
 80   addRoute: function(route, func) {
 81     var parts = route.split('/') ;
 82     if (!this._routes) this._routes = SC.Routes._Route.create() ;
 83     this._routes.addRoute(parts,func) ;
 84   },
 85   
 86   // eval routes.
 87   gotoRoute: function(route) {
 88     var params = {} ; var parts, route, func ;
 89 
 90     // save this route for window location sensing
 91     this._lastRoute = route ;
 92     
 93     // step 1: split out parameters
 94     var parts = route.split('&') ;
 95     if (parts && parts.length > 0) {
 96       route = parts.shift() ;
 97       parts.each(function(part) {
 98         var param = part.split('=') ;
 99         if (param && param.length > 1) params[param[0]] = decodeURIComponent(param[1]) ;
100       }) ;
101     } else route = '' ;
102     
103     // step 2: split our route parts
104     parts = route.split('/') ;
105     
106     // step 3: evaluate route.
107     if (!this._routes) this._routes = SC.Routes._Route.create() ;
108     func = this._routes.functionForRoute(parts,params) ;
109     if (func) func(params) ;
110     //else console.log('could not find route for: "'+route+'"') ;
111   },

114   init: function() {
115     arguments.callee.base.call(this) ;
116     if (SC.isSafari() && !SC.isSafari3()) {
117       Object.extend(this,this.browserFuncs.safari) ;  
118     } else if (SC.isIE()) {
119       Object.extend(this,this.browserFuncs.ie) ;  
120     }
121     this._didSetupHistory = false ;
122   },

124   // _checkWindowLocation and _setWindowLocation are implemented separately for
125   // each browser.  Below are the implementations, which get copied during init.
126   browserFuncs: {
127     
128     // for Safari2 and earlier.
129     safari: {
130 
131       _setupHistory: function() {
132         // get initial cloc.
133         var cloc = location.hash ;
134         cloc = (cloc && cloc.length > 0) ? cloc.slice(1,cloc.length) : '' ;
135         this._cloc = cloc ;
136 
137         // create back stack.
138         this._backStack = [] ;
139         this._backStack.length = history.length ;
140         this._backStack.push(cloc) ;
141 
142         // create forward stack.
143         this._forwardStack = [] ;
144 
145         this.invokeLater(this._checkWindowLocation, 1000) ;
146       },
147       
148       _checkWindowLocation: function() { 
149         // The way we know the user has moved forward or back in the history 
150         // is when the length of the history array no longer matched our own 
151         // copy of the history. However, when we first change locations, it 
152         // takes a little while for Safari to catch up.  So what we do instead 
153         // is first check to see if Safari's length has changed from its last 
154         // known length and only then check for a delta.
155         var historyDidChange = (history.length - this._lastLength) != 0;
156         var delta = (historyDidChange) ? (history.length - this._backStack.length) : 0 ;
157         this._lastLength = history.length ;
158 
159         if (historyDidChange) console.log('historyDidChange') ;
160         
161         // if the history length has changed, then we need to move forward or 
162         // back in the history.
163         if (delta) {
164           if (delta < 0) { // back button has been pushed
165             
166             // shift out the current loc.
167             this._forwardStack.push(this._cloc) ; 
168             
169             // shift out other items.
170             for(var i=0; i < Math.abs(delta+1);i++) {
171               this._forwardStack.push(this._backStack.pop()) ;
172             }
173             
174             // set new cloc.
175             this._cloc = this._backStack.pop() ;
176             
177             
178           } else { // forward button has been pushed
179             
180             // shift out the current loc.
181             this._backStack.push(this._cloc) ;
182             
183             for(var i=0; i < (delta-1); i++) {
184               this._backStack.push(this._forwardStack.pop()) ;
185             }
186             
187             this._cloc = this._forwardStack.pop() ;
188           }
189 
190         // if the history has changed but the delta hasn't, then that means
191         // a new location was set via _setWindowLocation().  Normally we would
192         // call gotoRoute in that method, but doing so will crash Safari.
193         // Instead, we wait until Safari registers the change and then do the
194         // route change.
195         } else if (historyDidChange && this._locationDidChange) {
196           this.gotoRoute(this._cloc) ;
197           this._locationDidChange = false ;
198         }
199 
200         var cloc = this._cloc ;
201         var loc = this.get('location') ;
202         if (cloc != loc) {
203           this.set('location',(cloc) ? cloc : '') ;
204           this.gotoRoute(cloc) ;
205         }
206         
207         this.invokeLater(this._checkWindowLocation, 50) ;
208       },
209 
210       _setWindowLocation: function(loc) {        
211         var cloc = this._cloc ;
212         if (cloc != loc) {
213           this._backStack.push(this._cloc) ;
214           this._forwardStack.length = 0 ;
215           this._cloc = loc ;          
216           location.hash = (loc && loc.length > 0) ? loc : '' ;
217           this._locationDidChange = true ;
218         }
219       }            
220       
221     },

226         this.invokeLater(this._checkWindowLocation, 1000) ;
227       },

230         var loc = this.get('location') ;
231         var cloc = location.hash ;
232         cloc = (cloc && cloc.length > 0) ? cloc.slice(1,cloc.length) : '' ;
233         if (cloc != loc) this.set('location',(cloc) ? cloc : '') ;
234         this.invokeLater(this._checkWindowLocation, 100) ;
235       },

238         //console.log('_setWindowLocation('+loc+')') ;

240         cloc = (cloc && cloc.length > 0) ? cloc.slice(1,cloc.length) : '' ;
241         if (cloc != loc) {
242           location.hash = (loc && loc.length > 0) ? loc : '#' ;
243         }
244         this.gotoRoute(loc) ;
245       }
246     }

273   // This object handles a single route level.  
274   _Route: SC.Object.extend({
275     
276     // a route that ends here gets this func.
277     _func: null,
278     
279     // staticly named routes.
280     _static: null,
281     
282     // dynamically named routes.
283     _dynamic: null,
284     
285     // set the wildcard route name here.
286     _wildcard: null,
287     
288     addRoute: function(parts, func) {
289       if (!parts || parts.length == 0) {
290         this._func = func ;
291         
292       // add to route table.
293       } else {
294         var part = parts.shift() ; // get next route.
295         var nextRoute = null ;
296         switch(part.slice(0,1)) {
297           
298           // add a dynamic route
299           case ':':
300             part = part.slice(1,part.length) ;
301             var routes = this._dynamic[part] || [] ;
302             nextRoute = SC.Routes._Route.create() ;
303             routes.push(nextRoute) ;
304             this._dynamic[part] = routes ;
305             break ;
306             
307           // setup wildcard route
308           case '*':
309             part = part.slice(1,part.length) ;
310             this._wildcard = part ;
311             this._func = func ;
312             break ;
313             
314           // setup a normal static route.
315           default:
316             var routes = this._static[part] || [] ;
317             nextRoute = SC.Routes._Route.create() ;
318             routes.push(nextRoute) ;
319             this._static[part] = routes ;
320         }
321         
322         // if we need to go another level deeper, call nextRoute
323         if (nextRoute) nextRoute.addRoute(parts,func) ;
324       }
325     },
326     
327     // process the next level of the route and pass on.
328     functionForRoute: function(parts, params) {
329       // if parts it empty, then we are here, so return func
330       if (!parts || parts.length == 0) {
331         return this._func ;
332         
333       // process the next part
334       } else {
335         var part = parts.shift() ;
336         var routes, nextRoute, ret, loc ;
337         
338         // try to match to static
339         routes = this._static[part] ;
340         if (routes) for(loc=0;(loc < routes.length) && (ret==null);loc++) {
341           var clone = parts.slice() ;
342           ret = routes[loc].functionForRoute(clone, params) ;
343         }
344         
345         // try to match dynamic if no static match was found.
346         if (ret == null) for(var key in this._dynamic) {
347           routes = this._dynamic[key] ;
348           if (routes) for(loc=0;(loc<routes.length) && (ret == null);loc++) {
349             var clone = parts.slice() ;
350             ret = routes[loc].functionForRoute(clone,params) ;
351             
352             // if a route was found, save the current part in params.
353             if (ret && params) params[key] = part ;
354           }
355           if (ret) break ; 
356         }
357         
358         // if nothing still found, and there is a wildcard, match that.
359         if ((ret == null) && this._wildcard) {
360           parts.unshift(part) ;
361           if (params) params[this._wildcard] = parts.join('/') ;
362           ret = this._func ;
363         }
364         
365         return ret ;
366       }
367     },
368     
369     init: function() {
370       arguments.callee.base.call(this) ;
371       this._static = {} ; this._dynamic = {} ;
372     }
373   })
