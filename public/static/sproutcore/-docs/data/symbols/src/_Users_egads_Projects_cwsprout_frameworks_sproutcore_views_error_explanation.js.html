  6 require('views/label');
  7 
  8 /**
  9   @class
 10   
 11   The ErrorExplanation view is a special type of label view that can display
 12   one or more errors related to a form or field.  This view will set itself
 13   to visible only if it has errors.
 14   
 15   @extends SC.View
 16   @extends SC.Control
 17   
 18 */
 19 SC.ErrorExplanationView = SC.View.extend(SC.Control,
 20 /** @scope SC.ErrorExplanationView.prototype */ {
 21 
 22   emptyElement: '<ul class="errors"></ul>',
 23   explanationTemplate: '<li>%@</li>',
 24   
 25   _errorsFor: function(errors) {
 26     if (!errors || errors.length == 0) return [] ;
 27     return errors.map(function(er) {
 28       return ($type(er) == T_ERROR) ? er : null ;
 29     }).compact() ;
 30   },
 31   
 32   valueBindingDefault: SC.Binding.Multiple,
 33   
 34   formatter: function(errors, view) {
 35     errors = view._errorsFor(errors) ;
 36     if (!errors || errors.length == 0) return '' ;
 37     return errors.map(function(er) {
 38       er = er.get('description') ; 
 39       if (er.escapeHTML) er = er.escapeHTML() ;
 40       return view.explanationTemplate.fmt(er); 
 41     }).join("") ;
 42   },
 43   
 44   escapeHTML: false,
 45   
 46   _valueObserver: function() {
 47     var errors = this._errorsFor(this.get('value')) ;
 48     var isVisible = errors && errors.length > 0 ;
 49     if (this.get('isVisible') != isVisible) this.set('isVisible',isVisible);
 50     this.set('innerHTML', this.formatter(errors, this)) ;
 51   }.observes('value'),
 52   
 53   init: function() {
 54     sc_super() ;
 55     this._valueObserver() ;
 56   }
 57   
 58 });
