 12 /**
 13   @class
 14   
 15   The inline text editor is used to display an editable area for controls 
 16   that are not always editable such as label views and source list views.
 17   
 18   You generally will not use the inline editor directly but instead will
 19   invoke beginEditing() and endEditing() on the views yous are 
 20   editing. If you would like to use the inline editor for your own views, 
 21   you can do that also by using the editing API described here.
 22   
 23   h2. Using the Inline Editor in Your Own Views
 24 
 25   If you need to use the inline editor with custom views you have written,
 26   you will need to work with the class methods to begin, commit, and discard
 27   editing changes.
 28   
 29   h3. Starting the Editor
 30   
 31   The inline editor works by positioning itself over the top of your view 
 32   with the same offset, width, and font information.  As the user types, the
 33   field will automatically resize vertically to make room for the user's text.
 34   
 35   To activate the inline editor you must call beginEdition() with at least 
 36   the target view you want the editor to position itself on top of:
 37   
 38   {{{
 39     SC.InlineTextFieldView.beginEditing({
 40       target: view, validator: validator
 41     }) ;
 42   }}}
 43 
 44   You can pass a variety of options to this method to configure the inline
 45   editor behavior, including:
 46 
 47   - *frame* The editors initial frame in viewport coordinates (REQ)
 48   - *exampleElement* A DOM element to use when copying styles.
 49   - *delegate* Optional delegate to receive update notices.  If not passed, the target view will be treated as the delegate. (REQ)
 50   - *value* The initial value of the edit field.  If not passed, the value property of the target view will be used instead.
 51   - *multiline* If YES then the hitting return will add to the value instead of exiting the inline editor.
 52   - *selectedRange* The range of text that should be selected.  If omitted, then the insertion point will be placed at the end of the value.
 53   - *commitOnBlur* If YES then bluring will commit the value, otherwise it will discard the current value.  Defaults to YES.
 54   - *validator* Optional validator will be attached to the field.
 55   
 56   If the inline editor is currently in use elsewhere, it will automatically
 57   close itself over there and begin editing for your view instead.  The 
 58   editor expects your source view to implement the InlineTextFieldViewDelegate
 59   protocol.
 60 
 61   h2. Commiting or Discarding Changes
 62   
 63   Normally the editor will automatically commit or discard its changes 
 64   whenever the user exits the edit mode.  If you need to force the editor to
 65   end editing, you can do so by calling commitEditing() or discardEditing():
 66   
 67   {{{
 68     SC.InlineTextFieldView.commitEditing();
 69     SC.InlineTextFieldView.discardEditing();
 70   }}}
 71   
 72   Both methods will try to end the editing context and will call the 
 73   relevent delegate methods on the delegate you passed to beginEditing().
 74   
 75   Note that it is possible an editor may not be able to commit editing 
 76   changes because either the delegate disallowed it or because its validator
 77   failed.  In this case commitEditing() will return NO.  If you want to
 78   end editing anyway, you can discard the editing changes instead by calling
 79   discardEditing().  This method will generally suceed unless your delegate
 80   refuses it as well.
 81   
 82   @extends SC.View
 83   @extends SC.DelegateSupport
 84   @extends SC.InlineEditorDelegate
 85   @since SproutCore 1.0
 86 */
 87 SC.InlineTextFieldView = SC.View.extend(SC.DelegateSupport, SC.InlineEditorDelegate,
 88 /** @scope SC.InlineTextFieldView.prototype */ {
 89 
 90   /**
 91     Invoked by the class method to begin editing on an inline editor.
 92     
 93     You generally should call the class method beginEditing() instead of
 94     this one since it will make sure to create and use the shared editor
 95     instance.
 96 
 97     @params options {Hash} hash of options for editing
 98     @returns {Boolean} YES if editor began editing, NO if it failed.
 99   */
100   beginEditing: function(options) {
101     
102     // end existing editing if necessary
103     this.beginPropertyChanges();
104     if (this.get('isEditing') && !this.blurEditor()) {
105       this.endPropertyChanges();  return NO ;
106     }
107 
108     this._optframe = options.frame ;
109     this._exampleElement = options.exampleElement ;
110     this._delegate = options.delegate ;
111 
112     if (!this._optframe || !this._delegate) {
113       throw "At least frame and delegate options are required for inline editor";
114     }
115     
116     this._originalValue = options.value || '' ;
117     this._multiline = (options.multiline !== undefined) ? options.multiline : NO ;
118     this._commitOnBlur =  (options.commitOnBlur !== undefined) ? options.commitOnBlur : YES ;
119 
120     // set field values
121     var field = this.outlet('field') ;
122     field.set('validator', options.validator) ;
123     field.set('value', this._originalValue) ;
124     field.set('selectedRange', options.selectedRange || { start: this._originalValue.length, length: 0 }) ;
125 
126     this.set('isEditing', YES) ;
127     
128     // add to window.
129     SC.app.get("keyPane").appendChild(this) ;
130     
131     // get style for view.
132     this.updateViewStyle() ;
133 
134     var del = this._delegate ;
135 
136     this._className = this.getDelegateProperty(del,"inlineEditorClassName");
137     if(this._className && !this.hasClassName(this._className)) {
138         this.setClassName(this._className,true);
139       }
140     
141     this.invokeDelegateMethod(del, 'inlineEditorWillBeginEditing', this) ;
142     this.resizeToFit(field.getFieldValue()) ;
143 
144     // allow notifications to go
145     this.endPropertyChanges() ;
146     
147     // and become first responder
148     this.field.becomeFirstResponder() ;
149 
150     this.invokeDelegateMethod(del, 'inlineEditorDidBeginEditing', this) ;
151   },
152   
153   
154   /**
155     Tries to commit the current value of the field and end editing.  
156     
157     Do not use this method, use the class method instead.
158   */
159   commitEditing: function() {
160     // try to validate field.  If it fails, return false.  
161     var field = this.outlet('field') ;
162     if (!$ok(field.validateSubmit())) return NO ;
163     return this._endEditing(field.get('value')) ;
164   },
165   
166   /**
167     Tries to discard the current value of the field and end editing.
168     
169     Do not use this method, use the class method instead.
170   */
171   discardEditing: function() {
172     return this._endEditing(this._originalValue) ;
173   },
174   
175   /**
176     Invoked whenever the editor loses (or should lose) first responder 
177     status to commit or discard editing.
178   */
179   blurEditor: function() {
180     if (!this.get('isEditing')) return YES ;
181     return (this._commitOnBlur) ? this.commitEditing() : this.discardEditing();  
182   },
183   
184   /** @private
185     Called by commitEditing and discardEditing to actually end editing.
186 
187     @returns {Boolean} NO if editing did not exit
188   */
189   _endEditing: function(finalValue) {
190     if (!this.get('isEditing')) return YES ;
191     
192     // get permission from the delegate.
193     var del = this._delegate ;
194     if (!this.invokeDelegateMethod(del, 'inlineEditorShouldEndEditing', this, finalValue)) return NO ; 
195 
196     // OK, we are allowed to end editing.  Notify delegate of final value
197     // and clean up.
198     this.invokeDelegateMethod(del, 'inlineEditorDidEndEditing', this, finalValue) ;
199 
200     // If the delegate set a class name, let's clean it up:
201     if(this._className) this.setClassName(this._className, false);
202     
203     // cleanup cached values
204     this._originalValue = this._delegate = this._exampleElement =  this._optframe = this._className = null ;
205     this.set('isEditing', NO) ;
206 
207     // resign first responder if not done already.  This may call us in a 
208     // loop but since isEditing is already NO, nothing will happen.
209     if (this.field.get('isFirstResponder')) this.field.resignFirstResponder();
210     if (this.get('parentNode')) this.removeFromParent() ;
211     
212     return YES ;
213   },
214   
215   /**
216     YES if the editor is currently visible and editing.
217   
218     @type {Boolean}
219   */
220   isEditing: NO,
221   
222   /** @private */
223   emptyElement: [
224     '<div class="sc-inline-text-field-view">',
225       '<div class="sizer"></div>',
226       '<textarea class="inner-field" wrap="virtual"></textarea>',
227     '</div>'
228   ].join(''),
229 
230   /**
231     Collects the appropriate style information from the targetView to 
232     make the inline editor appear similar.
233   */
234   updateViewStyle: function() {
235     // collect font and frame from target.
236     var f= this._optframe ;
237     var el = this._exampleElement ;
238     
239     var styles = {
240       fontSize:       Element.getStyle(el,'font-size'),
241       fontFamily:     Element.getStyle(el,'font-family'),
242       fontWeight:     Element.getStyle(el,'font-weight'),
243       paddingLeft:    Element.getStyle(el,'padding-left'),
244       paddingRight:   Element.getStyle(el,'padding-right'),
245       paddingTop:     Element.getStyle(el,'padding-top'),
246       paddingBottom:  Element.getStyle(el,'padding-bottom'),
247       lineHeight:     Element.getStyle(el,'line-height'),
248       textAlign:      Element.getStyle(el,'text-align')
249     } ;
250       
251     var field = this.outlet('field') ;
252     var sizer = this.outlet('sizer') ;
253     
254     field.setStyle(styles) ;
255     
256     styles.opacity = 0 ;
257     sizer.setStyle(styles) ;
258     sizer.recacheFrames() ;
259     
260     this.set('frame', f) ;
261   },
262   
263   /**
264     Resizes the visible textarea to fix the actual text in the text area.
265     
266     This method works by keeping a div positioned immediately beneath the 
267     text area with an opacity of 0 that contains the same text as the 
268     input text field itself.  This is then used to calculate the required 
269     size for the text area.
270   */
271   resizeToFit: function(newValue)
272   {
273     var sizer  = this.outlet('sizer');
274     var field  = this.outlet('field');
275     
276     // XSS attack waiting to happen... escape the form input;
277     var text = (newValue || '').escapeHTML();
278 
279     // convert the textarea's newlines into something comparable for the sizer 
280     // div appending a space to give a line with no text a visible height.
281     text = text.replace(/  /g, "  ").replace(/\n/g, "<br /> ");
282     
283     // get the text size
284     sizer.set('innerHTML', text || " ");
285     sizer.recacheFrames() ;
286     var h = sizer.get('frame').height;
287     this.set('frame', { height: h }) ;
288   },
289 
290   /** 
291     The actual text field view used for editing.
292   */
293   field: SC.TextareaFieldView.extend({
294 
295     // handle mouseDown event if we are currently editing so that events
296     // don't go any further?
297     mouseDown: function(e) {
298       arguments.callee.base.call(this, e) ;
299       return this.owner.get('isEditing');
300     },
301 
302     // [Safari] if you don't take key focus away from an element before you 
303     // remove it from the DOM key events are no longer sent to the browser.
304     willRemoveFromParent: function() {
305       this.get('rootElement').blur();
306     },
307     
308     // ask owner to end editing.
309     willLoseFirstResponder: function()
310     {
311       // should have been covered by willRemoveFromParent, but this was needed 
312       // too.
313       this.get('rootElement').blur();
314       return this.owner.blurEditor() ;
315     },
316 
317     // invoked when the user presses escape.  Returns true to ignore
318     // keystroke
319     cancel: function() { 
320       this.owner.discardEditing(); 
321       return YES;
322     },
323     
324     // do it here instead of waiting on the binding to make sure the UI
325     // updates immediately.
326     fieldValueDidChange: function(partialChange) {
327       arguments.callee.base.call(this, partialChange) ;
328       this.owner.resizeToFit(this.getFieldValue()) ;
329     },
330     
331     // invoked when the user presses return.  If this is a multi-line field,
332     // then allow the newine to proceed.  Otherwise, try to commit the 
333     // edit.
334     insertNewline: function(evt) { 
335       if (this.owner._multiline) {
336         return arguments.callee.base.call(this, evt) ;
337       } else {
338         this.owner.commitEditing() ;
339         return YES ;
340       }
341     },
342     
343     // Tries to find the next key view when tabbing.  If the next view is 
344     // editable, begins editing.
345     
346     insertTab: function(evt)
347     {
348       var next = this.get("owner")._delegate.nextValidKeyView();
349       this.owner.commitEditing() ;
350       if(next) next.beginEditing();
351       return YES ;
352     },
353 
354     insertBacktab: function(evt)
355     {
356       var prev = this.get("owner")._delegate.previousValidKeyView();
357       this.owner.commitEditing() ;
358       if(prev) prev.beginEditing();
359       return YES ;
360     }
361 
362   }).outletFor('.inner-field?'),
363   
364   sizer: SC.View.outletFor('.sizer?')
365 
366 });

369 SC.InlineTextFieldView.mixin(
370 /** @static SC.InlineTextFieldView */ {
371   /** Call this method to make the inline editor begin editing for your view.
372       
373       If the inline editor is already being used for another value it will
374       try to dismiss itself from the other editor and attach itself to the
375       new view instead.  If this process fails for some reason (for example
376       if the other view did not allow the view to end editing) then this
377       method will return false.
378 
379       You should pass a set of options that at least includes the target
380       view.  See class definition for options.
381       
382       @params options {Hash} hash of options for editing
383       @returns {Boolean} YES if editor began editing, NO if it failed.
384   */
385   beginEditing: function(options) {
386     if (!this.sharedEditor) this.sharedEditor = this.create() ;
387     return this.sharedEditor.beginEditing(options) ;
388   },
389   
390   /** Save the current value of the inline editor and exit edit mode.
391   
392     If the inline editor is being used it will try to end the editing and
393     close.  If the inline editor could not end for some reason (for example
394     if the delegate did not allow the editing to end) then this method will
395     return NO.
396     
397     @returns {Boolean} YES if the inline editor ended or no edit was in 
398       progress.
399   */
400   commitEditing: function() {
401     return (this.sharedEditor) ? this.sharedEditor.commitEditing() : YES ;
402   },
403 
404   /** Discard the current value of the inline editor and exit edit mode.
405   
406     If the inline editor is in use, this method will try to end the editing,
407     restoring the original value of the target view.  If the inline editor
408     could not end for some reason (for example if the delegate did not 
409     allow editing to end) then this method will return NO.
410     
411     @returns {Boolean} YES if the inline editor ended or no edit was in progress.
412   */
413   discardEditing: function() {
414     return (this.sharedEditor) ? this.sharedEditor.discardEditing() : YES ;  
415   },
416   
417 
418   
419   /**
420     The current shared inline editor.  This property will often remain NULL
421     until you actually begin editing for the first time.
422     
423     @type {SC.InlineTextFieldView}
424   */
425   sharedEditor: null
426   
427 }) ;
