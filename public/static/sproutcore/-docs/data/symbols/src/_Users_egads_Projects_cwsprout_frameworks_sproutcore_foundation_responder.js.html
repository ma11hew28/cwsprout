 10 /**
 11   @class 
 12   
 13   A responder is capable of responding to events such as mouse clicks and key 
 14   presses.  
 15   
 16   All views are a type of responder, but you can extend this class yourself 
 17   as well to implement your own responders if you need to.
 18   
 19   @extends SC.Object
 20   
 21 */
 22 SC.Responder = SC.Object.extend(
 23   /** @scope SC.Responder.prototype */
 24   {
 25     
 26   // .......................................................................
 27   // Managing Responder States
 28   //
 29   
 30   /**  
 31     This method will be called just before you become first responder to see
 32     if you want to be first responder.  If you don't want to handle
 33     keyboard events, return false. (this is the default).  Note that this is
 34     a property.
 35   */
 36   acceptsFirstResponder: false,
 37   
 38   /**
 39     This property is the next responder that should be allowed to handle
 40     events if you don't want to handle it.  For views, this is automatically
 41     set to your parentNode.
 42   */
 43   nextResponder: null,
 44   
 45   /**  
 46     This property is true whenever you are the first responder.  Observable.
 47   */
 48   isFirstResponder: false,
 49 
 50   pane: null,
 51   
 52   /**
 53     Call this method to become the first responder.  You will now be the
 54     first to receive keyboard events, etc.  Return false if you could not
 55     be made first responder for some reason.
 56   */
 57   becomeFirstResponder: function()
 58   {
 59     if (!this.get('acceptsFirstResponder')) return false; // responder doesn't want it
 60 
 61     var pane = this.get('pane');
 62     if (!pane) return false;                              // there's no pane of which to become FR.. doesn't matter if it *does* want it...
 63 
 64     if (pane.get('firstResponder') == this) return true;  // we already got it... and setting it again will trigger the willResign callback
 65 
 66     pane.set('firstResponder',this);
 67     return true;
 68   },
 69   
 70   /**
 71     Call this method if you no longer want to be first responder.  Normally
 72     you don't need to do this since you will lose firstResponder status
 73     when the user focuses a different view.
 74   */
 75   resignFirstResponder: function()
 76   {
 77     var pane = this.get('pane');
 78     if (!pane) return false;
 79     
 80     if (pane.get('firstResponder') != this) return false;
 81     
 82     pane.set('firstResponder',null);
 83     return true;
 84   },
 85   
 86   noResponderFor: function() {},
 87   
 88   /**
 89     This method will be called just before you become the first responder
 90     so you can make changes.
 91   */
 92   didBecomeFirstResponder: function() {},
 93   
 94   /**
 95     This method will be called just before you lose first responder status.
 96   */
 97   willLoseFirstResponder: function() {},

100   // .......................................................................
101   // Keyboard Events
102   //
103   
104   /**
105     Set this value to an SC.InputManager instance if you want to use a 
106     custom input manager for this specific instance.  If you want to override
107     the input manager used for all responders, replace SC.Responder.inputManager.
108   */
109   inputManager: function()
110   {
111     return SC.Responder.inputManager;
112   }.property(),
113 
114   /**
115     this method is invoked on the firstResponder whenever the user presses
116     a key.  You can interpret this yourself or call interpretKeyEvents()
117     which will use the inputManager to handle the event.  The default
118     version just passes this onto the next responder.
119   */
120   keyDown: function(evt) {
121     var nr = this.get('nextResponder') ;
122     return (nr && nr.keyDown) ? nr.keyDown(evt) : false;
123   },
124   
125   /**  
126     This method is invoked on the firstResponder whenever the user releases
127     a key.  Default version just passes onto the next responder.
128   */
129   keyUp: function(evt) {
130     var nr = this.get('nextResponder') ;
131     return (nr && nr.keyUp) ? nr.keyUp(evt) : false ;
132   },
133   
134   /**  
135     This method is invoked whenever the user changes the flag keys that are
136     pressed.  pressedFlags is a hash.  The default version passes on to the
137     nextRespoonder.
138   */
139   flagsChanged: function(pressedFlags, evt) {
140     //var nr = this.get('nextResponder') || SC.window.get('firstResponder') ;
141     var nr = this.get('nextResponder') ;
142     return (nr && nr.flagsChanges) ? nr.flagsChanges(pressedFlags, evt) : false;
143   },
144 
145   /**
146     This will be called if a keydown event isn't handled by the responder chain.
147   */
148   performKeyEquivalent: function(keystring, evt) {
149     return false;
150   },
151 
152   /*
153     call this method with one or more event objects to process the incoming
154     text.  This will use the current inputManager (see above) to process
155     the event and then to make an appropriate call on the receiver.
156   */
157   interpretKeyEvents: function(evts) {
158     var inputManager = this.get('inputManager');
159     if (inputManager) return inputManager.interpretKeyEvents(evts, this) ;
160     return false ;
161   },
162 
163   /**
164   * Attempt to perform a method. If either the method does not exist, or the method returns false, send the request to the nextResponder.
165   * @param {String} method The name of the method to perform
166   * @return {SC.Responder} The responder that handled the command, or false
167   */
168   doCommand: function(method)
169   {
170     var responder = this;
171     var args      = SC.$A(arguments);
172     var method    = args.shift();
173     var aliases   = this._commandAliases[method];
174     var handled   = false;
175     
176     do {
177       // attempt to handle the method
178       if (this._responderHandledCommand(responder, method, args)) return responder;
179       // perhaps under an alias?
180       if (aliases)
181       {
182         for (var i=0, n=aliases.length; i < n; i++)
183         {
184           if (this._responderHandledCommand(responder, aliases[i], args)) return responder;
185         }
186       }
187       // nope... next responder, please...
188     } while (responder = responder.get('nextResponder'));
189     // ran out of responders... 
190     return false;
191   },
192   
193   /** @private */
194   _responderHandledCommand: function( responder, method, args )
195   {
196     // if the method does not explicitly return a false-y value (false, "", 0), it's considered "handled"
197     // WARNING: returning null, or void is *not* == false
198     return (responder.respondsTo(method) && (responder[method].apply(responder, args) != false));
199   },
200   /** @private */
201   _commandAliases: {
202     'mouseDown':   ['didMouseDown'],
203     'mouseUp':     ['didMouseUp'],
204     'doubleClick': ['didDoubleClick'],
205     'click':       ['didClick'],
206     'mouseDown':   ['didMouseDown']
207   }
