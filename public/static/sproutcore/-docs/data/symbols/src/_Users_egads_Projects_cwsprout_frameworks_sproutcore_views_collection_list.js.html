  6 require('views/label');

  9 /** @class
 10 
 11   A list view renders vertical lists of items.  It is a specialized form of
 12   collection view that is simpler than the table view, but more refined than
 13   a generic collection.
 14   
 15   You can use a list view just like a collection view, except that often you
 16   also should provide a default rowHeight.  Setting this value will allow 
 17   the ListView to optimize its rendering.
 18 
 19   @extends SC.CollectionView
 20   @author    Charles Jolley  
 21   @version 1.0
 22 */
 23 SC.ListView = SC.CollectionView.extend(
 24 /** @scope SC.ListView.prototype */ {
 25   
 26   emptyElement: '<div class="list-view"></div>',
 27   
 28   /** 
 29     The common row height for list view items.
 30     
 31     If you set this property, then the ListView will be able to use this
 32     property to perform absolute layout of its children and to minimize t
 33     number of actual views it has to create.
 34     
 35     The value should be an integer expressed in pixels.
 36   */
 37   rowHeight: 20,
 38   
 39   /**
 40     The default example item view will render text-based items.
 41     
 42     You can override this as you wish.
 43   */
 44   exampleView: SC.ListItemView,
 45   
 46   insertionOrientation: SC.VERTICAL_ORIENTATION,
 47   
 48   contentRangeInFrame: function(frame) {
 49     var rowHeight = this.get('rowHeight') || 0 ;
 50     var min = Math.max(0,Math.floor(SC.minY(frame) / rowHeight)-1) ;
 51     var max = Math.ceil(SC.maxY(frame) / rowHeight) ;
 52     var ret = { start: min, length: max - min } ; 
 53     return ret ;
 54   },
 55   
 56   /** @private */
 57   layoutItemView: function(itemView, contentIndex, firstLayout) {
 58 
 59     var rowHeight = this.get('rowHeight') || 0 ;
 60     var parentView = itemView.get('parentView') ;
 61     var f = { 
 62       x: 0, 
 63       y: contentIndex*rowHeight,
 64       height: rowHeight, 
 65       width: (parentView || this).get('innerFrame').width 
 66     } ;
 67     
 68     if (firstLayout || !SC.rectsEqual(itemView.get('frame'), f)) {
 69       itemView.set('frame', f) ;      
 70       itemView.setStyle({ zIndex: contentIndex.toString() }) ;
 71     }
 72   },
 73   
 74   computeFrame: function() {
 75     var content = this.get('content') ;
 76     var rows = (content) ? content.get('length') : 0 ;
 77     var rowHeight = this.get('rowHeight') || 20 ;
 78     
 79     var parent = this.get('parentNode') ;
 80     var f = (parent) ? parent.get('innerFrame') : { width: 100, height: 100 } ;
 81 
 82     f.x = f.y = 0;
 83     f.height = Math.max(f.height, rows * rowHeight) ;
 84     return f ;
 85   },
 86   
 87   insertionPointClass: SC.View.extend({
 88     emptyElement: '<div class="list-insertion-point"><span class="anchor"></span></div>'
 89   }),
 90 
 91   showInsertionPoint: function(itemView, dropOperation) {
 92     if (!itemView) return ;
 93 
 94     // if drop on, then just add a class...
 95     if (dropOperation === SC.DROP_ON) {
 96       if (itemView !== this._dropOnInsertionPoint) {
 97         this.hideInsertionPoint() ;
 98         itemView.addClassName('drop-target') ;
 99         this._dropOnInsertionPoint = itemView ;
100       }
101       
102     } else {
103       
104       if (this._dropOnInsertionPoint) {
105         this._dropOnInsertionPoint.removeClassName('drop-target') ;
106         this._dropOnInsertionPoint = null ;
107       }
108     
109       if (!this._insertionPointView) {
110         this._insertionPointView = this.insertionPointClass.create() ;
111       } ;
112     
113       var insertionPoint = this._insertionPointView ;
114       f = { height: 0, x: 8, y: itemView.get('frame').y, width: itemView.owner.get('frame').width };
115       insertionPoint.set('frame', f) ;
116 
117       if (insertionPoint.parentNode != itemView.parentNode) {
118         itemView.parentNode.appendChild(insertionPoint) ;
119       }
120     }
121     
122   },
123   
124   hideInsertionPoint: function() {
125     var insertionPoint = this._insertionPointView ;
126     if (insertionPoint) insertionPoint.removeFromParent() ;
127 
128     if (this._dropOnInsertionPoint) {
129       this._dropOnInsertionPoint.removeClassName('drop-target') ;
130       this._dropOnInsertionPoint = null ;
131     }
132   },
133 
134   // We can do this much faster programatically using the rowHeight
135   insertionIndexForLocation: function(loc, dropOperation) {  
136     var f = this.get('innerFrame') ;
137     var sf = this.get('scrollFrame') ;
138     var rowHeight = this.get('rowHeight') || 0 ;
139 
140     // find the row and offset to work with
141     var offset = loc.y - f.y - sf.y ;
142     var retOp = SC.DROP_BEFORE ;
143     var ret = Math.floor(offset / this.get('rowHeight')) ;
144      
145     // find the percent through the row...
146     var percentage = (offset / rowHeight) - ret ;
147     
148     // if the dropOperation is SC.DROP_ON and we are in the center 60%
149     // then return the current item.
150     if (dropOperation === SC.DROP_ON) {
151       if (percentage > 0.80) ret++ ;
152       if ((percentage >= 0.20) && (percentage <= 0.80)) {
153         retOp = SC.DROP_ON;
154       }
155     } else {
156       if (percentage > 0.45) ret++ ;
157     }
158     
159     return [ret, retOp] ;
160   }
161   
162 }) ;
