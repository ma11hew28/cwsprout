  7 require('foundation/date');

 10 /** @namespace SC.Benchmark
 11 
 12   This bit of meta-programming magic can install a benchmark handler on any
 13   object method.  When a benchmark is installed, the time required to execute
 14   the method will be printed to the console log everytime the method is 
 15   called.
 16 
 17   This class can be used to implement benchmarking.  To use this object, just
 18   call start() with a key name and end() with a keyname.  The benchmark will 
 19   be logged.  If you set verbose = true, then benchmark will log everytime it 
 20   saves a bench.  Otherwise, it just keeps stats.  You can get the stats by
 21   calling report().
 22 
 23   Benchmark does not require anything other than the date.js class.  It also
 24   does not rely on SC.Object so that you can benchmark code in that area as
 25   well.
 26 
 27 */
 28 SC.Benchmark = {
 29 
 30   /**
 31     If true, then benchmarks will be logged to the console as they are 
 32     recorded.
 33   
 34     @type bool
 35   */
 36   verbose: NO,
 37   
 38   /**
 39     If false, benchmarking will be disabled.  You might want to disable this
 40     during production to maximize performance.
 41   
 42     @type bool
 43   */
 44   enabled: YES,
 45   
 46 /**
 47    This hash stores collected stats.  It contains key value pairs.  The value
 48    will be a hash with the following properties:
 49    
 50   * * *runs*: the number of times this stat has run
 51   * * *amt*: the total time consumed by this (divide by runs to get avg)
 52   * * *name*: an optional longer name you assigned to the stat key.  Set this  using name().
 53   * * *_starts*: this array is used internally. 
 54 */
 55   stats: {},
 56 
 57   /**
 58     Call this method at the start of whatever you want to collect.
 59     if topLevelOnly is passed, then recursive calls to the start will be 
 60     ignored and only the top level call will be benchmarked.
 61     
 62     @param key {String} A unique key that identifies this benchmark.  All calls to start/end with the same key will be groups together.
 63     @param topLevelOnly {Boolean} If true then recursive calls to this method with the same key will be ignored.  
 64     @param time {Integer} Only pass if you want to explicitly set the start time.  Otherwise the start time is now.
 65   */
 66   start: function(key, topLevelOnly, time) {
 67     if (!this.enabled) return ;
 68     var stat = this._statFor(key) ;
 69     
 70     if (topLevelOnly && stat._starts.length > 0) {
 71       stat._starts.push('ignore') ;
 72     } else {
 73       stat._starts.push(time || Date.now()) ;
 74     }
 75   },
 76   
 77   /**
 78     Call this method at the end of whatever you want to collect.  This will
 79     save the collected benchmark.
 80     
 81     @param key {String} The benchmark key you used when you called start()
 82     @param time {Integer} Only pass if you want to explicitly set the end time.  Otherwise start time is now.
 83   */
 84   end: function(key, time, runs) {
 85     if (!this.enabled) return ;
 86     var stat = this._statFor(key) ;
 87     var start = stat._starts.pop() ;
 88     if (!start) {
 89       console.log('SC.Benchmark "%@" ended without a matching start.  No information was saved.'.fmt(key));
 90       return ;
 91     }
 92 
 93     // top level only.
 94     if (start == 'ignore') return ; 
 95     
 96     stat.amt += (time || Date.now()) - start ;
 97     stat.runs += (runs || 1) ;
 98     
 99     if (this.verbose) this.log(key) ;
100   },
101 
102   /**
103     This is a simple way to benchmark a function.  The function will be 
104     run with the name you provide the number of times you indicate.  Only the
105     function is a required param.
106   */  
107   bench: function(func, key, reps, context) {
108     if (!key) key = "bench%@".fmt(this._benchCount++) ;
109     if (!reps) reps = 1 ;
110     var ret ;
111     
112     var runs = reps; 
113     SC.Benchmark.start(key) ;
114     while(--reps >= 0) {
115       ret = func(context);
116     }
117     SC.Benchmark.end(key, null, runs) ; 
118     
119     return ret ;
120   },
121   
122   /**  
123     This bit of metaprogramming magic install a wrapper around a method and
124     benchmark it whenever it is run.
125   */  
126   install: function(object,method, topLevelOnly) {
127     
128     // vae the original method.
129     var __func = object['b__' + method] = object[method] ;
130     
131     // replace with this helper.
132     object[method] = function() {
133       var key = '%@(%@)'.fmt(method, SC.$A(arguments).join(', ')) ;
134       SC.Benchmark.start(key, topLevelOnly) ;
135       var ret = __func.apply(this, arguments) ;
136       SC.Benchmark.end(key) ;
137       return ret ;
138     } ;
139   },
140   
141   /**
142     Restore the original method, deactivating the benchmark.
143   
144     @param {object} object the object to change
145     @param {string} method the method name as a string.
146   
147   */  
148   restore: function(object,method) {
149     object[method] = object['b__' + method] ;
150   },
151   
152   /**
153     This method will return a string containing a report of the stats
154     collected so far.  If you pass a key, only the stats for that key will
155     be returned.  Otherwise, all keys will be used.
156   */
157   report: function(key) {
158     if (key) return this._genReport(key) ;
159     var ret = [] ;
160     
161     // find the longest stat name...
162     var maxLen = 0 ;
163     for(var key in this.stats) {
164       if (!this.stats.hasOwnProperty(key)) continue ;
165       if (key.length > maxLen) maxLen = key.length;
166     }
167     
168     // now gen report...
169     var keys = [] ;
170     for(var key in this.stats) {
171       if (!this.stats.hasOwnProperty(key)) continue ;
172       keys.push(key) ;
173     }
174     keys = keys.sort() ;
175     var max = keys.length ;
176     for(var idx=0;idx<max;idx++) {
177       ret.push(this._genReport(keys[idx], maxLen)) ;
178     }
179     return ret.join("\n") ;
180   },
181 
182   /**
183     This method is just like report() except that it will log the results to
184     the console.
185   */  
186   log: function(key) {
187     console.log(this.report(key)) ;
188   },
189   
190   /**
191     This will activate profiling if you have Firebug installed.  Otherwise
192     does nothing.
193   */
194   startProfile: function(key) {
195     if (!this.enabled) return ;
196     if (console && console.profile) console.profile(key) ;
197   },
198   
199   endProfile: function(key) {
200     if (!this.enabled) return ;
201     if (console && console.profileEnd) console.profileEnd(key) ;
202   },
203   
204   // PRIVATE METHODS
205   
206   _genReport: function(key, nameLength) {
207     var stat = this._statFor(key) ;
208     var avg = (stat.runs > 0) ? (Math.floor(stat.amt * 100000 / stat.runs) / 100000) : 0 ;
209      
210     // Generate the name, adding padding spaces if needed.
211     var name = (stat.name || key)  ;
212     nameLength = (nameLength) ? nameLength : 0; 
213     if (nameLength > name.length) {
214       var toJoin = [name] ;
215       nameLength -= name.length ;
216       while(--nameLength >= 0) toJoin.push(' ') ;
217       name = toJoin.join('') ;
218     }
219     
220     return 'BENCH | %@1 | avg: %@4 msec | total: %@2 msec | reps: %@3x '.fmt(name, stat.amt, stat.runs, avg) ;  
221   },
222   
223   // @private
224   // returns a stats hash for the named key.  If the hash does not exist yet,
225   // creates it.
226   _statFor: function(key) {
227     var ret = this.stats[key] ;
228     if (!ret) ret = this.stats[key] = {
229       runs: 0, amt: 0, name: key, _starts: []      
230     };
231     return ret ;
232   },
233   
234   reset: function() { this.stats = {} ;  /*debugger;*/},
235   
236   // This is private, but it is used in some places, so we are keeping this for
237   // compatibility.
238   _bench: function(func, name) {
239     SC.Benchmark.bench(func, name, 1) ;
240   },
241   
242   _benchCount: 1
243   
244 } ;
