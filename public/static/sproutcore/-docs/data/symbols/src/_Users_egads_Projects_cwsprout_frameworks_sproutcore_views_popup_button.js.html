1 require('views/button/button') ;
  2 
  3 /**
  4   @class
  5 
  6   @extends SC.ButtonView
  7   @author Skip Baney
  8   @copyright 2006-2008, Sprout Systems, Inc. and contributors.
  9   @version 1.0
 10 */
 11 SC.PopupButtonView = SC.ButtonView.extend({
 12 
 13   /**
 14     Overriding the default SC.ButtonView#performKeyEquivalent method to pass 
 15     it onto the menu
 16 
 17     @param {string} keystring method name corresponding to the keys pressed 
 18     (i.e alt_shift_z)
 19     @param {DOMMouseEvent} evt mousedown event
 20   */
 21   performKeyEquivalent: function( keystring, evt )
 22   {
 23     if (!this.get('isEnabled')) return false;
 24     
 25     // is it our own keyEquivalent? 
 26     if (sc_super()) return true;
 27 
 28     // is it any of our menu items keyEquivalent?
 29     var menu = this.get('menu');
 30     return (!!menu && menu.performKeyEquivalent(keystring, evt));
 31   },
 32   
 33   /**
 34     Name of the menu view to use.
 35     @type {string}
 36   */
 37   menuName: null,
 38 
 39   /**
 40     PopupMenu reference. Will be lazy-loaded from the 'menuName' string.
 41     @type {SC.PopupMenu}
 42   */
 43   menu: function( key, value )
 44   {
 45     if ( value !== undefined ) 
 46     {
 47       value.set('isVisible', false);
 48       this.set('_menu', value);
 49     }
 50     if ( !this._menu )
 51     {
 52       var menu = SC.page.get(this.get('menuName'));
 53       if (menu) menu.set('isVisible', false);
 54       // calling set so that the isSelectedBinding is triggered
 55       this.set('_menu', menu);
 56     }
 57     return this._menu;
 58   }.property(),
 59   
 60   /**
 61     Binds the button's selection state to the menu's visibility.
 62     @private
 63   */
 64   isSelectedBinding: '*_menu.isVisible',
 65   
 66   /**
 67     Button action handler
 68     @param {DOMMouseEvent} evt mouseup event that triggered the action
 69   */
 70   action: function( evt )
 71   {
 72     var menu = this.get('menu');
 73     // no menu to toggle... bail...
 74     if (!menu) return false;
 75     
 76     if (!this._didFirstRun) {
 77       // for some reason the menu#isVisible is true the first time we get 
 78       // it... and since this#isSelected is bound to it... we get an incorrect 
 79       // conditional check. hacking it here to keep moving.
 80       menu.popup(this, evt);
 81       this._didFirstRun = true;
 82     } else {
 83       // toggle the menu...
 84       this.get('isSelected') ? menu.set('isVisible', false) : menu.popup(this, evt);
 85     }
 86     
 87     return true;
 88   }
 89 
 90 });
