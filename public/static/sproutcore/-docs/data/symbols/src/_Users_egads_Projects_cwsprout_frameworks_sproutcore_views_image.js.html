 11 SC.IMAGE_STATE_NONE = 'none';
 12 SC.IMAGE_STATE_LOADING = 'loading';
 13 SC.IMAGE_STATE_LOADED = 'loaded';
 14 SC.IMAGE_STATE_FAILED = 'failed';

 16 /**
 17   URL to a transparent GIF.  Used for spriting.
 18 */
 19 SC.BLANK_IMAGE_URL = static_url('blank.gif');

 21 /**
 22   @class
 23 
 24   Displays an image in the browser.  
 25   
 26   The ImageView can be used to efficiently display images in the browser.
 27   It includes a built in support for a number of features that can improve
 28   your page load time if you use a lot of images including a image loading
 29   queue and automatic support for CSS spriting.
 30 
 31   @extends SC.View
 32   @extends SC.Control
 33   @author Charles Jolley
 34 */
 35 SC.ImageView = SC.View.extend(SC.Control, 
 36 /** @scope SC.ImageView.prototype */ {
 37   
 38   /** Image views contain an img tag. */
 39   emptyElement: '<img src="%@" class="sc-image-view" />'.fmt(SC.BLANK_IMAGE_URL),
 40   
 41   /**
 42     Current load status of the image.
 43     
 44     This status changes as an image is loaded from the server.  If spriting
 45     is used, this will always be loaded.  Must be one of the following
 46     constants: SC.IMAGE_STATE_NONE, SC.IMAGE_STATE_LOADING, 
 47     SC.IMAGE_STATE_LOADED, SC.IMAGE_STATE_FAILED
 48   */
 49   status: SC.IMAGE_STATE_NONE,
 50   
 51   /**
 52     A url or CSS class name.
 53     
 54     This is the image you want the view to display.  It should be either a
 55     url or css class name.  You can also set the content and 
 56     contentValueKey properties to have this value extracted 
 57     automatically.
 58     
 59     If you want to use CSS spriting, set this value to a CSS class name.  If
 60     you need to use multiple class names to set your icon, separate them by
 61     spaces.
 62   */
 63   value: null,
 64   _value: null,
 65 
 66   /**
 67     Invoked whenever the content or value changes. (To be removed in 
 68     SproutCore 1.0)
 69     
 70     This method is no longer necessary since we have the standard SC.Control
 71     behavior.
 72     
 73     @deprecated
 74     @param {Object} content The content object.
 75     @returns {String} the URL or CSS class name
 76   */
 77   transform: function(content) { return content; },
 78   
 79   valueObserver: function() {
 80     
 81     // get the new URL.
 82     var value = this.get('value') ;
 83     
 84     // invoke the old transform method if it is defined
 85     if (this.transform !== SC.ImageView.prototype.transform) {
 86       var content = this.get('content') || '' ;
 87       value = this.transform(content) ;
 88     }
 89     
 90     // if the value has not changed, do nothing.
 91     if (value == this._value) return ;
 92     
 93     // if the old value was a class name, then we need to remove it.
 94     if (this._value && this._value.length>0 && !SC.ImageView.valueIsUrl(this._value)) {
 95       var classNames = this._value.split(' ') ;
 96       var idx = classNames.length ;
 97       while(--idx >= 0) { 
 98         this.removeClassName(classNames[idx]); 
 99       }
100       this.removeClassName('sc-sprite') ;
101     }
102     this._value = value ;
103     
104     // if the new value is empty, just clear the img.
105     if (!value || value.length == 0) {
106       this.rootElement.src = SC.BLANK_IMAGE_URL;
107       this.set('status', SC.IMAGE_STATE_NONE) ;
108       this._imageUrl = null; //clear
109       
110     // if a new value was set that is a URL, load the image URL.
111     } else if (SC.ImageView.valueIsUrl(value)) {
112       this.beginPropertyChanges() ;
113       this.set('status', SC.IMAGE_STATE_LOADING) ;
114       this._imageUrl = value ; // save to verify later.
115       SC.imageCache.loadImage(value, this, this._onLoadComplete) ;
116       this.endPropertyChanges() ;
117       
118     // if the new is a CSS class name, set an empty image and add class name 
119     } else {
120       var classNames = value.split(' ');
121       var idx = classNames.length ;
122       while(--idx >= 0) this.addClassName(classNames[idx]) ;
123       this.addClassName('sc-sprite') ;
124       this.rootElement.src = SC.BLANK_IMAGE_URL ;
125       this.set('status', SC.IMAGE_STATE_LOADED) ;
126     }
127   }.observes('value'),
128   
129 
130   /** 
131     Invoked once an image loads.  If an image has already been loaded,
132     this method will be invoked immediately.
133   */
134   _onLoadComplete: function(url, status, img) {  
135     
136     // sometimes this method gets called later after the url has already
137     // changed.  If this is the case, bail...
138     if (url !== this._imageUrl) return ;
139     
140     this.beginPropertyChanges() ;
141     this.set('imageWidth', parseInt(img.width,0)) ;
142     this.set('imageHeight', parseInt(img.height,0)) ;
143     this.set('status',status) ;
144     this.endPropertyChanges() ;
145     
146     if (status == SC.IMAGE_STATE_LOADED) {
147       if (this.imageDidLoad) this.imageDidLoad(url) ;
148       this.rootElement.src =  url ;
149     } else {
150       if (this.imageDidFail) this.imageDidFail(url, status) ;
151     }
152   },
153   
154   init: function() {
155     sc_super() ;
156     this.valueObserver() ;
157     if (this.rootElement.src) {
158       this.set('imageWidth',parseInt(this.rootElement.width,0)) ;
159       this.set('imageHeight',parseInt(this.rootElement.height,0)) ;
160     }
161   }
162   
163 }) ;
164 
165 /**
166   Returns YES if the passed value looks like an URL and not a CSS class
167   name.
168 */
169 SC.ImageView.valueIsUrl = function(value) {
170   return (value.indexOf('/') >= 0) || (value.indexOf('.') >= 0) ;
171 } ;

173 /**
174   The image cache will create Image objects to preload a set of 
175   images.  This will control the number of images being loaded to maximize
176   browser throughput.
177 */
178 SC.imageCache = SC.Object.create({  
179   
180   // this restricts the maximum number of images that can load in.
181   loadLimit: 4,
182   
183   // this is the primary entry point for the imageCache.  Use this method to
184   // ask the cache to load an image and then invoke the callback when its
185   // available.  You can pass either a function or an object + a function.
186   // Your callback should have this pattern:
187   //
188   // method(url, status, imgObject)
189   // url: the original URL you passed in.
190   // status: loaded | error | aborted
191   // imgObject: the image object.  This parameter is optional.
192   //
193   loadImage: function(url, objOrFunc, method) {
194     var dta = this._images[url] = (this._images[url] || { url: url,
195       img: null, handlers: [], status: 'unknown' }) ;
196 
197     if (dta.img == null) {
198       this._queue.push(dta) ;
199       if (!this._imgTimeout) {
200         this._imgTimeout = this.invokeLater(this.loadNextImage, 100) ;
201       }
202     }
203 
204     // you can pass either just a function or an object + a method.  This will
205     // handle both.
206     var handler = (method) ? [objOrFunc, method] : [this, objOrFunc] ;
207     if (dta.status == 'unknown') {
208       dta.handlers.push(handler) ;
209     } else if (handler[1]) {
210       handler[1].call(handler[0], url, dta.status, dta.img) ;
211     }
212   },
213   
214   // this is called to load images that need 
215   loadNextImage: function() {
216     this._imgTimeout = null ;
217     while((this._queue.length > 0) && (this._loading.length < this.loadLimit)) {   
218       var dta = this._queue.pop() ;
219       var url = dta.url ;
220       dta.img = new Image() ;
221       dta.img.onabort = this._onAbort.bind(this,url) ;
222       dta.img.onerror = this._onError.bind(this,url) ;
223       dta.img.onload = this._onLoad.bind(this, url) ;
224       dta.img.src = dta.url ;
225 
226       // add to loading queue.
227       this._loading.push(dta.url) ;
228     }
229   },
230   
231   _onAbort: function(url) { this._changeStatus(url, 'aborted') ; },
232   _onError: function(url) { this._changeStatus(url, 'error'); },
233   _onLoad: function(url) { this._changeStatus(url, 'loaded'); },
234 
235   // update the status in the queue and call any queued handlers.
236   _changeStatus: function(url, status) {
237     var dta = this._images[url] ;
238     if (!dta) return ;
239     dta.status = status ;
240     
241     var handler ;
242     while(handler = dta.handlers.pop()) {
243       if (handler[1]) handler[1].call(handler[0], url, dta.status, dta.img) ;
244     }
245     
246     // get the next image, if needed.
247     this._loading = this._loading.without(dta.url) ;
248     this.loadNextImage() ;
249   },
250   
251   _images: {},
252   _loading: [],
253   _queue: []
254     
255 });
