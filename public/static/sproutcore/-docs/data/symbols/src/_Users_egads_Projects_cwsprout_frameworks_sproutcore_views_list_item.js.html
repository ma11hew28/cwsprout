 14 /** @class

 16   Displays a list item with a label, optional icon, count, and right icon.

 26 /** @scope SC.ListItemView.prototype */ {
 27   
 28   /** A ListItemView has an img tag, label, count, optional right button, and
 29     an optional right arrow.  These items are made visible/hidden based on a
 30     variety of CSS classes.
 31   */
 32   emptyElement: '<div class="sc-list-item-view sc-collection-item"></div>',
 33   
 34   /**
 35     The content object the list item will display.
 36   */
 37   content: null,
 38   
 39   /**
 40     (displayDelegate) True if you want the item view to display an icon.
 41     
 42     If false, the icon on the list item view will be hidden.  Otherwise,
 43     space will be left for the icon next to the list item view.
 44   */
 45   hasContentIcon: NO,
 46   
 47   /**
 48     (displayDelegate) True if you want space to be allocated for a branch 
 49     arrow.
 50     
 51     If false, the space for the branch arrow will be collapsed.
 52   */
 53   hasContentBranch: NO,
 54   
 55   /**
 56     (displayDelegate) The name of the property used for the checkbox value.
 57     
 58     The checkbox will only be visible if this key is not null.
 59     
 60     @type {String}
 61   */
 62   contentCheckboxKey: null,
 63   
 64   /**
 65     (displayDelegate) Property key to use for the icon url
 66 
 67     This property will be checked on the content object to determine the 
 68     icon to display.  It must return either a URL or a CSS class name.
 69   */
 70   contentIconKey: null,
 71   
 72   /**
 73     (displayDelegate) The name of the property used for label itself
 74     
 75     If null, then the content object itself will be used..
 76   */
 77   contentValueKey: null,
 78   
 79   /**
 80     (displayDelegate) The name of the property used to find the count of 
 81     unread items. 
 82     
 83     The count will only be visible if this property is not null and the 
 84     returned value is not 0.
 85   */
 86   contentUnreadCountKey: null,
 87   
 88   /**
 89     (displayDelegate) The name of the property used to determine if the item
 90     is a branch or leaf (i.e. if the branch icon should be displayed to the
 91     right edge.)
 92     
 93     If this is null, then the branch view will be completely hidden.
 94     Otherwise space will be allocated for it.
 95   */
 96   contentIsBranchKey: null,
 97   
 98   /**
 99     YES if the item view is currently editing.
100   */
101   isEditing: NO,
102   
103   contentPropertyDidChange: function() {
104     if (this.get('isEditing')) this.discardEditing() ;
105     this.render() ;  
106   },

109   /**
110     Regenerates the innerHTML for this view and updates it if necessary.
111   */
112   render: function() {
113     var html = [] ;
114     var content = this.get('content') ;
115     var del = this.displayDelegate ;
116     
117     // handle checkbox
118     var checkboxKey = this.getDelegateProperty(del, 'contentCheckboxKey') ;
119     if (checkboxKey) {
120       var checkboxValue = (content && content.get) ? content.get(checkboxKey) : false ;
121       html.push(this.renderCheckboxHtml(checkboxValue)) ;
122     }
123 
124     // handle icon
125     if (this.getDelegateProperty(del, 'hasContentIcon')) {
126        var iconKey = this.getDelegateProperty(del,'contentIconKey') ;
127        var icon = (iconKey && content && content.get) ? content.get(iconKey) : null ;
128        html.push(this.renderIconHtml(icon));
129     }
130     
131     // handle label
132     var labelKey = this.getDelegateProperty(del, 'contentValueKey') ;
133     var label = (labelKey && content && content.get) ? content.get(labelKey) : null ;
134     html.push(this.renderLabelHtml(label));
135     
136     // handle unread count
137     var countKey = this.getDelegateProperty(del, 'contentUnreadCountKey') ;
138     var count = (countKey && content && content.get) ? content.get(countKey) : null ;
139     if ((count != null) && (count != 0)) {
140       html.push(this.renderCountHtml(count));
141     }
142     
143     // handle action 
144     var actionKey = this.getDelegateProperty(del, 'listItemActionProperty') ;
145     var actionClassName = (actionKey && content && content.get) ? content.get(actionKey) : null ;
146     if (actionClassName) {
147        html.push(this.renderActionHtml(actionClassName));
148     }
149     this.setClassName('sc-has-action', actionClassName) ;
150     
151     // handle branch
152     if (this.getDelegateProperty(del, 'hasContentBranch')) {
153       var branchKey = this.getDelegateProperty(del, 'contentIsBranchKey');
154       var hasBranch = (branchKey && content && content.get) ? content.get(branchKey) : false ;
155       html.push(this.renderBranchHtml(hasBranch));
156       this.setClassName('sc-has-branch', true) ;
157     } else this.setClassName('sc-has-branch', false) ;
158     
159     html = html.join('') ;
160     if (html != this._lastRenderedHtml) {
161       this._lastRenderedHtml = html ;
162       this.set('innerHTML', html) ;
163     }
164   },
165   
166   /**
167     Generates the HTML string used to represent the checkbox for your list
168     item.  Override this to return your own custom HTML.  The default version
169     will use the HTML provided by SC.CheckboxView.
170     
171     @returns {String}
172     @param state {String} the checkbox state.  YES, NO, or SC.MIXED_STATE
173   */
174   renderCheckboxHtml: function(state) {
175     var ret ;
176     
177     // Note: this basically takes the HTML from the checkbox view and then
178     // inserts class names as necessary.  This is cached to avoid using too
179     // much memory.
180     if (state === SC.MIXED_STATE) {
181       ret = SC.ListItemView._mixedCheckboxHtml ;
182       if (!ret) {
183         ret = SC.CheckboxView.prototype.emptyElement ;
184         ret = ret.replace('class="', 'class="mixed ') ;
185         SC.ListItemView._mixedCheckboxHtml = ret ;
186       }
187     } else if (state) {
188       ret = SC.ListItemView._selectedCheckboxHtml ;
189       if (!ret) {
190         ret = SC.CheckboxView.prototype.emptyElement ;
191         ret = ret.replace('class="', 'class="sel ') ;
192         SC.ListItemView._selectedCheckboxHtml = ret ;
193       }
194     } else {
195       ret = SC.ListItemView._normalCheckboxHtml ;
196       if (!ret) {
197         ret = SC.CheckboxView.prototype.emptyElement ;
198         SC.ListItemView._normalCheckboxHtml = ret ;
199       }
200     }
201     return ret ;
202   },
203   
204   /** 
205      renderIconHtml generates the html string used to represent the icon for 
206      your list item.  override this to return your own custom HTML
207      
208      @returns {String}
209      @param icon {String} the icon property based on your view's contentIconKey
210    */
211    renderIconHtml: function(icon){
212      var html = [];
213      // get a class name and url to include if relevant
214      var url = null, className = null ;
215      if (icon && SC.ImageView.valueIsUrl(icon)) {
216        url = icon; className = '' ;
217      } else {
218        className = icon; url = static_url('blank.gif') ;
219      }
220      html.push('<img class="sc-icon ');
221      html.push(className || '');
222      html.push('" src="');
223      html.push(url || static_url('blank.gif')) ;
224      html.push('" />') ;
225      html=html.join('');
226      return html;
227    },
228    
229    /** 
230        renderLabelHtml generates the html string used to represent the label 
231        for your list item.  override this to return your own custom HTML
232        
233        @returns {String}
234        @param label {String} the label property based on your view's 
235         contentValueKey
236      */
237    renderLabelHtml: function(label){
238      var html = [];
239      html.push('<span class="sc-label">') ;
240      html.push(label || '') ;
241      html.push('</span>') ;
242      return html.join('');    
243    },

245    /**
246       Finds and retrieves the element containing the label.  This is used
247       for inline editing.  If you override renderLabelHtml() you probably
248       need to override this as well.
249   */
250    findLabelElement: function() {
251      return this.$class('sc-label') ;
252    },
253    
254    /** 
255         renderCountHtml generates the html string used to represent the count 
256         (like unread count) for your list item.  override this to return your 
257         own custom HTML
258         
259         @returns {String}
260         @param count {Integer} the label property based on your view's 
261          contentValueKey
262     */
263    renderCountHtml: function(count) {
264      var html= [];
265       html.push('<span class="sc-count"><span class="inner">') ;
266        html.push(count.toString()) ;
267        html.push('</span></span>') ;
268        return html.join('');
269    },
270    
271    /** 
272       renderActionHtml generates the html string used to represent the 
273       action item for your list item.  override this to return your own 
274       custom HTML
275     
276       @returns {String}
277       @param actionClassName {String} the name of the action item.
278     */
279    renderActionHtml: function(actionClassName){
280      var html = [];
281      html.push('<img src="') ;
282      html.push(static_url('blank.gif')) ;
283      html.push('" class="sc-action" />') ;
284      return html.join('');
285    },
286    
287    /** 
288      renderBranchHtml generates the html string used to represent the 
289      branch arrow. override this to return your own custom HTML
290      @returns {String}
291      @arguments {Boolean} whehter the branch is 
292    */
293    
294    renderBranchHtml: function(hasBranch) {
295      var html = [];
296      html.push('<span class="sc-branch ');
297      html.push(hasBranch ? 'sc-branch-visible' : 'sc-branch-hidden') ;
298      html.push('"> </span>');
299      return html.join('');
300    },
301    
302    _isInsideElementWithClassName: function(className, evt) {
303      var el = Event.element(evt) ;
304      var rootElement = this.rootElement;
305      var ret = NO ;
306      while(!ret && el && (el !== rootElement)) {
307        if (Element.hasClassName(el, className)) ret = YES ;
308        el = el.parentNode ;
309      }
310      
311      rootElement = el = null ; //avoid memory leaks
312      return ret ;
313    },

315    /** @private 
316     mouseDown is handled only for clicks on the checkbox view or or action
317     button.
318    */
319    mouseDown: function(evt) {
320      var del = this.displayDelegate ;
321      var checkboxKey = this.getDelegateProperty(del, 'contentCheckboxKey') ;
322      if (checkboxKey) {
323        if (this._isInsideElementWithClassName('sc-checkbox-view', evt)) {
324          this._addCheckboxActiveState() ;
325          this._isMouseDownOnCheckbox = YES ;
326          this._isMouseInsideCheckbox = YES ;
327          return true ;
328        }
329      }  
330      
331      return false ; // otherwise let normal handlers do it...
332    },
333 
334    mouseUp: function(evt) {
335      var ret= NO ;
336      if (this._isMouseDownOnCheckbox) {
337        
338        // update only if mouse inside on mouse up...
339        if (this._isMouseInsideCheckbox) {
340          var del = this.displayDelegate ;
341          var checkboxKey = this.getDelegateProperty(del, 'contentCheckboxKey') ;
342          var content = this.get('content') ;
343          if (content && content.get) {
344            var value = content.get(checkboxKey) ;
345            value = (value === SC.MIXED_STATE) ? YES : !value ;
346            content.set(checkboxKey, value) ;
347          }
348        }
349 
350        this._removeCheckboxActiveState() ;
351        ret = YES ;
352      } 
353 
354      this._isMouseInsideCheckbox = this._isMouseDownOnCheckbox = NO ;
355      return ret ;
356    },
357    
358    mouseOut: function(evt) {
359      if (this._isMouseDownOnCheckbox) {
360        this._removeCheckboxActiveState() ;
361        this._isMouseInsideCheckbox = NO ;
362      }  
363      return NO ;
364    },
365    
366    mouseOver: function(evt) {
367      if (this._isMouseDownOnCheckbox) {
368        this._addCheckboxActiveState() ;
369        this._isMouseInsideCheckbox = YES ;
370      }  
371      return NO ;
372    },
373    
374    _addCheckboxActiveState: function() {
375      var el = this.$sel('.sc-checkbox-view') ;
376      if (this.get('isEnabled')) Element.addClassName(el, 'active') ;
377      el = null ;
378    },
379    
380    _removeCheckboxActiveState: function() {
381      var el = this.$sel('.sc-checkbox-view') ;
382      Element.removeClassName(el, 'active') ;
383      el = null ;
384    },

386    /** 
387     Returns true if a click is on the label text itself to enable editing.
388     
389     Note that if you override renderLabelHtml(), you probably need to override 
390     this as well.
391   
392     @param evt {Event} the mouseUp event.
393     @returns {Boolean} YES if the mouse was on the content element itself.
394    */
395    contentHitTest: function(evt) {
396      
397      // if not content value is returned, not much to do.
398      var del = this.displayDelegate ;
399      var labelKey = this.getDelegateProperty(del, 'contentValueKey') ;
400      if (!labelKey) return NO ;
401      
402      // get the element to check for.
403      var el = this.findLabelElement() ;
404      if (!el) return NO ; // no label to check for.
405      
406      var cur = Event.element(evt) ;
407      while(cur && (cur != (this.rootElement)) && (cur != window)) {
408        if (cur === el) return YES ;
409        cur = cur.parentNode ;
410      }
411      
412      return NO;
413    },
414    
415    beginEditing: function() {
416      
417      if (this.get('isEditing')) return YES ;
418      
419      var content = this.get('content') ;
420      var del = this.displayDelegate ;
421      var labelKey = this.getDelegateProperty(del, 'contentValueKey') ;
422      var v = (labelKey && content && content.get) ? content.get(labelKey) : null ;
423      
424      var f = this.get('frame') ;
425      var el = this.findLabelElement() ;
426      if (!el) return NO ;
427 
428      // if the label has a large line height, try to adjust it to something
429      // more reasonable so that it looks right when we show the popup editor.
430      var oldLineHeight = Element.getStyle(el, 'lineHeight') ;
431      var fontSize = parseInt(Element.getStyle(el, 'fontSize'), 0) ;
432      var lineHeight = parseInt(oldLineHeight, 0) ;
433      var lineHeightShift = 0;
434      
435      if (fontSize && lineHeight) {
436        var targetLineHeight = fontSize * 1.5 ;
437        if (targetLineHeight < lineHeight) {
438          Element.setStyle(el, { lineHeight: '1.5' }) ;
439          lineHeightShift = (lineHeight - targetLineHeight) / 2; 
440        } else oldLineHeight = null ;
441      }
442      
443      f.x += el.offsetLeft ;
444      f.y += el.offsetTop + lineHeightShift - 2;
445      f.height = el.offsetHeight ;
446      f.width = (f.width - 30 - el.offsetLeft) ;
447      f = this.convertFrameToView(f, null) ;
448      
449      var ret = SC.InlineTextFieldView.beginEditing({
450        frame: f, 
451        exampleElement: el, 
452        delegate: this, 
453        value: v
454      }) ;
455 
456      // restore old line height for original item if the old line height 
457      // was saved.
458      if (oldLineHeight) Element.setStyle(el, { lineHeight: oldLineHeight }) ;
459      
460      // Done!  If this failed, then set editing back to no.
461      return ret ;
462    },
463    
464    commitEditing: function() {
465      if (!this.get('isEditing')) return YES ;
466      return SC.InlineTextFieldView.commitEditing();
467    },
468    
469    discardEditing: function() {
470      if (!this.get('isEditing')) return YES ;
471      return SC.InlineTextFieldView.discardEditing();
472    },

475      Set editing to true so edits will no longer be allowed.
476    */
477    inlineEditorWillBeginEditing: function(inlineEditor) {
478      this.set('isEditing', YES);
479    },
480 
481    /** @private 
482      Hide the label view while the inline editor covers it.
483    */
484    inlineEditorDidBeginEditing: function(inlineEditor) {
485      var el = this.findLabelElement() ;
486      this._oldOpacity = Element.getStyle(el, 'opacity') ;
487      Element.setStyle(el, { opacity: 0.0 }) ;
488    },
489 
490    /** @private
491      Could check with a validator someday...
492    */
493    inlineEditorShouldEndEditing: function(inlineEditor, finalValue) {
494      return YES ;
495    },
496 
497    /** @private
498      Update the field value and make it visible again.
499    */
500    inlineEditorDidEndEditing: function(inlineEditor, finalValue) {
501      this.set('isEditing', NO) ;
502 
503      var content = this.get('content') ;
504      var del = this.displayDelegate ;
505      var labelKey = this.getDelegateProperty(del, 'contentValueKey') ;
506      if (labelKey && content && content.set) {
507        content.set(labelKey, finalValue) ;
508      }
509      
510      // force a refresh, otherwise the label will never be visible again
511      // b/c its opacity is 0.
512      this._lastRenderedHtml = null;
513      this.render() ;
514    }
